019package org.wattdepot.common.util;
020
021import java.io.File;
022
023/**
024 * UserHome - Provides a utility that returns the desired parent directory where
025 * the ".wattdepot3" directory will be created for storage of WattDepot
026 * preferences, logs, database, etc. This defaults to the user.home System
027 * Property, but can be overridden by the user by providing the property
028 * wattdepot3.user.home.
029 * 
030 * @author Cam Moore
031 * 
032 */
033public final class UserHome {
034
035  /** Name of property used to store the preferred location for user home directory. */
036  public static final String USER_HOME_PROPERTY = "wattdepot.user.home";
037
038  /**
039   * Hide the default constructor.
040   */
041  private UserHome() {
042    
043  }
044
045  /**
046   * Return a File instance representing the desired location of the .wattdepot3 directory. Note that
047   * this directory may or may not exist.
048   * 
049   * @return A File instance representing the desired user.home directory.
050   */
051  public static File getHome() {
052    return new File(getHomeString());
053  }
054
055  /**
056   * Return a String representing the desired location of the .wattdepot3 directory. Note that this
057   * directory may or may not exist.
058   * 
059   * @return A String instance representing the desired user.home directory.
060   */
061  public static String getHomeString() {
062    return System.getProperty(USER_HOME_PROPERTY, System.getProperty("user.home"));
063  }  
064}
