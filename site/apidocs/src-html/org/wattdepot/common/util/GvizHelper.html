047/**
048 * GvizHelper - Utility class that handles Google Visualization using the Google Visualization
049 * Datasource library. 
050 *
051 * * @see <a
052 * href="http://code.google.com/apis/chart/interactive/docs/dev/implementing_data_source.html">Google
053 * Visualization Datasource API</a>
054 *
055 * @author Yongwen Xu
056 *
057 */
058public class GvizHelper {
059  /** Conversion factor for milliseconds per minute. */
060  private static final long MILLISECONDS_PER_MINUTE = 60L * 1000;
061
062  /**
063   * @param server
064   *          the restlet ServerResource
065   * @param type
066   *          type of the gviz query string, such as tqx, or tq
067   * @return the value of the gviz query string from the request
068   */
069  public static String getGvizQueryString(ServerResource server, String type) {
070    String qs = server.getRequest().getResourceRef().getQueryAsForm().getFirstValue(type);
071    try {
072      if (qs != null) {
073        qs = URLDecoder.decode(qs, "UTF-8");
074      }
075    }
076    catch (UnsupportedEncodingException e) {
077      qs = null;
078    }
079    return qs;
080  }
081
082  /**
083   * @param resource
084   *          server resource object
085   * @param tqxString
086   *          gviz tqx query string, i.e., request id
087   * @param tqString
088   *          gviz tq query string, selectable fields
089   * @return gviz response 
090   */
091  public static String getGvizResponse(Object resource, String tqxString, String tqString) {
092    DataTable table = null;
093    try {
094      if (resource instanceof InterpolatedValue) {
095        table = getDataTable((InterpolatedValue) resource);
096      }
097      if (resource instanceof MeasurementList) {
098        table = getDataTable((MeasurementList) resource);
099      }
100      if (resource instanceof InterpolatedValueList) {
101        table = getDataTable((InterpolatedValueList) resource);
102      }
103      if (resource instanceof XYInterpolatedValueList) {
104        table = getDataTable((XYInterpolatedValueList) resource);
105      }
106    }
107    catch (DataSourceException e) {
108      return getGvizDataErrorResponse(e);
109    }
110
111    return getGvizResponseFromDataTable(table, tqxString/*, tqString*/);
112  }

114  /**
115   * @param mValue
116   *          mesured value
117   * @return gviz DataTable
118   * @throws DataSourceException
119   *          data source exception
120   */
121  private static DataTable getDataTable(InterpolatedValue mValue) throws DataSourceException {
122
123    MeasurementType mType = mValue.getMeasurementType();
124
125    DataTable data = new DataTable();
126
127    try {
128      data.addColumn(
129          new ColumnDescription("Date", ValueType.DATETIME, "Date & Time"));
130      data.addColumn(
131          new ColumnDescription(mType.getName(), ValueType.NUMBER, mType.toString()));
132
133      TableRow row = new TableRow();
134
135      XMLGregorianCalendar xgcal = DateConvert.convertDate(mValue.getStart());
136      row.addCell(new DateTimeValue(convertTimestamp(xgcal)));
137      row.addCell(mValue.getValue());
138
139      data.addRow(row);
140    }
141    catch (TypeMismatchException | DatatypeConfigurationException e) {
142      throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Problem adding data to table"); // NOPMD
143    }
144    return data;
145  }
146
147  /**
148   * @param mList
149   *          measurement list
150   * @return gviz data table
151   * @throws DataSourceException
152   *          data source exception
153   */
154  private static DataTable getDataTable(MeasurementList mList) throws DataSourceException {
155    DataTable data = new DataTable();
156
157    // Sets up the columns requested by any SELECT in the datasource query
158    ArrayList<Measurement> measurements = mList.getMeasurements();
159    if (!measurements.isEmpty()) {
160      String type = measurements.get(0).getMeasurementType();
161      try {
162        data.addColumn(
163            new ColumnDescription("Timestamp", ValueType.DATETIME, "Date & Time"));
164        data.addColumn(
165            new ColumnDescription(type, ValueType.NUMBER, type));
166
167        for (Measurement measurement : measurements) {
168          TableRow row = new TableRow();
169          XMLGregorianCalendar xgcal = DateConvert.convertDate(measurement.getDate());
170          row.addCell(new DateTimeValue(convertTimestamp(xgcal)));
171          row.addCell(measurement.getValue());
172          data.addRow(row);
173        }
174      }
175      catch (NumberFormatException e) {
176        // String value in database couldn't be converted to a number.
177        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Found bad number in database"); // NOPMD
178      }
179      catch (TypeMismatchException | DatatypeConfigurationException e) {
180        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Problem adding data to table"); // NOPMD
181      }
182    }
183    return data;
184  }

186  /**
187   * @param mList
188   *          measured value list
189   * @return gviz data table
190   * @throws DataSourceException
191   *          data source exception
192   */
193  private static DataTable getDataTable(InterpolatedValueList mList) throws DataSourceException {
194    DataTable data = new DataTable();
195
196    // Sets up the columns requested by any SELECT in the datasource query
197    ArrayList<InterpolatedValue> values = mList.getInterpolatedValues();
198    if (!values.isEmpty()) {
199      String type = values.get(0).getMeasurementType().getUnits();
200      try {
201        data.addColumn(
202            new ColumnDescription("Timestamp", ValueType.DATETIME, "Date & Time"));
203        data.addColumn(
204            new ColumnDescription(type, ValueType.NUMBER, type));
205
206        for (InterpolatedValue value : values) {
207          TableRow row = new TableRow();
208          XMLGregorianCalendar xgcal = DateConvert.convertDate(value.getStart());
209          row.addCell(new DateTimeValue(convertTimestamp(xgcal)));
210          if (value.getValue() != null) {
211            row.addCell(value.getValue());
212          }
213          data.addRow(row);
214        }
215      }
216      catch (NumberFormatException e) {
217        // String value in database couldn't be converted to a number.
218        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Found bad number in database"); // NOPMD
219      }
220      catch (TypeMismatchException | DatatypeConfigurationException e) {
221        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Problem adding data to table"); // NOPMD
222      }
223    }
224    return data;
225  }

227  /**
228   * @param list The XYInterpolatedValueList.
229   * @return The gviz data table.
230   * @throws DataSourceException if there is a data source problem.
231   */
232  protected static DataTable getDataTable(XYInterpolatedValueList list) throws DataSourceException {
233    DataTable data = new DataTable();
234
235    ArrayList<XYInterpolatedValue> values = list.getValues();
236    if (!values.isEmpty()) {
237      String xType = values.get(0).getXMeasurementType().getUnits();
238      String yType = values.get(0).getYMeasurementType().getUnits();
239      try {
240        data.addColumn(new ColumnDescription(xType, ValueType.NUMBER, xType));
241        data.addColumn(new ColumnDescription(yType, ValueType.NUMBER, yType));
242
243        for (XYInterpolatedValue value : values) {
244          TableRow row = new TableRow();
245          if (value.getXValue() != null && value.getYValue() != null) {
246            row.addCell(value.getXValue());
247            row.addCell(value.getYValue());
248          }
249          data.addRow(row);
250        }
251
252      }
253      catch (NumberFormatException e) {
254        // String value in database couldn't be converted to a number.
255        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Found bad number in database"); // NOPMD
256      }
257      catch (TypeMismatchException e) {
258        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Problem adding data to table"); // NOPMD
259      }
260    }
261    return data;
262  }

264  /**
265   * @param e
266   *          DataSourceException
267   * @return the error message
268   */
269  protected static String getGvizDataErrorResponse(DataSourceException e) {
270    return "({status:'error',errors:[{reason:'"
271        + e.getReasonType().toString() + "',message:'" + e.getMessageToUser() + "'}]});";
272  }

274  /**
275   * @param table
276   *          gviz data table
277   * @param tqxString
278   *          gviz query string
279   * @return the gviz json response
280   */
281  protected static String getGvizResponseFromDataTable(DataTable table, String tqxString/*, String tqString*/) {
282    String response = "google.visualization.Query.setResponse";
283
284    String reqId = null;
285    if (tqxString != null) {
286      String[] tqxArray = tqxString.split(";");
287      for (String s : tqxArray) {
288        if (s.contains("reqId")) {
289          reqId = s.substring(s.indexOf(":") + 1, s.length());
290        }
291      }
292    }
293
294    String tableString = JsonRenderer.renderDataTable(table, true, true, true).toString();
295
296    response += "({status:'ok',";
297
298    if (reqId != null) {
299      response += "reqId:'" + reqId + "',";
300    }
301    response += "table:" + tableString + "});";
302
303    return response;
304  }

306  /**
307   * Takes an XMLGregorianCalendar (the native WattDepot timestamp format) and returns a 
308   * com.ibm.icu.util.GregorianCalendar (<b>note:</b>
309   * this is <b>not</b> the same as a java.util.GregorianCalendar!) with the time zone set to GMT,
310   * but with the timestamp adjusted by the provided offset.
311   *
312   * For example, if the input value is 10 AM Hawaii Standard Time (HST is -10 hours from UTC), and
313   * the offset is -600 minutes then the output will be 10 AM UTC, effectively subtracting 10 hours
314   * from the timestamp (in addition to the conversion between types).
315   *
316   * This rigamarole is needed because the Google Visualization data source library only accepts
317   * timestamps in UTC, and the Annonated Timeline visualization displays the UTC values. Without
318   * this conversion, any graphs of meter data recorded in the Hawaii time zone (for instance) would
319   * display the data 10 hours later in the graph, which is not acceptable. The timezoneOffset is
320   * provided if in the future we want the capability to display data normalized to an arbitrary
321   * time zone, rather than just the one the data was collected in.
322   *
323   * @param timestamp the input XMLGregorianCalendar timestamp
324   * @return a com.ibm.icu.util.GregorianCalendar suitable for use by the Google visualization data
325   * source library, and normalized by timeZoneOffset but with timeZone field set to GMT.
326   */
327  private static com.ibm.icu.util.GregorianCalendar convertTimestamp(XMLGregorianCalendar timestamp) {
328    // Calendar conversion hell. GViz library uses com.ibm.icu.util.GregorianCalendar, which is
329    // different from java.util.GregorianCalendar. So we go from our native format
330    // XMLGregorianCalendar -> GregorianCalendar, extract milliseconds since epoch, feed that
331    // to the IBM GregorianCalendar.
332
333    int timeZoneOffset = timestamp.getTimezone();
334
335    com.ibm.icu.util.GregorianCalendar gvizCal = new com.ibm.icu.util.GregorianCalendar();
336
337    // Added bonus, DateTimeValue only accepts GregorianCalendars if the timezone is GMT.
338    gvizCal.setTimeZone(com.ibm.icu.util.TimeZone.getTimeZone("GMT"));
339
340    java.util.GregorianCalendar standardCal = timestamp.toGregorianCalendar();
341
342    // Add the timeZoneOffset to the epoch time after converting to milliseconds
343    gvizCal.setTimeInMillis(standardCal.getTimeInMillis()
344        + timeZoneOffset * MILLISECONDS_PER_MINUTE);
345
346    return gvizCal;
347  }
