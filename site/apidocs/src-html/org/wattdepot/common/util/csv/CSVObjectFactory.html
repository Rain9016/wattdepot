048  /**
049   * @param cpd the CollectorProcessDefinition.
050   * @return The CSV entity for the given CollectorProcessDefinition.
051   */
052  public static String toCSV(CollectorProcessDefinition cpd) {
053    StringBuffer buf = new StringBuffer();
054    // class name
055    buf.append(cpd.getClass().getSimpleName());
056    buf.append(",");
057    // name
058    buf.append(cpd.getName());
059    buf.append(",");
060    // sensor id
061    buf.append(cpd.getSensorId());
062    buf.append(",");
063    // polling
064    buf.append(cpd.getPollingInterval());
065    buf.append(",");
066    // depository id
067    buf.append(cpd.getDepositoryId());
068    buf.append(",");
069    buf.append(cpd.getOrganizationId());
070    buf.append(",");
071    buf.append(getPropertiesCSV(cpd.getProperties()));
072    return buf.toString();
073  }

075  /**
076   * @param csv the CSV to parse.
077   * @return the CollectorProcessDefinition
078   * @throws IOException if there is a problem parsing the csv.
079   */
080  public static CollectorProcessDefinition buildCPD(String csv) throws IOException {
081    CSVReader reader = new CSVReader(new StringReader(csv));
082    try {
083      String[] line = reader.readNext();
084      if (CollectorProcessDefinition.class.getSimpleName().equals(line[0]) && line.length >= 7) {
085        String name = line[1];
086        String sensorId = line[2];
087        Long polling = Long.parseLong(line[3]);
088        String depositoryId = line[4];
089        String orgId = line[5];
090        Set<Property> properties = buildProperties(line, 6);
091        return new CollectorProcessDefinition(Slug.slugify(name), name, sensorId, polling,
092            depositoryId, properties, orgId);
093      }
094    }
095    finally {
096      reader.close();
097    }
098    return null;
099  }

124  /**
125   * @param csv The CSV entity to parse.
126   * @return The Depository
127   * @throws IOException if there is a problem parsing the String.
128   */
129  public static Depository buildDepository(String csv) throws IOException {
130    CSVReader reader = new CSVReader(new StringReader(csv));
131    try {
132      String[] line = reader.readNext();
133      if (Depository.class.getSimpleName().equals(line[0]) && line.length == 5) {
134        String name = line[1];
135        String measTypeName = line[2];
136        String measTypeUnits = line[3];
137        MeasurementType type = new MeasurementType(measTypeName, Unit.valueOf(measTypeUnits));
138        String orgId = line[4];
139        return new Depository(Slug.slugify(name), name, type, orgId);
140      }
141    }
142    finally {
143      reader.close();
144    }
145    return null;
146  }

198  /**
199   * @param group The SensorGroup to convert.
200   * @return The CSV for the group.
201   */
202  public static String toCSV(SensorGroup group) {
203    StringBuffer buf = new StringBuffer();
204    // class name
205    buf.append(group.getClass().getSimpleName());
206    buf.append(",");
207    // name
208    buf.append(group.getName());
209    buf.append(",");
210    // orgId
211    buf.append(group.getOrganizationId());
212    buf.append(",");
213    buf.append(group.getSensors().size());
214    buf.append(",");
215    for (String sensor : group.getSensors()) {
216      buf.append(sensor);
217      buf.append(",");
218    }
219    return buf.toString();
220  }

222  /**
223   * @param csv The CSV representation of the SensorGroup
224   * @return The SensorGroup
225   * @throws IOException if there is a problem parsing the String.
226   */
227  public static SensorGroup buildSensorGroup(String csv) throws IOException {
228    CSVReader reader = new CSVReader(new StringReader(csv));
229    try {
230      String[] line = reader.readNext();
231      if (line.length >= 4 && SensorGroup.class.getSimpleName().equals(line[0])) {
232        String name = line[1];
233        String orgId = line[2];
234        Integer numSensors = Integer.parseInt(line[3]);
235        Set<String> sensors = new HashSet<String>();
236        for (int i = 0; i < numSensors; i++) {
237          sensors.add(line[4 + i]);
238        }
239        return new SensorGroup(Slug.slugify(name), name, sensors, orgId);
240      }
241    }
242    finally {
243      reader.close();
244    }
245    return null;
246  }

281  /**
282   * @param csv The CSV representation of the SensorGroup
283   * @return The SensorGroup
284   * @throws IOException if there is a problem parsing the String.
285   */
286  public static MeasurementPruningDefinition buildMeasurementPruningDefinition(String csv)
287      throws IOException {
288    CSVReader reader = new CSVReader(new StringReader(csv));
289    try {
290      String[] line = reader.readNext();
291      if (line.length == 8 && MeasurementPruningDefinition.class.getSimpleName().equals(line[0])) {
292        String name = line[1];
293        String depositoryId = line[2];
294        String sensorId = line[3];
295        String orgId = line[4];
296        Integer ignore = Integer.parseInt(line[5]);
297        Integer collect = Integer.parseInt(line[6]);
298        Integer gap = Integer.parseInt(line[7]);
299        return new MeasurementPruningDefinition(name, depositoryId, sensorId, orgId, ignore,
300            collect, gap);
301      }
302    }
303    finally {
304      reader.close();
305    }
306    return null;
307
308  }

310  /**
311   * @param properties Set of properties.
312   * @return The properties as a CSV entry with trailing ,.
313   */
314  private static String getPropertiesCSV(Set<Property> properties) {
315    StringBuffer buf = new StringBuffer();
316    buf.append(properties.size());
317    buf.append(",");
318    for (Property p : properties) {
319      buf.append(p.getKey());
320      buf.append(",");
321      buf.append(p.getValue());
322      buf.append(",");
323    }
324    return buf.toString();
325  }
326
327  /**
328   * @param values The String [] holding the properties.
329   * @param index The index to where the properties are stored.
330   * @return a Set of Properties.
331   */
332  private static Set<Property> buildProperties(String[] values, int index) {
333    Set<Property> properties = new HashSet<Property>();
334    int numProps = Integer.parseInt(values[index]);
335    for (int i = 0; i < numProps; i++) {
336      Property p = new Property();
337      p.setKey(values[index + i * 2 + 1]);
338      p.setValue(values[index + i * 2 + 2]);
339      properties.add(p);
340    }
341    return properties;
342  }
