001package org.wattdepot.common.http.api;
002
003import org.restlet.resource.Get;
004import org.wattdepot.common.domainmodel.InterpolatedValueList;
005
006/**
007 * DepositoryLatestValuesResource - HTTP Interface for getting an
008 * InterpolatedValueList of the latest values for a sensor or sensor group. <br>
009 * (/wattdepot/{org-id}/depository/{depository-id}/values/latest/)
010 *
011 * @author Cam Moore
012 *
013 */
014@SuppressWarnings("PMD.UnusedModifier")
015public interface DepositoryLatestValuesResource {
016  /**
017   * Defines GET <br/>
018   * /wattdepot/{org-id}/depository/{depository-id}/values/latest/?sensor={sensor_id}.
019   *
020   * @return The InterpolatedValueList.
021   */
022  @Get("json")
023  // Use JSON as transport encoding.
024  public InterpolatedValueList retrieve();
025}
