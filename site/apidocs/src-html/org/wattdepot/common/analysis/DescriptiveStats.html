019package org.wattdepot.common.analysis;
020
021import java.util.ArrayList;
022
023import org.apache.commons.math3.stat.descriptive.DescriptiveStatistics;
024import org.wattdepot.common.domainmodel.InterpolatedValue;
025import org.wattdepot.common.domainmodel.Labels;
026import org.wattdepot.common.domainmodel.InterpolatedValueList;
027import org.wattdepot.common.domainmodel.Measurement;
028import org.wattdepot.common.domainmodel.MeasurementList;
029
030/**
031 * DescriptiveStats - Simple way of getting descriptive statistics for different
032 * WattDepot measurement and interpolated values lists.
033 * 
034 * @author Cam Moore
035 * 
036 */
037public class DescriptiveStats {
038
039  private DescriptiveStatistics data;
040
041  /**
042   * @param meas MeasurementList.
043   */
044  public DescriptiveStats(MeasurementList meas) {
045    this.data = new DescriptiveStatistics();
046    for (Measurement m : meas.getMeasurements()) {
047      data.addValue(m.getValue());
048    }
049  }
050  
051  /**
052   * @param meas the Measurements.
053   * @param valueType The type of value to use either Labels.POINT or Labels.INTERVAL.
054   */
055  public DescriptiveStats(MeasurementList meas, String valueType) {
056    this.data = new DescriptiveStatistics();
057    if (Labels.POINT.equals(valueType)) {
058      for (Measurement m : meas.getMeasurements()) {
059        data.addValue(m.getValue());
060      }
061    }
062    else if (Labels.INTERVAL.equals(valueType)) {
063      ArrayList<Measurement> measurements = meas.getMeasurements();
064      for (int i = 1; i < measurements.size(); i++) {
065        Measurement start = measurements.get(i - 1);
066        Measurement end = measurements.get(i);
067        data.addValue(end.getValue() - start.getValue());
068      }
069    }
070  }
071  
072  /**
073   * @param meas MeasuredValueList.
074   */
075  public DescriptiveStats(InterpolatedValueList meas) {
076    this.data = new DescriptiveStatistics();
077    for (InterpolatedValue v : meas.getInterpolatedValues()) {
078      if (v.getValue() != null) {
079        data.addValue(v.getValue());
080      }
081    }
082  }
083
084  /**
085   * @param v the value to add.
086   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#addValue(double)
087   */
088  public void addValue(double v) {
089    data.addValue(v);
090  }
091
092  /**
093   * 
094   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#clear()
095   */
096  public void clear() {
097    data.clear();
098  }
099
100  /**
101   * @param index the index.
102   * @return get the value at the index.
103   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getElement(int)
104   */
105  public double getElement(int index) {
106    return data.getElement(index);
107  }
108
109  /**
110   * @return the geometric mean of the values.
111   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getGeometricMean()
112   */
113  public double getGeometricMean() {
114    return data.getGeometricMean();
115  }
116
117  /**
118   * @return the Kurtosis of the values.
119   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getKurtosis()
120   */
121  public double getKurtosis() {
122    return data.getKurtosis();
123  }
124
125  /**
126   * @return the maximum value.
127   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getMax()
128   */
129  public double getMax() {
130    return data.getMax();
131  }
132
133  /**
134   * @return the mean value.
135   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getMean()
136   */
137  public double getMean() {
138    return data.getMean();
139  }
140
141  /**
142   * @return the minimum value.
143   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getMin()
144   */
145  public double getMin() {
146    return data.getMin();
147  }
148
149  /**
150   * @return the number of values.
151   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getN()
152   */
153  public long getN() {
154    return data.getN();
155  }
156
157  /**
158   * Returns an estimate for the pth percentile of the values.
159   * @param p the requested percentile scaled from 0 to 100.
160   * @return An estimate for the pth percentile.
161   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getPercentile(double)
162   */
163  public double getPercentile(double p) {
164    return data.getPercentile(p);
165  }
166
167  /**
168   * @return the population variance of the values.
169   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getPopulationVariance()
170   */
171  public double getPopulationVariance() {
172    return data.getPopulationVariance();
173  }
174
175  /**
176   * @return the skewness of the values.
177   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getSkewness()
178   */
179  public double getSkewness() {
180    return data.getSkewness();
181  }
182
183  /**
184   * @return the values in sorted order.
185   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getSortedValues()
186   */
187  public double[] getSortedValues() {
188    return data.getSortedValues();
189  }
190
191  /**
192   * @return the standard deviation of the values.
193   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getStandardDeviation()
194   */
195  public double getStandardDeviation() {
196    return data.getStandardDeviation();
197  }
198
199  /**
200   * @return the sum of the values.
201   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getSum()
202   */
203  public double getSum() {
204    return data.getSum();
205  }
206
207  /**
208   * @return the sum of the square of the values
209   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getSumsq()
210   */
211  public double getSumsq() {
212    return data.getSumsq();
213  }
214
215  /**
216   * @return an array of the values.
217   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getValues()
218   */
219  public double[] getValues() {
220    return data.getValues();
221  }
222
223  /**
224   * @return the variance of the values.
225   * @see org.apache.commons.math3.stat.descriptive.DescriptiveStatistics#getVariance()
226   */
227  public double getVariance() {
228    return data.getVariance();
229  }
230  
231  
232}
