033/**
034 * Measurement - represents a measurement from a sensor.
035 * 
036 * @author Cam Moore
037 * 
038 */
039public class Measurement {
040  private String id;
041  private String sensorId;
042  private Date timestamp;
043  private Double value;
044  private Amount<?> amount;
045  private Unit<?> units;
046
047  /**
048   * Hide the default constructor.
049   */
050  public Measurement() {
051
052  }
053
054  /**
055   * @param sensorId The id of the sensor that made the measurement.
056   * @param timestamp The time of the measurement.
057   * @param value The value measured.
058   * @param units The type of the measurement.
059   */
060  public Measurement(String sensorId, Date timestamp, Double value, Unit<?> units) {
061    // Can a sensor create two measurements at the same time with the same type?
062    this.id = Slug.slugify(sensorId + timestamp.getTime() + units);
063    this.sensorId = sensorId;
064    this.timestamp = new Date(timestamp.getTime());
065    this.amount = Amount.valueOf(value, units);
066    this.value = amount.getEstimatedValue();
067    this.units = units;
068  }
069
070  /*
071   * (non-Javadoc)
072   * 
073   * @see java.lang.Object#equals(java.lang.Object)
074   */
075  @Override
076  public boolean equals(Object obj) {
077    if (this == obj) {
078      return true;
079    }
080    if (obj == null) {
081      return false;
082    }
083    if (!getClass().isAssignableFrom(obj.getClass())
084        && !obj.getClass().isAssignableFrom(getClass()) && getClass() != obj.getClass()) {
085      return false;
086    }
087    if (!(obj instanceof Measurement)) {
088      return false;
089    }
090    Measurement other = (Measurement) obj;
091    if (units == null) {
092      if (other.units != null) {
093        return false;
094      }
095    }
096    else if (!units.equals(other.units)) {
097      return false;
098    }
099    if (sensorId == null) {
100      if (other.sensorId != null) {
101        return false;
102      }
103    }
104    else if (!sensorId.equals(other.sensorId)) {
105      return false;
106    }
107    if (timestamp == null) {
108      if (other.timestamp != null) {
109        return false;
110      }
111    }
112    else if (!timestamp.equals(other.timestamp)) {
113      return false;
114    }
115    if (value == null) {
116      if (other.value != null) {
117        return false;
118      }
119    }
120    else if (!value.equals(other.value)) {
121      return false;
122    }
123    return true;
124  }
125
126  /**
127   * @param value a MeasuredValue
128   * @return true if this Measurement has the same sensorId, time,
129   *         MeasurementType, and value as the MeasuredValue.
130   */
131  public boolean equivalent(InterpolatedValue value) {
132    if (sensorId == null) {
133      if (value.getSensorId() != null) {
134        return false;
135      }
136    }
137    else if (!sensorId.equals(value.getSensorId())) {
138      return false;
139    }
140    if (timestamp == null) {
141      if (value.getStart() != null) {
142        return false;
143      }
144    }
145    else if (!timestamp.equals(value.getStart())) {
146      return false;
147    }
148    if (this.value == null) {
149      if (value.getValue() != null) {
150        return false;
151      }
152    }
153    else if (value.getValue() == null || Math.abs(this.value - value.getValue()) > 0.0001) {
154      return false;
155    }
156    if (getMeasurementType() == null) {
157      if (value.getMeasurementType() != null) {
158        return false;
159      }
160    }
161    else if (value.getMeasurementType() == null || !getMeasurementType().equals(value.getMeasurementType().getUnits())) {
162      return false;
163    }
164
165    return true;
166  }

168  /**

172    return new Date(timestamp.getTime());
173  }
174
175  /**
176   * @return the measurementType

178  public String getMeasurementType() {
179    if (amount != null) {

182    return null;
183  }

188  public String getSensorId() {

194   */

196    return id;
197  }
198

200   * @return the value
201   */
202  public Double getValue() {
203    return value;
204  }
205
206  /*
207   * (non-Javadoc)
208   * 
209   * @see java.lang.Object#hashCode()
210   */
211  @Override
212  public int hashCode() {
213    final int prime = 31;
214    int result = 1;
215    result = prime * result + ((units == null) ? 0 : units.hashCode());
216    result = prime * result + ((sensorId == null) ? 0 : sensorId.hashCode());
217    result = prime * result + ((timestamp == null) ? 0 : timestamp.hashCode());
218    result = prime * result + ((value == null) ? 0 : value.hashCode());
219    return result;
220  }
221
222  /**
223   * @param timestamp the timestamp to set
224   */
225  public void setDate(Date timestamp) {
226    this.timestamp = new Date(timestamp.getTime());
227  }
228
229  /**
230   * The String measurementType must be a javax.measure.unit.Unit toString()
231   * value. See http://jscience.org/api/javax/measure/unit/NonSI.html or
232   * http://jscience.org/api/javax/measure/unit/SI.html
233   * 
234   * @param measurementType the measurementType to set
235   */
236  public void setMeasurementType(String measurementType) {
237    this.units = Unit.valueOf(measurementType);
238    if (this.value != null) {
239      this.amount = Amount.valueOf(this.value, this.units);
240    }
241  }
242
243  /**
244   * @param sensorId the id of the sensor to set
245   */
246  public void setSensorId(String sensorId) {
247    this.sensorId = sensorId;
248  }
249
250  /**
251   * @param id the id to set
252   * @exception BadSlugException if the slug is not valid.
253   */
254  public void setId(String id) throws BadSlugException {
255    if (Slug.validateSlug(id)) {
256      this.id = id;
257    }
258    else {
259      throw new BadSlugException(id + " is not a valid slug.");
260    }
261  }
262
263  /**
264   * @param value the value to set
265   */
266  public void setValue(Double value) {
267    this.value = value;
268    if (this.amount == null) {
269      this.amount = Amount.valueOf(value, this.units);
270    }
271  }
272
273  /*
274   * (non-Javadoc)
275   * 
276   * @see java.lang.Object#toString()
277   */
278  @Override
279  public String toString() {
280    try {
281      return "Measurement [id= " + id + ", sensorId=" + sensorId + ", timestamp="
282          + DateConvert.convertDate(timestamp) + ", value=" + value + ", units=" + units + "]";
283    }
284    catch (DatatypeConfigurationException e) {
285      // shouldn't happen
286      e.printStackTrace();
287    }
288    return "Measurement [id= " + id + ", sensorId=" + sensorId + ", timestamp=" + timestamp
289        + ", value=" + value + ", units=" + units + "]";
290  }
291
292  /**
293   * @return The units for this measurement.
294   */
295  public Unit<?> units() {
296    return this.units;
297  }
