019package org.wattdepot.common.domainmodel;
020
021import java.util.HashSet;
022import java.util.Set;
023
024import org.wattdepot.common.exception.BadSlugException;
025import org.wattdepot.common.util.Slug;
026
027/**
028 * CollectorProcessDefinition - Represents a process that queries a Sensor and
029 * produces measurements.
030 * 
031 * @author Cam Moore
032 * 
033 */
034public class CollectorProcessDefinition implements IDomainModel {
035  /** A unique id for the CollectorProcessDefinition. */
036  private String id;
037  /** The human readable name. */
038  private String name;
039  /** The id of the sensor making the measurements. */
040  protected String sensorId;
041  /** The number of seconds between polls. */
042  protected Long pollingInterval;
043  /** The id of the depository where the measurements are stored. */
044  protected String depositoryId;
045  /** Additional properties for the Collector. */
046  protected Set<Property> properties;
047  /** The id of the owner of this collector. */
048  private String ownerId;
049
050  /**
051   * Hide the default constructor.
052   */
053  protected CollectorProcessDefinition() {
054    properties = new HashSet<Property>();
055  }
056
057  /**
058   * @param name The name of the CollectorProcessDefinition.
059   * @param sensorId The id of the sensor that measures the environment.
060   * @param poll The number of seconds between polls.
061   * @param depositoryId The depository_id where measurements are stored.
062   * @param ownerId the id of the owner of the collector.
063   */
064  public CollectorProcessDefinition(String name, String sensorId, Long poll, String depositoryId,
065      String ownerId) {
066    this(Slug.slugify(name), name, sensorId, poll, depositoryId, new HashSet<Property>(), ownerId);
067  }
068
069  /**
070   * @param id The unique id for the CollectorProcessDefinition.
071   * @param name The name of the CollectorProcessDefinition.
072   * @param sensorId The id of the sensor that measures the environment.
073   * @param poll The number of seconds between polls.
074   * @param depositoryId The depository_id where measurements are stored.
075   * @param properties The properties associated with this
076   *        CollectorProcessDefinition.
077   * @param ownerId the id of the owner of the collector.
078   */
079  public CollectorProcessDefinition(String id, String name, String sensorId, Long poll,
080      String depositoryId, Set<Property> properties, String ownerId) {
081    this.id = id;
082    this.name = name;
083    this.sensorId = sensorId;
084    this.pollingInterval = poll;
085    this.depositoryId = depositoryId;
086    this.properties = properties;
087    this.ownerId = ownerId;
088  }
089
090  /**
091   * @param e The Property to add.
092   * @return true if added.
093   * @see java.util.List#add(java.lang.Object)
094   */
095  public boolean addProperty(Property e) {
096    return properties.add(e);
097  }

099  /*
100   * (non-Javadoc)
101   * 
102   * @see java.lang.Object#equals(java.lang.Object)
103   */
104  @Override
105  public boolean equals(Object obj) {
106    if (this == obj) {
107      return true;
108    }
109    if (obj == null) {
110      return false;
111    }
112    if (!getClass().isAssignableFrom(obj.getClass())
113        && !obj.getClass().isAssignableFrom(getClass()) && getClass() != obj.getClass()) {
114      return false;
115    }
116    if (obj.getClass().equals(Object.class)) {
117      return false;
118    }
119    CollectorProcessDefinition other = (CollectorProcessDefinition) obj;
120    if (depositoryId == null) {
121      if (other.depositoryId != null) {
122        return false;
123      }
124    }
125    else if (!depositoryId.equals(other.depositoryId)) {
126      return false;
127    }
128    if (id == null) {
129      if (other.id != null) {
130        return false;
131      }
132    }
133    else if (!id.equals(other.id)) {
134      return false;
135    }
136    if (ownerId == null) {
137      if (other.ownerId != null) {
138        return false;
139      }
140    }
141    else if (!ownerId.equals(other.ownerId)) {
142      return false;
143    }
144    if (pollingInterval == null) {
145      if (other.pollingInterval != null) {
146        return false;
147      }
148    }
149    else if (!pollingInterval.equals(other.pollingInterval)) {
150      return false;
151    }
152    if (properties == null) {
153      if (other.properties != null) {
154        return false;
155      }
156    }
157    else if (!properties.equals(other.properties)) {
158      return false;
159    }
160    if (sensorId == null) {
161      if (other.sensorId != null) {
162        return false;
163      }
164    }
165    else if (!sensorId.equals(other.sensorId)) {
166      return false;
167    }
168    if (name == null) {
169      if (other.name != null) {
170        return false;
171      }
172    }
173    else if (!name.equals(other.name)) {
174      return false;
175    }
176    return true;
177  }

241  /*
242   * (non-Javadoc)
243   * 
244   * @see java.lang.Object#hashCode()
245   */
246  @Override
247  public int hashCode() {
248    final int prime = 31;
249    int result = 1;
250    result = prime * result + ((depositoryId == null) ? 0 : depositoryId.hashCode());
251    result = prime * result + ((id == null) ? 0 : id.hashCode());
252    result = prime * result + ((ownerId == null) ? 0 : ownerId.hashCode());
253    result = prime * result + ((pollingInterval == null) ? 0 : pollingInterval.hashCode());
254    result = prime * result + ((properties == null) ? 0 : properties.hashCode());
255    result = prime * result + ((sensorId == null) ? 0 : sensorId.hashCode());
256    result = prime * result + ((name == null) ? 0 : name.hashCode());
257    return result;
258  }

296    }
297  }

324    this.sensorId = sensorId;
