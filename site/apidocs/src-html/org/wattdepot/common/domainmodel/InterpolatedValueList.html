019package org.wattdepot.common.domainmodel;
020
021import java.util.ArrayList;
022import java.util.Date;
023
024/**
025 * MeasuredValueList - a list of measured values.
026 * 
027 * @author Yongwen Xu
028 * 
029 */
030public class InterpolatedValueList {
031  private ArrayList<InterpolatedValue> interpolatedValues;
032  private ArrayList<InterpolatedValue> missingData;
033
034  /**
035   * Default Constructor.
036   */
037  public InterpolatedValueList() {
038    interpolatedValues = new ArrayList<InterpolatedValue>();
039    missingData = new ArrayList<InterpolatedValue>();
040  }
041
042  /**
043   * @return the measuredValues
044   */
045  public ArrayList<InterpolatedValue> getInterpolatedValues() {
046    return interpolatedValues;
047  }
048
049  /**
050   * @param interpolatedValues the measuredValues to set
051   */
052  public void setInterpolatedValues(ArrayList<InterpolatedValue> interpolatedValues) {
053    this.interpolatedValues = interpolatedValues;
054  }
055
056  /**
057   * @return the missing data InterpolatedValues with null for the value.
058   */
059  public ArrayList<InterpolatedValue> getMissingData() {
060    return missingData;
061  }
062
063  /**
064   * @param missingData the missing data to set.
065   */
066  public void setMissingData(ArrayList<InterpolatedValue> missingData) {
067    this.missingData = missingData;
068  }
069
070  /*
071     * (non-Javadoc)
072     *
073     * @see java.lang.Object#toString()
074     */
075  @Override
076  public String toString() {
077    return "InterpolatedValueList [InterpolatedValues=" + interpolatedValues + ", MissingData=" + missingData + "]";
078  }
079
080  /**
081   * Collapses the missing data combining adjacent InterpolatedValues into a single InterpolatedValue.
082   */
083  public void collapseMissingData() {
084    ArrayList<InterpolatedValue> temp = new ArrayList<InterpolatedValue>();
085    Date collapseStart = null;
086    Date collapseEnd = null;
087    String sensorId = null;
088    MeasurementType type = null;
089    for (InterpolatedValue v : missingData) {
090      sensorId = v.getSensorId();
091      type = v.getMeasurementType();
092      Date start = v.getStart();
093      Date end = v.getEnd();
094      if (collapseStart == null) {
095        collapseStart = start;
096        collapseEnd = end;
097      }
098      else {
099        if (collapseEnd.getTime() == start.getTime()) { // old end == start collapse the interval
100          collapseEnd = end;
101        }
102        else { // not adjacent
103          temp.add(new InterpolatedValue(sensorId, null, type, collapseStart, collapseEnd));
104          collapseStart = null;
105          collapseEnd = null;
106        }
107      }
108    }
109    if (collapseStart != null && collapseEnd != null) {
110      temp.add(new InterpolatedValue(sensorId, null, type, collapseStart, collapseEnd));
111    }
112    this.missingData = temp;
113  }
114
115  /**
116   * Collapses the missing data combining adjacent InterpolatedValues into a single InterpolatedValue from
117   * now to the past.
118   */
119  public void collapseMissingDataNowToPast() {
120    ArrayList<InterpolatedValue> temp = new ArrayList<InterpolatedValue>();
121    Date collapseStart = null;
122    Date collapseEnd = null;
123    String sensorId = null;
124    MeasurementType type = null;
125    for (InterpolatedValue v : missingData) {
126      sensorId = v.getSensorId();
127      type = v.getMeasurementType();
128      Date start = v.getStart();
129      Date end = v.getEnd();
130      if (collapseStart == null) {
131        collapseStart = start;
132        collapseEnd = end;
133      }
134      else {
135        if (collapseStart.getTime() == end.getTime()) { // old end == start collapse the interval
136          collapseStart = start;
137        }
138        else { // not adjacent
139          temp.add(new InterpolatedValue(sensorId, null, type, collapseStart, collapseEnd));
140          collapseStart = null;
141          collapseEnd = null;
142        }
143      }
144    }
145    if (collapseStart != null && collapseEnd != null) {
146      temp.add(new InterpolatedValue(sensorId, null, type, collapseStart, collapseEnd));
147    }
148    this.missingData = temp;
149  }
150}
