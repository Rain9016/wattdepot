020package org.wattdepot.extension.openeis.http.api;
021
022import org.restlet.data.LocalReference;
023import org.restlet.data.MediaType;
024import org.restlet.data.Status;
025import org.restlet.ext.freemarker.TemplateRepresentation;
026import org.restlet.representation.Representation;
027import org.restlet.resource.ClientResource;
028import org.restlet.resource.Get;
029import org.wattdepot.common.domainmodel.Depository;
030import org.wattdepot.common.domainmodel.InterpolatedValue;
031import org.wattdepot.common.domainmodel.Sensor;
032import org.wattdepot.common.exception.IdNotFoundException;
033import org.wattdepot.common.exception.MisMatchedOwnerException;
034import org.wattdepot.common.exception.NoMeasurementException;
035import org.wattdepot.common.http.api.API;
036import org.wattdepot.extension.openeis.OpenEISLabels;
037import org.wattdepot.server.http.api.WattDepotServerResource;
038
039import java.util.ArrayList;
040import java.util.Date;
041import java.util.HashMap;
042import java.util.List;
043import java.util.Map;
044import java.util.logging.Level;
045
046/**
047 * OpenEISUIServerResource - User interface for the OpenEIS algorithms. Handles /wattdepot/{orgId}/openeis/ui requests.
048 *
049 * @author Cam Moore
050 *         Created by carletonmoore on 4/10/15.
051 */
052public class OpenEISUIServerResource extends WattDepotServerResource {
053
054  /**
055   * @return The Organization summary information as an HTML Representation.
056   */
057  @Get()
058  public Representation toHtml() {
059    getLogger().log(Level.INFO,
060        "GET " + API.BASE_URI + "{" + orgId + "}/" + OpenEISLabels.OPENEIS + "/ui/");
061    if (isInRole(orgId)) {
062      Map<String, Object> dataModel = new HashMap<String, Object>();
063      dataModel.put("orgId", orgId);
064      Representation rep = null;
065      TemplateRepresentation template = null;
066      try {
067        depot.getOrganization(orgId, true);
068        // get the depositories
069        List<Depository> depos = depot.getDepositories(orgId, false);
070        for (Depository d : depos) {
071          String typeName = d.getMeasurementType().getName();
072          if (!typeName.startsWith("Power") && !typeName.startsWith("Energy") && !typeName.startsWith("Temperature") && !typeName.startsWith("Area")) {
073            depos.remove(d); // remove the ones we aren't interested in.
074          }
075        }
076        Map<String, List<Sensor>> depoSensors = new HashMap<String, List<Sensor>>();
077        List<Sensor> sensors = new ArrayList<Sensor>();
078        List<Sensor> powerSensors = new ArrayList<Sensor>();
079        List<Depository> powerDepositories = new ArrayList<Depository>();
080        List<Sensor> temperatureSensors = new ArrayList<Sensor>();
081        List<Depository> temperatureDepositories = new ArrayList<Depository>();
082        List<Sensor> energySensors = new ArrayList<Sensor>();
083        List<Depository> energyDepositories = new ArrayList<Depository>();
084        Map<String, Map<String, List<Date>>> depsitorySensorInfo = new HashMap<>();
085        for (Depository d : depos) {
086          Map<String, List<Date>> sensorInfo = new HashMap<String, List<Date>>();
087          List<Sensor> sensorList = new ArrayList<Sensor>();
088          depoSensors.put(d.getId(), sensorList);
089          for (String sensorId : depot.listSensors(d.getId(), orgId, false)) {
090            Sensor s = depot.getSensor(sensorId, orgId, false);
091            sensorList.add(s);
092            if (!sensors.contains(s)) {
093              sensors.add(s);
094            }
095            if (d.getMeasurementType().getName().startsWith("Power") && !powerSensors.contains(s)) {
096              powerSensors.add(s);
097              powerDepositories.add(d);
098            }
099            else if (d.getMeasurementType().getName().startsWith("Energy") && !energySensors.contains(s)) {
100              energySensors.add(s);
101              energyDepositories.add(d);
102            }
103            else if (d.getMeasurementType().getName().startsWith("Temperature") && !temperatureSensors.contains(s)) {
104              temperatureSensors.add(s);
105              temperatureDepositories.add(d);
106            }
107            try {
108              InterpolatedValue earliest = depot.getEarliestMeasuredValue(
109                  d.getId(), orgId, sensorId, false);
110              InterpolatedValue latest = depot.getLatestMeasuredValue(
111                  d.getId(), orgId, sensorId, false);
112              List<Date> info = new ArrayList<Date>();
113              info.add(earliest.getStart());
114              info.add(latest.getEnd());
115              sensorInfo.put(sensorId, info);
116              depsitorySensorInfo.put(d.getId(), sensorInfo);
117            }
118            catch (NoMeasurementException e) {
119              // TODO Auto-generated catch block
120              e.printStackTrace();
121            }
122          }
123        }
124        dataModel.put("depositories", depos);
125        dataModel.put("depoSensors", depoSensors);
126        dataModel.put("sensors", sensors);
127        dataModel.put("power_sensors", powerSensors);
128        dataModel.put("power_depositories", powerDepositories);
129        dataModel.put("energy_sensors", energySensors);
130        dataModel.put("energy_depositories", energyDepositories);
131        dataModel.put("temperature_sensors", temperatureSensors);
132        dataModel.put("temperature_depositories", temperatureDepositories);
133        dataModel.put("depository_sensor_info", depsitorySensorInfo);
134      }
135      catch (IdNotFoundException e) {
136        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
137        return null;
138      }
139      catch (MisMatchedOwnerException e) {
140        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
141        return null;
142      }
143      rep = new ClientResource(LocalReference.createClapReference(getClass()
144          .getPackage()) + "/OpenEISUI.ftl").get();
145      template = new TemplateRepresentation(rep, dataModel, MediaType.TEXT_HTML);
146      return template;
147    }
148    return null;
149  }
150}
