020package org.wattdepot.extension.openeis.http.api;
021
022import org.wattdepot.common.domainmodel.Labels;
023import org.wattdepot.extension.openeis.OpenEISLabels;
024
025/**
026 * API - Defines the Strings that make up the WattDepot OPENEIS HTTP API.
027 *
028 * @author Cam Moore
029 */
030public class OpenEISAPI {
031
032/************************************************************************************/
033/********************* OpenEIS http://eis.lbl.gov/openeis.html API ******************/
034/************************************************************************************/
035
036  /**
037   * <b>/wattdepot/{org-id}/openeis/</b> Base URI.
038   */
039  public static final String BASE_URI = org.wattdepot.common.http.api.API.BASE_URI + Labels.ORGANIZATION_ID_VAR + "/"
040      + OpenEISLabels.OPENEIS + "/";
041
042  /**
043   * <b>/wattdepot/{org-id}/openeis/ui/</b> URI for the OpenEIS user interface.
044   */
045  public static final String OPENEIS_UI_URI = BASE_URI + "ui" + "/";
046
047  /**
048   * <b>/wattdepot/{org-id}/openeis/time-series-load-profiling/data/</b> URI to get
049   * last month's power data. Use GET requests.
050   */
051  public static final String OPENEIS_TIME_SERIES_LOAD_DATA_URI = BASE_URI + OpenEISLabels.TIME_SERIES_LOAD_PROFILING + "/" + Labels.DATA + "/";
052
053  /**
054   * <b>/wattdepot/{org-id}/openeis/time-series-load-profiling/gviz/</b> URI to get
055   * last month's power data. Use GET requests.
056   */
057  public static final String OPENEIS_TIME_SERIES_LOAD_GVIZ_URI = BASE_URI + OpenEISLabels.TIME_SERIES_LOAD_PROFILING + "/" + Labels.GVIZ + "/";
058
059  /**
060   * <b>/wattdepot/{org-id}/openeis/heat-map/data/</b> URI to get the last year's hourly power data. Use GET requests.
061   */
062  public static final String OPENEIS_HEAT_MAP_DATA_URI = BASE_URI + OpenEISLabels.HEAT_MAP + "/" + Labels.DATA + "/";
063
064  /**
065   * <b>/wattdepot/{org-id}/openeis/heat-map/gviz/</b> URI to get the last year's hourly power data. Use GET requests.
066   */
067  public static final String OPENEIS_HEAT_MAP_GVIS_URI = BASE_URI + OpenEISLabels.HEAT_MAP + "/" + Labels.GVIZ + "/";
068
069  /**
070   * <b>/wattdepot/{org-id}/openeis/energy-signature/data/</b> URI to get the last year's hourly power data. Use GET requests.
071   */
072  public static final String OPENEIS_ENERGY_SIGNATURE_DATA_URI = BASE_URI + OpenEISLabels.ENERGY_SIGNATURE + "/" + Labels.DATA + "/";
073
074  /**
075   * <b>/wattdepot/{org-id}/openeis/energy-signature/gviz/</b> URI to get the last year's hourly power data. Use GET requests.
076   */
077  public static final String OPENEIS_ENERGY_SIGNATURE_GVIZ_URI = BASE_URI + OpenEISLabels.ENERGY_SIGNATURE + "/" + Labels.GVIZ + "/";
078
079
080  /**
081   * <b>/wattdepot/{org-id}/openeis/longitudinal-benchmarking/data/</b> URI to get a longitudinal benchmark comparison data.
082   * Use GET requests.
083   */
084  public static final String OPENEIS_LONGITUDINAL_BASELINE_DATA_URI = BASE_URI + OpenEISLabels.LONGITUDINAL_BENCHMARKING + "/"
085      + Labels.DATA + "/";
086
087  /**
088   * <b>/wattdepot/{org-id}/openeis/longitudinal-benchmarking/gviz/</b> URI to get a longitudinal benchmark comparison data for google
089   * visualization. Use GET requests.
090   */
091  public static final String OPENEIS_LONGITUDINAL_BASELINE_GVIZ_URI = BASE_URI + OpenEISLabels.LONGITUDINAL_BENCHMARKING + "/"
092      + Labels.GVIZ + "/";
093
094  /**
095   * <b>/wattdepot/{org-id}/openeis/load-analysis/</b> URI to get a load analysis. Use GET requests.
096   */
097  public static final String OPENEIS_LOAD_ANALYSIS = BASE_URI + OpenEISLabels.LOAD_ANALYSIS + "/";
098
099  /**
100   * <b>/wattdepot/{org-id}/openeis/load-duration-curve/data/</b> URI to get a load analysis. Use GET requests.
101   */
102  public static final String OPENEIS_LOAD_DURATION_CURVE_DATA_URI = BASE_URI + OpenEISLabels.LOAD_DURATION_CURVE + "/" + Labels.DATA + "/";
103
104  /**
105   * <b>/wattdepot/{org-id}/openeis/load-duration-curve/gviz/</b> URI to get a load analysis. Use GET requests.
106   */
107  public static final String OPENEIS_LOAD_DURATION_CURVE_GVIZ_URI = BASE_URI + OpenEISLabels.LOAD_DURATION_CURVE + "/" + Labels.GVIZ + "/";
108
109}
