020package org.wattdepot.extension.openeis.server;
021
022import org.restlet.data.Status;
023import org.restlet.resource.ResourceException;
024import org.wattdepot.common.domainmodel.Depository;
025import org.wattdepot.common.domainmodel.InterpolatedValue;
026import org.wattdepot.common.domainmodel.InterpolatedValueList;
027import org.wattdepot.common.domainmodel.Labels;
028import org.wattdepot.common.exception.IdNotFoundException;
029import org.wattdepot.common.util.tstamp.Tstamp;
030import org.wattdepot.extension.openeis.OpenEISLabels;
031import org.wattdepot.extension.openeis.domainmodel.TimeInterval;
032
033import javax.xml.datatype.XMLGregorianCalendar;
034import java.util.Date;
035import java.util.logging.Level;
036
037/**
038 * Base class for creating longitudinal benchmarks.
039 *
040 * @author Cam Moore
041 *         Created by carletonmoore on 4/18/15.
042 */
043public class LongitudinalBenchmarkingServer extends OpenEISServer {
044  private String depositoryId;
045  private String sensorId;
046  private String baseStartString;
047  private TimeInterval interval;
048  private String comparisonStartString;
049  private Integer numberOfIntervals;
050
051  /*
052 * (non-Javadoc)
053 *
054 * @see org.restlet.resource.Resource#doInit()
055 */
056  @Override
057  protected void doInit() throws ResourceException {
058    super.doInit();
059    this.depositoryId = getQuery().getValues(Labels.DEPOSITORY);
060    this.sensorId = getQuery().getValues(Labels.SENSOR);
061    this.baseStartString = getQuery().getValues(OpenEISLabels.BASELINE_START);
062    this.interval = TimeInterval.fromParameter(getQuery().getValues(OpenEISLabels.BASELINE_DURATION));
063    this.comparisonStartString = getQuery().getValues(OpenEISLabels.COMPARISON_START);
064    this.numberOfIntervals = Integer.parseInt(getQuery().getValues(OpenEISLabels.NUM_INTERVALS));
065  }
066
067  /**
068   * Calculates the baseline and comparison values.
069   *
070   * @return An InterpolatedValueList of the baseline and comparison values.
071   */
072  public InterpolatedValueList doRetrieve() {
073    getLogger().log(
074        Level.INFO,
075        "GET /wattdepot/{" + orgId + "}/" + OpenEISLabels.OPENEIS + "/" + OpenEISLabels.LONGITUDINAL_BENCHMARKING +
076            "/?" + Labels.DEPOSITORY + "={" + depositoryId + "}&" + Labels.SENSOR + "={" + sensorId + "}&" +
077            OpenEISLabels.BASELINE_START + "={" + baseStartString + "}&" + OpenEISLabels.BASELINE_DURATION +
078            "={" + interval + "&" + OpenEISLabels.COMPARISON_START + "={" + comparisonStartString + "}&" +
079            OpenEISLabels.NUM_INTERVALS + "={" + numberOfIntervals + "}");
080    try {
081      Depository depository = depot.getDepository(depositoryId, orgId, true);
082      InterpolatedValueList ret = null;
083      if (depository.getMeasurementType().getName().startsWith("Energy")) {
084        XMLGregorianCalendar baseCal = Tstamp.makeTimestamp(baseStartString);
085        Date baseStart = baseCal.toGregorianCalendar().getTime();
086        XMLGregorianCalendar compareCal = Tstamp.makeTimestamp(comparisonStartString);
087        Date compDate = compareCal.toGregorianCalendar().getTime();
088        ret = getDifferenceValues(depositoryId, sensorId, baseStart, interval, 1, true);
089        InterpolatedValueList compareValues = getDifferenceValues(depositoryId, sensorId, compDate, interval, numberOfIntervals, true);
090        for (InterpolatedValue val : compareValues.getInterpolatedValues()) {
091          ret.getInterpolatedValues().add(val);
092        }
093      }
094      return ret;
095    }
096    catch (IdNotFoundException e) {
097      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
098      return null;
099    }
100    catch (Exception e) {
101      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
102      return null;
103    }
104  }
105}
