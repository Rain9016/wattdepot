020package org.wattdepot.extension.openeis.server;
021
022import org.restlet.data.Status;
023import org.restlet.resource.ResourceException;
024import org.wattdepot.common.domainmodel.Depository;
025import org.wattdepot.common.domainmodel.InterpolatedValue;
026import org.wattdepot.common.domainmodel.InterpolatedValueList;
027import org.wattdepot.common.domainmodel.InterpolatedValueValueComparator;
028import org.wattdepot.common.domainmodel.Labels;
029import org.wattdepot.common.exception.IdNotFoundException;
030import org.wattdepot.common.util.tstamp.Tstamp;
031import org.wattdepot.extension.openeis.OpenEISLabels;
032
033import java.util.ArrayList;
034import java.util.Collections;
035import java.util.Date;
036import java.util.logging.Level;
037
038/**
039 * LoadDurationCurveServer - Base class for handling Load Duration Curve requests.
040 *
041 * @author Cam Moore
042 *         Created by carletonmoore on 4/22/15.
043 */
044public class LoadDurationCurveServer extends OpenEISServer {
045  private String depositoryId;
046  private String sensorId;
047  private String startString;
048  private String endString;
049
050  /*
051 * (non-Javadoc)
052 *
053 * @see org.restlet.resource.Resource#doInit()
054 */
055  @Override
056  protected void doInit() throws ResourceException {
057    super.doInit();
058    this.depositoryId = getQuery().getValues(Labels.DEPOSITORY);
059    this.sensorId = getQuery().getValues(Labels.SENSOR);
060    this.startString = getQuery().getValues(Labels.START);
061    this.endString = getQuery().getValues(Labels.END);
062  }
063
064  /**
065   * Retrieves the hourly power data for the given depository, sensor, start and end dates.
066   *
067   * @return An InterpolatedValueList of the hourly power data.
068   */
069  public InterpolatedValueList doRetrieve() {
070    getLogger().log(
071        Level.INFO,
072        "GET /wattdepot/{" + orgId + "}/" + OpenEISLabels.OPENEIS + "/" + OpenEISLabels.LOAD_DURATION_CURVE +
073            "/?" + Labels.DEPOSITORY + "={" + depositoryId + "}&" + Labels.SENSOR + "={" + sensorId + "}&" +
074            Labels.START + "={" + startString + "}&" + Labels.END + "={" + endString + "}");
075    try {
076      Depository depository = depot.getDepository(depositoryId, orgId, true);
077      Date start = Tstamp.makeTimestamp(startString).toGregorianCalendar().getTime();
078      Date end = Tstamp.makeTimestamp(endString).toGregorianCalendar().getTime();
079      InterpolatedValueList values;
080      if (depository.getMeasurementType().getName().startsWith("Power")) {
081        values = getHourlyPointData(depositoryId, sensorId, start, end, false);
082      }
083      else if (depository.getMeasurementType().getName().startsWith("Energy")) {
084        values = getHourlyDifferenceData(depositoryId, sensorId, start, end, false);
085      }
086      else {
087        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not a Power Depository.");
088        return null;
089      }
090      InterpolatedValueValueComparator c = new InterpolatedValueValueComparator();
091      ArrayList<InterpolatedValue> list = values.getInterpolatedValues();
092      Collections.sort(list, c);
093      Collections.reverse(list);
094      values.setInterpolatedValues(list);
095      return values;
096    }
097    catch (IdNotFoundException e) {
098      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
099      return null;
100    }
101    catch (Exception e) {
102      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
103      return null;
104    }
105  }
106
107}
