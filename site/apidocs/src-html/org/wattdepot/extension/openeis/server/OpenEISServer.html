018/**
019 * OpenEISServer - Base class that provides functionality for getting Hourly average values for point data and
020 * Hourly difference values.
021 *
022 * @author Cam Moore
023 */
024public class OpenEISServer extends WattDepotServerResource {
025
026  /**
027   * Returns the hourly point data for the given depositoryId, sensorId, and for the time interval.
028   *
029   * @param depositoryId The Depository.
030   * @param sensorId     The Sensor collecting the data.
031   * @param howLong      The length of time.
032   * @param keepNulls    if true returns InterpolatedValues with a value of null for missing data.
033   * @return An InterpolatedValueList of the hourly data.
034   */
035  public InterpolatedValueList getHourlyPointData(String depositoryId, String sensorId, TimeInterval howLong, boolean keepNulls) {
036    XMLGregorianCalendar now = Tstamp.makeTimestamp();
037    now.setTime(0, 0, 0, 0); // set now to midnight
038    Date nowDate = now.toGregorianCalendar().getTime();
039    XMLGregorianCalendar past = Tstamp.incrementDays(now, howLong.getNumDays() * -1);
040    Date pastDate = past.toGregorianCalendar().getTime();
041    return getHourlyPointData(depositoryId, sensorId, pastDate, nowDate, keepNulls);
042  }

044  /**

052  public InterpolatedValueList getHourlyPointDataYear(String depositoryId, String sensorId, boolean keepNulls) {
053    return getHourlyPointData(depositoryId, sensorId, TimeInterval.ONE_YEAR, keepNulls);

074  /**
075   * Returns an InterpolatedValueList of the values starting at start with an interval of howLong.
076   *
077   * @param depositoryId The Depository.
078   * @param sensorId     The Sensor.
079   * @param start        The start date.
080   * @param howLong      The period to calculate the difference.
081   * @param numIntervals How many values to calculate.
082   * @param keepNulls    if true returns InterpolatedValues with a value of null for missing data.
083   * @return an InterpolatedValueList of the values.
084   */
085  public InterpolatedValueList getDifferenceValues(String depositoryId, String sensorId, Date start, TimeInterval howLong, int numIntervals, boolean keepNulls) {
086    if (isInRole(orgId)) {
087      InterpolatedValueList ret = new InterpolatedValueList();
088      try {
089        Depository depository = depot.getDepository(depositoryId, orgId, true);
090        XMLGregorianCalendar startCal = Tstamp.makeTimestamp(start.getTime());
091        startCal.setTime(0, 0, 0, 0); // start at the beginning of the day.
092        XMLGregorianCalendar endCal = Tstamp.incrementDays(startCal, howLong.getNumDays());
093        for (int i = 0; i < numIntervals; i++) {
094          Date begin = startCal.toGregorianCalendar().getTime();
095          Date end = endCal.toGregorianCalendar().getTime();
096          Double val = null;
097          try {
098            val = depot.getValue(depositoryId, orgId, sensorId, begin, end, false);
099          }
100          catch (NoMeasurementException e) {
101            val = null;
102          }
103          InterpolatedValue v = new InterpolatedValue(sensorId, val, depository.getMeasurementType(), begin, end);
104          if (val == null) {
105            ret.getMissingData().add(v);
106            if (keepNulls) {
107              ret.getInterpolatedValues().add(v);
108            }
109          }
110          else {
111            ret.getInterpolatedValues().add(v);
112          }
113          startCal = Tstamp.incrementDays(startCal, howLong.getNumDays());
114          endCal = Tstamp.incrementDays(endCal, howLong.getNumDays());
115        }
116        return ret;
117      }
118      catch (IdNotFoundException e) {
119        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
120        return null;
121      }
122    }
123    else {
124      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
125      return null;
126    }
127  }

129  /**
130   * Returns the hourly difference data as an InterpolatedValueList.
131   *
132   * @param depositoryId The Depository.
133   * @param sensorId     The Sensor.
134   * @param start        The start of the period.
135   * @param end          The end of the period.
136   * @param keepNulls    if true inserts InterpolatedValues with a value of null in the return list.
137   * @return An InterpolatedValueList of the hourly differences.
138   */
139  public InterpolatedValueList getHourlyDifferenceData(String depositoryId, String sensorId, Date start, Date end, boolean keepNulls) {
140    if (isInRole(orgId)) {
141      try {
142        Depository depository = depot.getDepository(depositoryId, orgId, true);
143        XMLGregorianCalendar startCal = Tstamp.makeTimestamp(start.getTime());
144        startCal.setTime(0, 0, 0, 0);
145        XMLGregorianCalendar endCal = Tstamp.makeTimestamp(end.getTime());
146        endCal.setTime(0, 0, 0, 0);
147        endCal = Tstamp.incrementDays(endCal, 1);
148        List<XMLGregorianCalendar> times = Tstamp.getTimestampList(startCal, endCal, 60);
149        InterpolatedValueList ret = new InterpolatedValueList();
150        for (int i = 1; i < times.size(); i++) {
151          XMLGregorianCalendar begin = times.get(i - 1);
152          Date beginDate = begin.toGregorianCalendar().getTime();
153          XMLGregorianCalendar stop = times.get(i);
154          Date endDate = stop.toGregorianCalendar().getTime();
155          Double val = null;
156          try {
157            val = depot.getValue(depositoryId, orgId, sensorId, beginDate, endDate, false);
158          }
159          catch (NoMeasurementException e) { // if there are no measurements just add null
160            val = null;
161          }
162          InterpolatedValue iv = new InterpolatedValue(sensorId, val, depository.getMeasurementType(), beginDate, endDate);
163          if (!keepNulls) {
164            if (val != null) {
165              ret.getInterpolatedValues().add(iv);
166            }
167            else {
168              ret.getMissingData().add(iv);
169            }
170          }
171          else {
172            ret.getInterpolatedValues().add(iv);
173            if (val == null) {
174              ret.getMissingData().add(iv);
175            }
176          }
177        }
178        return ret;
179      }
180      catch (IdNotFoundException e) {
181        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
182        return null;
183      }
184    }
185    else {
186      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
187      return null;
188    }
189  }

191  /**
192   * Returns the hourly average data as an InterpolatedValueList.
193   *
194   * @param depositoryId The Depository.
195   * @param sensorId     The Sensor.
196   * @param start        The start of the period.
197   * @param end          The end of the period.
198   * @param keepNulls    if true insert InterpolatedValues with a null value.
199   * @return An InterpolatedValueList of the hourly averages.
200   */
201  public InterpolatedValueList getHourlyPointData(String depositoryId, String sensorId, Date start, Date end, boolean keepNulls) {
202    if (isInRole(orgId)) {
203      InterpolatedValueList ret = new InterpolatedValueList();
204      try {
205        Depository depository = depot.getDepository(depositoryId, orgId, true);
206        XMLGregorianCalendar startCal = Tstamp.makeTimestamp(start.getTime());
207        startCal.setTime(0, 0, 0, 0);
208        XMLGregorianCalendar endCal = Tstamp.makeTimestamp(end.getTime());
209        endCal.setTime(0, 0, 0, 0);
210        endCal = Tstamp.incrementDays(endCal, 1);
211        List<XMLGregorianCalendar> times = Tstamp.getTimestampList(startCal, endCal, 60);
212        for (int i = 1; i < times.size(); i++) {
213          XMLGregorianCalendar begin = times.get(i - 1);
214          Date beginDate = begin.toGregorianCalendar().getTime();
215          XMLGregorianCalendar stop = times.get(i);
216          Date endDate = stop.toGregorianCalendar().getTime();
217          Double val = 0.0;
218          List<Measurement> measurements = depot.getMeasurements(depositoryId, orgId, sensorId, beginDate, endDate, false);
219          if (measurements.size() > 0) {
220            for (Measurement m : measurements) {
221              val += m.getValue();
222            }
223            val = val / measurements.size();
224          }
225          else {
226            val = null;
227          }
228          InterpolatedValue iv = new InterpolatedValue(sensorId, val, depository.getMeasurementType(), beginDate, endDate);
229          if (!keepNulls) {
230            if (val != null) {
231              ret.getInterpolatedValues().add(iv);
232            }
233            else {
234              ret.getMissingData().add(iv);
235            }
236          }
237          else {
238            ret.getInterpolatedValues().add(iv);
239            if (val == null) {
240              ret.getMissingData().add(iv);
241            }
242          }
243        }
244        return ret;
245      }
246      catch (IdNotFoundException e) {
247        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
248        return null;
249      }
250    }
251    else {
252      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
253      return null;
254    }
255  }
