039/**
040 * WattDepotServer - The main class that starts up the WattDepotServer
041 * coordinates the different interfaces.
042 * 
043 * @author Cam Moore
044 * 
045 */
046public class WattDepotServer {
047
048  /** Holds the Restlet component for the server. */
049  private WattDepotComponent restletServer;
050  /** Holds the WattDepot persistent store for the server. */
051  private WattDepotPersistence depot;
052  /** Holds the ServerProperties instance associated with this server. */
053  private ServerProperties serverProperties;
054  /** Holds the WattDepotLogger for the server. */
055  private Logger logger = null;
056  /** Holds the hostname associated with this Server. */
057  private String hostName = null;
058

060   * Default constructor.

062   */
063  public WattDepotServer() {

092
093  /**
094   * Creates a new instance of the WattDepot server.
095   * 
096   * @param properties
097   *          The ServerProperties used to initialize this server.
098   * @return The WattDepotServer.
099   * @throws Exception
100   *           if there is a problem starting the server.
101   */
102  public static WattDepotServer newInstance(ServerProperties properties)
103      throws Exception {
104    int port = Integer.parseInt(properties.get(ServerProperties.PORT_KEY));
105    WattDepotServer wattDepotServer = new WattDepotServer();
106    // System.out.println("WattDepotServer.");
107//        LoggerUtil.showLoggers();
108    boolean enableLogging = Boolean.parseBoolean(properties
109        .get(ServerProperties.ENABLE_LOGGING_KEY));
110    wattDepotServer.serverProperties = properties;
111    wattDepotServer.hostName = wattDepotServer.serverProperties.getFullHost();
112
113    // Get the WattDepotPersistence implementation.
114    String depotClass = properties.get(ServerProperties.WATT_DEPOT_IMPL_KEY);
115    wattDepotServer.depot = (WattDepotPersistence) Class.forName(depotClass)
116        .getConstructor(ServerProperties.class).newInstance(properties);
117    if (wattDepotServer.depot.getSessionOpen() != wattDepotServer.depot.getSessionClose()) {
118      throw new RuntimeException("opens and closed mismatched.");
119    }
120    wattDepotServer.depot.initializeMeasurementTypes();
121    if (wattDepotServer.depot.getSessionOpen() != wattDepotServer.depot.getSessionClose()) {
122      throw new RuntimeException("opens and closed mismatched.");
123    }
124    wattDepotServer.depot.initializeSensorModels();
125    if (wattDepotServer.depot.getSessionOpen() != wattDepotServer.depot.getSessionClose()) {
126      throw new RuntimeException("opens and closed mismatched.");
127    }
128    wattDepotServer.depot.setServerProperties(properties);
129    wattDepotServer.restletServer = new WattDepotComponent(wattDepotServer.depot);
130
131    // Adds a HTTP or HTTPS server connector
132    if (properties.get(ServerProperties.SSL).equals(ServerProperties.TRUE)) {
133
134      Server server = new Server(new Context(), Protocol.HTTPS, port, wattDepotServer.restletServer);
135      Series<Parameter> parameters = server.getContext().getParameters();
136      parameters.add("sslContextFactory", "org.restlet.ext.ssl.DefaultSslContextFactory");
137      parameters.add("keyStorePath", properties.get(ServerProperties.SSL_KEYSTORE_PATH));
138      parameters.add("keyStorePassword", properties.get(ServerProperties.SSL_KEYSTORE_PASSWORD));
139      parameters.add("keyPassword", properties.get(ServerProperties.SSL_KEYSTORE_KEY_PASSWORD));
140      parameters.add("keyStoreType", properties.get(ServerProperties.SSL_KEYSTORE_TYPE));
141//      parameters.add("protocol", "TLS");
142//      parameters.add("wantClientAuthentication", "false");
143      wattDepotServer.restletServer.getServers().add(server);
144    }
145    else {
146      wattDepotServer.restletServer.getServers().add(Protocol.HTTP, port);
147    }
148
149    wattDepotServer.logger = wattDepotServer.restletServer.getLogger();
150
151    // Set up logging.
152    if (enableLogging) {
153      LoggerUtil.useConsoleHandler();
154      Logger base = LogManager.getLogManager().getLogger(Logger.GLOBAL_LOGGER_NAME);
155      base = base.getParent();
156      String level = properties.get(ServerProperties.LOGGING_LEVEL_KEY);
157     LoggerUtil.setLoggingLevel(base, level);
158
159      wattDepotServer.logger.info("Starting WattDepot server.");
160      wattDepotServer.logger.info("Host: " + wattDepotServer.hostName);
161      wattDepotServer.logger.info(wattDepotServer.serverProperties.echoProperties());
162    }
163    else {
164      LoggerUtil.useConsoleHandler();
165      Logger base = LogManager.getLogManager().getLogger(Logger.GLOBAL_LOGGER_NAME);
166      base = base.getParent();
167      LoggerUtil.setLoggingLevel(base, Level.SEVERE.toString());
168    }
169    wattDepotServer.restletServer.start();
170
171    wattDepotServer.logger.info("WattDepot server now running.");
172
173//    LoggerUtil.showLoggers();
174    return wattDepotServer;
175  }

211  /**
212   * Stops the WattDepotServer.
213   * 
214   * @throws Exception

216   */
217  public void stop() throws Exception {
218    this.depot.stop();
219    this.restletServer.stop();
220  }
221
222  /**
223   * @param args
224   *          commandline arguments.
225   * @throws Exception
226   *           if there is a problem starting the components.
227   */
228  public static void main(String[] args) throws Exception {
229    Options options = new Options();
230    options.addOption("h", "help", false, "Print this message");
231    options
232        .addOption("d", "directoryName", true,
233            "subdirectory under ~/.wattdepot where this server's files are to be kept.");
234
235    CommandLine cmd = null;
236    String directoryName = null;
237
238    CommandLineParser parser = new PosixParser();
239    HelpFormatter formatter = new HelpFormatter();
240    try {
241      cmd = parser.parse(options, args);
242    }
243    catch (ParseException e) {
244      System.err.println("Command line parsing failed. Reason: "
245          + e.getMessage() + ". Exiting.");
246      System.exit(1);
247    }
248    if (cmd.hasOption("h")) {
249      formatter.printHelp("WattDepotServer", options);
250      System.exit(0);
251    }
252    if (cmd.hasOption("d")) {
253      directoryName = cmd.getOptionValue("d");
254    }
255    LoggerUtil.disableLogging();
256    if (directoryName == null || directoryName.length() == 0) {
257      WattDepotServer.newInstance();
258    }
259    else {
260      WattDepotServer.newInstance(directoryName);
261    }
262  }
