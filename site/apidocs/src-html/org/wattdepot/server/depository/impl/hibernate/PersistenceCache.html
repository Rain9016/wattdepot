020package org.wattdepot.server.depository.impl.hibernate;
021
022import org.wattdepot.common.domainmodel.Depository;
023import org.wattdepot.common.domainmodel.Organization;
024import org.wattdepot.common.domainmodel.Sensor;
025
026import java.util.HashMap;
027
028/**
029 * In memory cache of Organizations, Depositories, and Sensors. Intended to speed up WattDepotPersistence. It is a lazy cache.
030 * @author Cam Moore
031 */
032public class PersistenceCache {
033  private HashMap<String, Organization> orgMap;
034  private HashMap<String, HashMap<String, Depository>> orgDepoMap;
035  private HashMap<String, HashMap<String, Sensor>> orgSensorMap;
036
037  /** Creates a new PersistenceCache. */
038  public PersistenceCache() {
039    this.orgMap = new HashMap<String, Organization>();
040    this.orgDepoMap = new HashMap<String, HashMap<String, Depository>>();
041    this.orgSensorMap = new HashMap<String, HashMap<String, Sensor>>();
042  }
043
044  /**
045   * Gets the Organization for the given orgId.
046   * @param orgId the organizations id.
047   * @return The Organization for the given id or null if the id is not in the cache.
048   */
049  public Organization getOrganization(String orgId) {
050    return orgMap.get(orgId);
051  }
052
053  /**
054   * Stores the Organization in the cache.
055   * @param org The organization to store.
056   * @return The old Organization in the cache or null if the organization is new to the cache.
057   */
058  public Organization putOrganization(Organization org) {
059    return orgMap.put(org.getId(), org);
060  }
061
062  /**
063   * Deletes the organization from the cache.
064   * @param org The organization to delete.
065   * @return The old organization from the cache or null if the organization wasn't in the cache.
066   */
067  public Organization deleteOrganization(Organization org) {
068    return orgMap.remove(org.getId());
069  }
070
071  /**
072   * Gets the Depository defined by the depId and orgId.
073   * @param depId The depository id.
074   * @param orgId The organization id.
075   * @return The Depository for the given ids, or null if it is undefined.
076   */
077  public Depository getDepository(String depId, String orgId) {
078    HashMap<String, Depository> depMap = orgDepoMap.get(orgId);
079    if (depMap != null) {
080      return depMap.get(depId);
081    }
082    return null;
083  }
084
085  /**
086   * Stores the Depository in the cache.
087   * @param dep The Depository to store.
088   * @return The old Depository in the cache or null if the depository wasn't in the cache.
089   */
090  public Depository putDepository(Depository dep) {
091    HashMap<String, Depository> depMap = orgDepoMap.get(dep.getOrganizationId());
092    if (depMap == null) {
093      depMap = new HashMap<String, Depository>();
094      orgDepoMap.put(dep.getOrganizationId(), depMap);
095    }
096    return depMap.put(dep.getId(), dep);
097  }
098
099  /**
100   * Deletes the depository from the cache.
101   * @param depository The Depository to delete.
102   * @return The old Depository or null if the depository wasn't in the cache.
103   */
104  public Depository deleteDepository(Depository depository) {
105    HashMap<String, Depository> depMap = orgDepoMap.get(depository.getOrganizationId());
106    if (depMap == null) {
107      return null;
108    }
109    return depMap.remove(depository.getId());
110  }
111
112  /**
113   * Gets the Sensor defined by the sensorId and orgId.
114   * @param sensorId The sensor id.
115   * @param orgId the organizations id.
116   * @return The Sensor from the cache or null if the sensor isn't defined.
117   */
118  public Sensor getSensor(String sensorId, String orgId) {
119    HashMap<String, Sensor> sensorMap = orgSensorMap.get(orgId);
120    if (sensorMap != null) {
121      return sensorMap.get(sensorId);
122    }
123    return null;
124  }
125
126  /**
127   * Stores the sensor in the cache.
128   * @param sensor The Sensor to store.
129   * @return The old Sensor from the cache or null if the Sensor wasn't in the cache.
130   */
131  public Sensor putSensor(Sensor sensor) {
132    HashMap<String, Sensor> sensorMap = orgSensorMap.get(sensor.getOrganizationId());
133    if (sensorMap == null) {
134      sensorMap = new HashMap<String, Sensor>();
135      orgSensorMap.put(sensor.getOrganizationId(), sensorMap);
136    }
137    return sensorMap.put(sensor.getId(), sensor);
138  }
139
140  /**
141   * Deletes the Sensor from the cache.
142   * @param sensor The Sensor to delete.
143   * @return The old sensor from the cache or null if the sensor wasn't in the cache.
144   */
145  public Sensor deleteSensor(Sensor sensor) {
146    HashMap<String, Sensor> sensorMap = orgSensorMap.get(sensor.getOrganizationId());
147    if (sensorMap == null) {
148      return null;
149    }
150    return sensorMap.remove(sensor.getId());
151  }
152}
