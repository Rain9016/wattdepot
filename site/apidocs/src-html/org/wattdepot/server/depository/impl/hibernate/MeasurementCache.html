020package org.wattdepot.server.depository.impl.hibernate;
021
022import org.wattdepot.common.domainmodel.InterpolatedValue;
023
024import java.util.HashMap;
025
026/**
027 * Caches the earliest and latest measurements organized by Organization, Depository and Sensor.
028 * Created by Cam Mmoore on 12/3/14.
029 * @author Cam Moore
030 */
031public class MeasurementCache {
032  // OrgId, DepositoryId, SensorId, MeasurementChachePair
033  private HashMap<String, HashMap<String, HashMap<String, MeasurementCachePair>>> cache;
034
035  /** Default constructor.*/
036  public MeasurementCache() {
037    this.cache = new HashMap<String, HashMap<String, HashMap<String, MeasurementCachePair>>>();
038  }
039
040  /**
041   * Gets the earliest measurement date for the given sensor.
042   * @param orgId The organization id.
043   * @param depotId The depository id.
044   * @param sensorId The sensor id.
045   * @return The earliest measurement as an InterpolatedValue or null if there aren't any measurements.
046   */
047  public InterpolatedValue getEarliest(String orgId, String depotId, String sensorId) {
048    HashMap<String, HashMap<String, MeasurementCachePair>> orgMap = cache.get(orgId);
049    if (orgMap == null) {
050      return null;
051    }
052    HashMap<String, MeasurementCachePair> depotMap = orgMap.get(depotId);
053    if (depotMap == null) {
054      return null;
055    }
056    MeasurementCachePair sensorInfo = depotMap.get(sensorId);
057    if (sensorInfo == null) {
058      return null;
059    }
060    else {
061      return sensorInfo.getEarliest();
062    }
063  }
064
065  /**
066   * Gets the date of the latest measurement for the given sensor.
067   * @param orgId The organization id.
068   * @param depotId The depository id.
069   * @param sensorId The sensor id.
070   * @return The latest measurement as an InterpolatedValue or null if there aren't any measurements.
071   */
072  public InterpolatedValue getLatest(String orgId, String depotId, String sensorId) {
073    HashMap<String, HashMap<String, MeasurementCachePair>> orgMap = cache.get(orgId);
074    if (orgMap == null) {
075      return null;
076    }
077    HashMap<String, MeasurementCachePair> depotMap = orgMap.get(depotId);
078    if (depotMap == null) {
079      return null;
080    }
081    MeasurementCachePair sensorInfo = depotMap.get(sensorId);
082    if (sensorInfo == null) {
083      return null;
084    }
085    else {
086      return sensorInfo.getLatest();
087    }
088  }
089
090  /**
091   * Updates the Sensor's information with the given InterpolatedValue.
092   * @param orgId The organization id.
093   * @param depotId The depository id.
094   * @param sensorId The sensor id.
095   * @param iv The interpolated value for the measurement.
096   */
097  public void update(String orgId, String depotId, String sensorId, InterpolatedValue iv) {
098    HashMap<String, HashMap<String, MeasurementCachePair>> orgMap = cache.get(orgId);
099    if (orgMap == null) {
100      orgMap = new HashMap<String, HashMap<String, MeasurementCachePair>>();
101      cache.put(orgId, orgMap);
102    }
103    HashMap<String, MeasurementCachePair> depotMap = orgMap.get(depotId);
104    if (depotMap == null) {
105      depotMap = new HashMap<String, MeasurementCachePair>();
106      orgMap.put(depotId, depotMap);
107    }
108    MeasurementCachePair sensorInfo = depotMap.get(sensorId);
109    if (sensorInfo == null) {
110      sensorInfo = new MeasurementCachePair(iv);
111      depotMap.put(sensorId, sensorInfo);
112    }
113    else {
114      sensorInfo.update(iv);
115    }
116  }
117}
