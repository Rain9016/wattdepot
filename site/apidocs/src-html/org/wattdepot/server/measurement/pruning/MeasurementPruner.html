047/**
048 * MeasurementGarbageCollector - Removes measurements from the WattDepot
049 * repository that are at a higher frequency sampling rate than desired.
050 * 
051 * @author Cam Moore
052 * 
053 */
054public class MeasurementPruner extends TimerTask {
055
056  /**
057   * The window to get the measurements. Hopefully allows for quicker
058   * performance.
059   */
060  public static final int PRUNE_WINDOW = 6 * 60;
061
062  private WattDepotPersistence persistance;
063  private MeasurementPruningDefinition definition;
064  private boolean debug;
065
066  /**
067   * Create a MeasurementGarbageCollector.
068   * 
069   * @param properties The ServerProperties that define the type of persistence.
070   * @param gcdId The id of the MeasurementPruningDefintion.
071   * @param orgId The id of the Organization.
072   * @param debug true if want debugging information.
073   * @throws Exception If there is a problem instantiating the
074   *         WattDepotPersistence.
075   */
076  public MeasurementPruner(ServerProperties properties, String gcdId, String orgId, boolean debug)
077      throws Exception {
078    // Get the WattDepotPersistence implementation.
079    String depotClass = properties.get(ServerProperties.WATT_DEPOT_IMPL_KEY);
080    this.persistance = (WattDepotPersistence) Class.forName(depotClass)
081        .getConstructor(ServerProperties.class).newInstance(properties);
082    this.definition = this.persistance.getMeasurementPruningDefinition(gcdId, orgId, true);
083    this.debug = debug;
084  }
085
086  /**
087   * @return the definition
088   */
089  public MeasurementPruningDefinition getDefinition() {
090    return definition;
091  }
092
093  /**
094   * @return The end of the collection window.
095   */
096  private Date getEndDate() {
097    XMLGregorianCalendar now;
098    Date ret = null;
099    try {
100      now = DateConvert.convertDate(new Date());
101      XMLGregorianCalendar endCal = Tstamp
102          .incrementDays(now, -1 * definition.getIgnoreWindowDays());
103      ret = DateConvert.convertXMLCal(endCal);
104    }
105    catch (DatatypeConfigurationException e) {
106      // TODO Auto-generated catch block
107      e.printStackTrace();
108    }
109    return ret;
110  }
111
112  /**
113   * @param sensorId the id of the sensor to get measurements for.
114   * @return The Measurements in the collection window.
115   * @throws IdNotFoundException if there is a problem with the
116   *         MeasurementPruningDefintion.
117   */
118  private List<Measurement> getMeasurementsToCheck(String sensorId) throws IdNotFoundException {
119    Date start = getStartDate();
120    Date end = getEndDate();
121    if (debug) {
122      SimpleDateFormat df = new SimpleDateFormat();
123      System.out.println("Collection window for " + sensorId + " is from " + df.format(start)
124          + " to " + df.format(end));
125    }
126    return this.persistance.getMeasurements(this.definition.getDepositoryId(),
127        this.definition.getOrgId(), sensorId, end, start, false);
128  }
129
130  /**
131   * @return A list of measurements to garbage collect. They are at a higher
132   *         sample rate than desired.
133   * @throws IdNotFoundException if there is a problem with the
134   *         MeasurementPruningDefinition.
135   */
136  public List<Measurement> getMeasurementsToDelete() throws IdNotFoundException {
137    Long startTime = 0l;
138    Long endTime = 0l;
139    Long diff = 0l;
140    if (debug) {
141      startTime = System.nanoTime();
142    }
143    List<Measurement> ret = new ArrayList<Measurement>();
144    SensorGroup group = this.persistance.getSensorGroup(this.definition.getSensorId(),
145        this.definition.getOrgId(), false);
146    if (group != null) {
147      for (String s : group.getSensors()) {
148        ret.addAll(getMeasurementsToDelete(s));
149      }
150    }
151    else {
152      ret = getMeasurementsToDelete(this.definition.getSensorId());
153    }
154    if (debug) {
155      endTime = System.nanoTime();
156      diff = endTime - startTime;
157      System.out.println("getMeasurementsToDelete() took " + (diff / 1E9) + " secs.");
158    }
159    return ret;
160  }
161
162  /**
163   * @param sensorId The id of the sensor making the measurements.
164   * @return The List of measurements to delete for the given sensor.
165   * @throws IdNotFoundException if there is a problem with the sensorId.
166   */
167  private List<Measurement> getMeasurementsToDelete(String sensorId) throws IdNotFoundException {
168    List<Measurement> ret = new ArrayList<Measurement>();
169    List<Measurement> check = getMeasurementsToCheck(sensorId);
170    int size = check.size();
171    int index = 1;
172    int baseIndex = 0;
173    while (index < size - 1) {
174      long secondsBetween = Math.abs((check.get(index).getDate().getTime() - check.get(baseIndex)
175          .getDate().getTime()) / 1000);
176      if (secondsBetween < definition.getMinGapSeconds()) {
177        ret.add(check.get(index++));
178      }
179      else {
180        baseIndex = index;
181        index++;
182      }
183    }
184    return ret;
185
186  }

195  /**
196   * @return The start of the collection window.
197   */
198  private Date getStartDate() {
199    XMLGregorianCalendar now;
200    Date ret = null;
201    try {
202      now = DateConvert.convertDate(new Date());
203      XMLGregorianCalendar startCal = Tstamp.incrementDays(now,
204          -1 * definition.getIgnoreWindowDays());
205      startCal = Tstamp.incrementDays(startCal, -1 * definition.getCollectWindowDays());
206      ret = DateConvert.convertXMLCal(startCal);
207    }
208    catch (DatatypeConfigurationException e) {
209      // TODO Auto-generated catch block
210      e.printStackTrace();
211    }
212    return ret;
213  }

215  /**
216   * Processes the command line arguments and runs the
217   * MeasurementGarbageCollector one time.
218   * 
219   * @param args The command line arguments.
220   * @throws Exception if there is a problem.
221   */
222  public static void main(String[] args) throws Exception {
223    Options options = new Options();
224    options
225        .addOption("h", "help", false,
226            "Usage: MeasurementPruner -o <orgId> -m <measurement pruning definition id>"
227                + " [-d] [-s]");
228    options.addOption("o", "orgId", true, "Organization Id.");
229    options.addOption("m", "mpd", true, "MeasurementPruningDefinition Id.");
230    options.addOption("d", "debug", false, "Display debugging information.");
231    options.addOption("s", "single", false, "Run gc only once, right away.");
232
233    CommandLine cmd = null;
234    String orgId = null;
235    String gcdId = null;
236    boolean debug = false;
237    boolean single = false;
238    CommandLineParser parser = new PosixParser();
239    HelpFormatter formatter = new HelpFormatter();
240    try {
241      cmd = parser.parse(options, args);
242    }
243    catch (ParseException e) {
244      System.err.println("Command line parsing failed. Reason: " + e.getMessage() + ". Exiting.");
245      System.exit(1);
246    }
247    if (cmd.hasOption("h")) {
248      formatter.printHelp("MeasurementGarbageCollector", options);
249      System.exit(0);
250    }
251    if (cmd.hasOption("o")) {
252      orgId = cmd.getOptionValue("o");
253    }
254    else {
255      orgId = Organization.ADMIN_GROUP.getId();
256    }
257    if (cmd.hasOption("m")) {
258      gcdId = cmd.getOptionValue("m");
259    }
260    debug = cmd.hasOption("d");
261    single = cmd.hasOption("s");
262    if (debug) {
263      System.out.println("Measurement Garbage Collection:");
264      System.out.println("Org Id = " + orgId);
265      System.out.println("GCD Id = " + gcdId);
266      System.out.println("Single run = " + single);
267    }
268    ServerProperties properties = new ServerProperties();
269    // if (debug) {
270    // properties.set(ServerProperties.SERVER_TIMING_KEY,
271    // ServerProperties.TRUE);
272    // }
273    MeasurementPruner mgc = new MeasurementPruner(properties, gcdId, orgId, debug);
274    if (single) {
275      mgc.pruneMeasurements();
276    }
277    else {
278      // Set up the TimerTask to run the gc at the right time.
279      if (debug) {
280        System.out.println("Setting up Timer for " + mgc);
281      }
282      Timer t = new Timer();
283      t.schedule(mgc, mgc.millisToNextRun(), mgc.getGCPeriod());
284    }
285  }
286
287  /**
288   * @return The number of milliseconds to wait till the next expected run time.
289   */
290  private long millisToNextRun() {
291    if (debug) {
292      System.out.print("milliseconds to next run is ");
293    }
294    if (definition.getNextRun() == null) {
295      if (debug) {
296        System.out.println("0");
297      }
298      return 0l;
299    }
300    else {
301      long delay = definition.getNextRun().getTime() - new Date().getTime();
302      if (debug) {
303        System.out.println(delay);
304      }
305      if (delay < 0) {
306        delay = 0l;
307      }
308      return delay;
309    }
310  }
311
312  /**
313   * @return The number of milliseconds to wait between GC runs.
314   */
315  private long getGCPeriod() {
316    int period = 24 * 60 * 60 * 1000; // defaults to once a day
317    if (definition.getCollectWindowDays() > 1) {
318      period = (definition.getCollectWindowDays() - 1) * 24 * 60 * 60 * 1000;
319      if (debug) {
320        System.out.println("GC period is " + period + " milliseconds.");
321      }
322    }
323    return period;
324  }

326  /*

331  @Override
332  public void run() {
333    try {
334      pruneMeasurements();
335    }
336    catch (DatatypeConfigurationException e) {
337      e.printStackTrace();
338    }
339    catch (IdNotFoundException e) {
340      e.printStackTrace();
341    }

344  /**
345   * Prunes the measurements.
346   * 
347   * @throws DatatypeConfigurationException if there is a problem with
348   *         DateConvert.
349   * @throws IdNotFoundException if there is a problem with persistence.
350   */
351  public void pruneMeasurements() throws DatatypeConfigurationException, IdNotFoundException {
352    Date lastStarted = new Date();
353    Integer deleted = 0;
354    if (debug) {
355      System.out.println("Starting run at " + new SimpleDateFormat().format(lastStarted));
356    }
357    // figure out the 6 hour windows to delete
358    XMLGregorianCalendar start = DateConvert.convertDate(getStartDate());
359    XMLGregorianCalendar end = DateConvert.convertDate(getEndDate());
360    List<XMLGregorianCalendar> windows = Tstamp.getTimestampList(start, end, PRUNE_WINDOW);
361    String sensorId = this.definition.getSensorId();
362    SensorGroup group = this.persistance
363        .getSensorGroup(sensorId, this.definition.getOrgId(), false);
364    if (group != null) {
365      for (String s : group.getSensors()) {
366        for (int i = 0; i < windows.size() - 1; i++) {
367          Date windowStart = DateConvert.convertXMLCal(windows.get(i));
368          Date windowEnd = DateConvert.convertXMLCal(windows.get(i + 1));
369          List<Measurement> check = this.persistance.getMeasurements(
370              this.definition.getDepositoryId(), this.definition.getOrgId(), s, windowStart,
371              windowEnd, false);
372          int size = check.size();
373          if (debug) {
374            System.out.println("Sensor " + s + ": " + windowStart + " to " + windowEnd + " has "
375                + size + " measurements");
376          }
377          int index = 1;
378          int baseIndex = 0;
379          while (index < size - 1) {
380            long secondsBetween = Math.abs((check.get(index).getDate().getTime() - check
381                .get(baseIndex).getDate().getTime()) / 1000);
382            if (secondsBetween < definition.getMinGapSeconds()) {
383              this.persistance.deleteMeasurement(this.definition.getDepositoryId(),
384                  this.definition.getOrganizationId(), check.get(index++).getId());
385              deleted++;
386            }
387            else {
388              baseIndex = index;
389              index++;
390            }
391          }
392        }
393
394      }
395    }
396    else {
397      for (int i = 0; i < windows.size() - 1; i++) {
398        Date windowStart = DateConvert.convertXMLCal(windows.get(i));
399        Date windowEnd = DateConvert.convertXMLCal(windows.get(i + 1));
400        List<Measurement> check = this.persistance.getMeasurements(
401            this.definition.getDepositoryId(), this.definition.getOrgId(), sensorId, windowStart,
402            windowEnd, false);
403        int size = check.size();
404        if (debug) {
405          System.out.println("Sensor " + sensorId + ": " + windowStart + " to " + windowEnd
406              + " has " + size + " measurements");
407        }
408        int index = 1;
409        int baseIndex = 0;
410        while (index < size - 1) {
411          long secondsBetween = Math.abs((check.get(index).getDate().getTime() - check
412              .get(baseIndex).getDate().getTime()) / 1000);
413          if (secondsBetween < definition.getMinGapSeconds()) {
414            this.persistance.deleteMeasurement(this.definition.getDepositoryId(),
415                this.definition.getOrganizationId(), check.get(index++).getId());
416            deleted++;
417          }
418          else {
419            baseIndex = index;
420            index++;
421          }
422        }
423      }
424    }
425    Date lastCompleted = new Date();
426    this.definition.setLastStarted(lastStarted);
427    this.definition.setLastCompleted(lastCompleted);
428    this.definition.setNumMeasurementsCollected(deleted);
429    try {
430      this.persistance.updateMeasurementPruningDefinition(this.definition);
431    }
432    catch (IdNotFoundException e) {
433      e.printStackTrace();
434    }
435    if (debug) {
436      System.out.println("Finished run at " + new SimpleDateFormat().format(lastCompleted));
437      System.out.println("Deleted " + deleted + " measurements.");
438    }
439  }
