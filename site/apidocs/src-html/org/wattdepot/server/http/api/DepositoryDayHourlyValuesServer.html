020package org.wattdepot.server.http.api;
021
022import org.restlet.data.Status;
023import org.restlet.resource.ResourceException;
024import org.wattdepot.common.domainmodel.Depository;
025import org.wattdepot.common.domainmodel.InterpolatedValue;
026import org.wattdepot.common.domainmodel.InterpolatedValueList;
027import org.wattdepot.common.domainmodel.Labels;
028import org.wattdepot.common.domainmodel.Measurement;
029import org.wattdepot.common.exception.IdNotFoundException;
030import org.wattdepot.common.exception.NoMeasurementException;
031import org.wattdepot.common.util.DateConvert;
032import org.wattdepot.common.util.tstamp.Tstamp;
033
034import javax.xml.datatype.DatatypeConfigurationException;
035import javax.xml.datatype.XMLGregorianCalendar;
036import java.text.ParseException;
037import java.util.Date;
038import java.util.List;
039import java.util.logging.Level;
040
041/**
042 * DepositoryHourlyServer - Base class for handling hourly sample HTTP requests.
043 *
044 * @author Cam Moore
045 */
046public class DepositoryDayHourlyValuesServer extends WattDepotServerResource {
047  private String depositoryId;
048  private String sensorId;
049  private String timestamp;
050  private String dataType;
051
052  /*
053   * (non-Javadoc)
054   *
055   * @see org.restlet.resource.Resource#doInit()
056   */
057  @Override
058  protected void doInit() throws ResourceException {
059    super.doInit();
060    this.sensorId = getQuery().getValues(Labels.SENSOR);
061    this.timestamp = getQuery().getValues(Labels.TIMESTAMP);
062    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
063    this.dataType = getQuery().getValues(Labels.VALUE_TYPE);
064  }
065
066
067  /**
068   * retrieve the hourly depository value list for a sensor.
069   *
070   * @return measurement list.
071   */
072  public InterpolatedValueList doRetrieve() {
073    getLogger().log(
074        Level.INFO,
075        "GET /wattdepot/{" + orgId + "}/" + Labels.DEPOSITORY + "/{" + depositoryId + "}/" + Labels.DAY_HOURLY + "/"
076            + Labels.VALUES + "/?" + Labels.SENSOR + "={" + sensorId + "}&" + Labels.TIMESTAMP + "={"
077            + timestamp + "}&" + Labels.VALUE_TYPE + "={" + dataType + "}");
078    if (isInRole(orgId)) {
079      if (timestamp != null && dataType != null) {
080        InterpolatedValueList ret = new InterpolatedValueList();
081        try {
082          Depository depository = depot.getDepository(depositoryId, orgId, true);
083          if (depository != null) {
084            XMLGregorianCalendar startTime = DateConvert.parseCalString(timestamp);
085            startTime.setTime(0, 0, 0, 0); // beginning of the day
086            XMLGregorianCalendar endTime = DateConvert.parseCalString(timestamp);
087            endTime = Tstamp.incrementDays(endTime, 1);
088            endTime.setTime(0, 0, 0, 0); // beginning of the next day.
089            List<XMLGregorianCalendar> times = Tstamp.getTimestampList(startTime, endTime, 60);
090            if (times != null) {
091              for (int i = 1; i < times.size(); i++) {
092                XMLGregorianCalendar begin = times.get(i - 1);
093                Date beginDate = begin.toGregorianCalendar().getTime();
094                XMLGregorianCalendar end = times.get(i);
095                Date endDate = end.toGregorianCalendar().getTime();
096                InterpolatedValue interpolatedValue = new InterpolatedValue(sensorId, 0.0, depository.getMeasurementType(), beginDate, endDate);
097                Double val = 0.0;
098                interpolatedValue.addDefinedSensor(sensorId);
099                if (dataType.equals("point")) {  // need to calculate the average value for the hourly intervals
100                  List<Measurement> measurements = depot.getMeasurements(depositoryId, orgId, sensorId, beginDate, endDate, false);
101                  if (measurements.size() > 0) {
102                    for (Measurement m : measurements) {
103                      val += m.getValue();
104                    }
105                    val = val / measurements.size();
106                    interpolatedValue.addReportingSensor(sensorId);
107                  }
108                  else {
109                    val = Double.NaN;
110                  }
111                }
112                else {  // calculate the difference
113                  try {
114                    val = depot.getValue(depositoryId, orgId, sensorId, beginDate, endDate, false);
115                    interpolatedValue.addReportingSensor(sensorId);
116                  }
117                  catch (NoMeasurementException nme) {
118                    val = Double.NaN;
119                  }
120                }
121                interpolatedValue.setValue(val);
122                ret.getInterpolatedValues().add(interpolatedValue);
123              }
124            }
125          }
126        }
127        catch (ParseException e) {
128          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
129          return null;
130        }
131        catch (DatatypeConfigurationException e) {
132          setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
133          return null;
134        }
135        catch (IdNotFoundException e) {
136          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
137          return null;
138        }
139        return ret;
140      }
141      else {
142        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Missing start and/or end times or value type.");
143        return null;
144      }
145    }
146    else {
147      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
148      return null;
149    }
150  }
151}
