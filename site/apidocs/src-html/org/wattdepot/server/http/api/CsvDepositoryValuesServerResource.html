001package org.wattdepot.server.http.api;
002
003import org.restlet.data.Disposition;
004import org.restlet.data.MediaType;
005import org.restlet.representation.StringRepresentation;
006import org.wattdepot.common.domainmodel.InterpolatedValue;
007import org.wattdepot.common.domainmodel.InterpolatedValueList;
008import org.wattdepot.common.http.api.CsvDepositoryValuesResource;
009
010import java.text.SimpleDateFormat;
011
012/**
013 * CsvDepositoryValuesServerResource - ServerResouce that handles the GET
014 * /wattdepot/{org-id}/depository/{depository-id}/values/csv/ response.
015 *
016 * @author Brian Fr√∏lund
017 */
018public class CsvDepositoryValuesServerResource extends DepositoryValuesServer implements
019    CsvDepositoryValuesResource {
020
021  /*
022  * (non-Javadoc)
023  *
024  * @see org.wattdepot.restlet.CsvDepositoryValuesResource#retrieve()
025  */
026  @Override
027  public StringRepresentation retrieve() {
028    InterpolatedValueList mList = doRetrieve();
029
030    //RFC4180 CSV http://tools.ietf.org/html/rfc4180
031    SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSXXX");
032    StringBuilder sb = new StringBuilder();
033    sb.append("Start,Value\r\n");
034    for (InterpolatedValue interpolatedValue : mList.getInterpolatedValues()) {
035      sb.append(sdf.format(interpolatedValue.getStart()));
036      sb.append(",");
037      sb.append(interpolatedValue.getValue());
038      sb.append("\r\n");
039    }
040    StringRepresentation response = new StringRepresentation(sb.toString(), MediaType.TEXT_CSV);
041    response.setDisposition(new Disposition(Disposition.TYPE_ATTACHMENT));
042    response.getDisposition().setFilename(String.format("%s_%s_%s.csv", depositoryId, sensorId, dataType));
043    return response;
044  }
045}
