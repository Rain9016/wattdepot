001package org.wattdepot.server.http.api;
002
003import org.restlet.data.Status;
004import org.restlet.resource.ResourceException;
005import org.wattdepot.common.domainmodel.CollectorProcessDefinition;
006import org.wattdepot.common.domainmodel.Depository;
007import org.wattdepot.common.domainmodel.InterpolatedValue;
008import org.wattdepot.common.domainmodel.Labels;
009import org.wattdepot.common.domainmodel.Sensor;
010import org.wattdepot.common.domainmodel.SensorGroup;
011import org.wattdepot.common.exception.IdNotFoundException;
012import org.wattdepot.common.exception.MisMatchedOwnerException;
013import org.wattdepot.common.exception.NoMeasurementException;
014import org.wattdepot.common.http.api.DepositoryLatestValueResource;
015
016import java.util.Date;
017import java.util.List;
018import java.util.logging.Level;
019
020/**
021 * DepositoryLatestValueServerResource - Returns the latest value for the sensor or sensor group stored in the
022 * depository as an InterpolatedValue with missing sensors.
023 *
024 * @author Cam Moore
025 */
026public class DepositoryLatestValueServerResource extends WattDepotServerResource implements DepositoryLatestValueResource {
027  private String depositoryId;
028  private String sensorId;
029  private String window;
030
031  /*
032 * (non-Javadoc)
033 *
034 * @see org.restlet.resource.Resource#doInit()
035 */
036  @Override
037  protected void doInit() throws ResourceException {
038    super.doInit();
039    this.sensorId = getQuery().getValues(Labels.SENSOR);
040    this.window = getQuery().getValues(Labels.WINDOW);
041    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
042  }
043
044
045  @Override
046  public InterpolatedValue retrieve() {
047    getLogger().log(
048        Level.INFO,
049        "GET /wattdepot/{" + orgId + "}/" + Labels.DEPOSITORY + "/{" + depositoryId + "}/"
050            + Labels.LATEST + "/" + Labels.VALUE + "/?" + Labels.SENSOR + "={" + sensorId + "}&"
051            + Labels.WINDOW + "={" + window + "}");
052    int width = 5;
053    if (window != null && !window.equals("")) {
054      try {
055        width = Integer.parseInt(window);
056      }
057      catch (NumberFormatException nfe) {
058        width = 5;
059      }
060    }
061    if (isInRole(orgId)) {
062      try {
063        Depository depository = depot.getDepository(depositoryId, orgId, true);
064        Sensor sensor = depot.getSensor(sensorId, orgId, false);
065        InterpolatedValue value = new InterpolatedValue(sensorId, 0.0, depository.getMeasurementType(), new Date());
066        if (sensor != null) {
067          value.addDefinedSensor(sensorId);
068          CollectorProcessDefinition cpd = findCPD(depositoryId, sensorId, orgId);
069          try {
070            if (cpd != null) {
071              return depot.getLatestMeasuredValue(depositoryId, orgId, sensorId, width * cpd.getPollingInterval(), false);
072            }
073            else {
074              return depot.getLatestMeasuredValue(depositoryId, orgId, sensorId, false);
075            }
076          }
077          catch (NoMeasurementException e) {
078            return value;
079          }
080        }
081        else {
082          SensorGroup group = depot.getSensorGroup(sensorId, orgId, false);
083          if (group != null) {
084            for (String s : group.getSensors()) {
085              sensor = depot.getSensor(s, orgId, false);
086              value.addDefinedSensor(s);
087              if (sensor != null) {
088                CollectorProcessDefinition cpd = findCPD(depositoryId, s, orgId);
089                try {
090                  InterpolatedValue latest;
091                  if (cpd != null) {
092                    latest = depot.getLatestMeasuredValue(depositoryId, orgId, s, width * cpd.getPollingInterval(), false);
093                  }
094                  else {
095                    latest = depot.getLatestMeasuredValue(depositoryId, orgId, s, false);
096                  }
097                  value.setValue(value.getValue() + latest.getValue());
098                  value.addReportingSensor(s);
099                  value.setStart(latest.getStart());
100                  value.setEnd(latest.getEnd());
101                }
102                catch (NoMeasurementException e) { //NOPMD
103
104                }
105              }
106            }
107            return value;
108          }
109        }
110      }
111      catch (IdNotFoundException e) {
112        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not a Depository id.");
113        return null;
114      }
115      catch (MisMatchedOwnerException e) {
116        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not in organization " + orgId + ".");
117        return null;
118      }
119    }
120    return null;
121  }
122
123  /**
124   * @param depositoryId The depository id.
125   * @param sensorId     The sensor id.
126   * @param orgId        The orgainzation id.
127   * @return The CollectorProcessDefinition for the depository and sensor, or null if not defined.
128   */
129  private CollectorProcessDefinition findCPD(String depositoryId, String sensorId, String orgId) {
130    try {
131      List<CollectorProcessDefinition> cpds = depot.getCollectorProcessDefinitions(orgId, false);
132      for (CollectorProcessDefinition cpd : cpds) {
133        if (cpd.getDepositoryId().equals(depositoryId) && cpd.getSensorId().equals(sensorId)) {
134          return cpd;
135        }
136      }
137    }
138    catch (IdNotFoundException e) {
139      e.printStackTrace();
140    }
141    return null;
142  }
143}
