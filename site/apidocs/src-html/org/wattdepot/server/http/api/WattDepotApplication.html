127  /**
128   * Creates a root Router to dispatch call to server resources.
129   * 
130   * @return the inbound root.
131   */
132  @Override
133  public Restlet createInboundRoot() {
134    Router router = new Router(getContext());
135
136    // Use CLAP (ClassLoader Access Protocol) to access SPA directory from
137    // filesystem or JAR
138    Directory directory = new Directory(getContext(), "clap://dist/");
139    directory.setListingAllowed(true);
140    router.attach("/webroot/", directory);
141
142    router.attach(API.BASE_URI, IsAliveServerResource.class);
143    // router.attach("/wattdepot/", LoginPageServerResource.class);
144    // router.attach("/wattdepot/login/", LoginServerResource.class);
145    // Group administration UI.
146    router.attach(API.ADMIN_URI, AdminServerResource.class);
147    router.attach(API.ORGANIZATION_SUMMARY_URI, OrgSummaryServerResource.class);
148    router.attach(API.ORGANIZATION_VISUALIZE_URI, OrgVisualizerServerResource.class);
149    // CollectorProcessDefinition
150    router.attach(API.COLLECTOR_PROCESS_DEFINITION_PUT_URI,
151        CollectorProcessDefinitionPutServerResource.class);
152    router.attach(API.COLLECTOR_PROCESS_DEFINITION_URI,
153        CollectorProcessDefinitionServerResource.class);
154    router.attach(API.COLLECTOR_PROCESS_DEFINITIONS_URI,
155        CollectorProcessDefinitionsServerResource.class);
156    // Depositories and Measurements
157    router.attach(API.DEPOSITORY_PUT_URI, DepositoryPutServerResource.class);
158    router.attach(API.DEPOSITORY_URI, DepositoryServerResource.class);
159    router.attach(API.DEPOSITORIES_URI, DepositoriesServerResource.class);
160    router.attach(API.DEPOSITORY_SENSORS_URI, DepositorySensorsServerResource.class);
161    router.attach(API.DEPOSITORY_SENSOR_SUMMARY_URI, DepositorySensorSummaryServerResource.class);
162    router.attach(API.MEASUREMENT_PRUNING_DEFINITION_PUT_URI, MeasurementPruningDefinitionPutServerResource.class);
163    router.attach(API.MEASUREMENT_PRUNING_DEFINITION_URI, MeasurementPruningDefinitionServerResource.class);
164    router.attach(API.MEASUREMENT_PRUNING_DEFINITIONS_URI, MeasurementPruningDefinitionsServerResource.class);
165    router.attach(API.MEASUREMENT_PUT_URI, DepositoryMeasurementPutServerResource.class);
166    router.attach(API.MEASUREMENT_URI, DepositoryMeasurementServerResource.class);
167    router.attach(API.MEASUREMENTS_URI, DepositoryMeasurementsServerResource.class);
168    router.attach(API.MEASUREMENTS_GVIZ_URI, GvizDepositoryMeasurementsServerResource.class);
169    router.attach(API.VALUE_URI, DepositoryValueServerResource.class);
170    router.attach(API.VALUE_GVIZ_URI, GvizDepositoryValueServerResource.class);
171    router.attach(API.VALUES_URI, DepositoryValuesServerResource.class);
172    router.attach(API.VALUES_GVIZ_URI, GvizDepositoryValuesServerResource.class);
173    router.attach(API.VALUES_AVERAGE_URI, DepositoryAverageValuesServerResource.class);
174    router.attach(API.VALUES_AVERAGE_GVIZ_URI, GvizDepositoryAverageValuesServerResource.class);
175    router.attach(API.VALUES_MAXIMUM_URI, DepositoryMaximumValuesServerResource.class);
176    router.attach(API.VALUES_MINIMUM_URI, DepositoryMinimumValuesServerResource.class);
177    // MeasurementTypes
178    router.attach(API.MEASUREMENT_TYPE_PUT_URI, MeasurementTypePutServerResource.class);
179    router.attach(API.MEASUREMENT_TYPE_URI, MeasurementTypeServerResource.class);
180    router.attach(API.MEASUREMENT_TYPES_URI, MeasurementTypesServerResource.class);
181    // Sensors
182    router.attach(API.SENSOR_PUT_URI, SensorPutServerResource.class);
183    router.attach(API.SENSOR_URI, SensorServerResource.class);
184    router.attach(API.SENSORS_URI, SensorsServerResource.class);
185    // Sensor Measurement Summary
186    router.attach(API.SENSOR_MEASUREMENT_SUMMARY_URI, SensorMeasurementServerResource.class);
187    // SensorGroups
188    router.attach(API.SENSOR_GROUP_PUT_URI, SensorGroupPutServerResource.class);
189    router.attach(API.SENSOR_GROUP_URI, SensorGroupServerResource.class);
190    router.attach(API.SENSOR_GROUPS_URI, SensorGroupsServerResource.class);
191    // SensorModels
192    router.attach(API.SENSOR_MODEL_PUT_URI, SensorModelPutServerResource.class);
193    router.attach(API.SENSOR_MODEL_URI, SensorModelServerResource.class);
194    router.attach(API.SENSOR_MODELS_URI, SensorModelsServerResource.class);
195    // Users, UserGroups, and UserPasswords
196    router.attach(API.USER_PUT_URI, UserInfoPutServerResource.class);
197    router.attach(API.USER_URI, UserInfoServerResource.class);
198    router.attach(API.USERS_URI, UserInfosServerResource.class);
199    router.attach(API.USER_PASSWORD_URI, UserPasswordServerResource.class);
200    router.attach(API.ORGANIZATION_PUT_URI, OrganizationPutServerResource.class);
201    router.attach(API.ORGANIZATION_URI, OrganizationServerResource.class);
202    router.attach(API.ORGANIZATIONS_URI, OrganizationsServerResource.class);
203
204    ChallengeAuthenticator authenticator = new ChallengeAuthenticator(getContext(),
205        ChallengeScheme.HTTP_BASIC, "WattDepot Realm");
206    authenticator.setNext(router);
207    return authenticator;
208
209    // return router;
210  }
