043/**
044 * DepositoryDescriptiveStatsServer - Base class for handling historical values HTTP requests.
045 *
046 * @author Cam Moore
047 */
048public class DepositoryDescriptiveStatsServer extends WattDepotServerResource {
049  private String depositoryId;
050  private String hourlyDailyChoice;
051  private String sensorId;
052  private String timestamp;
053  private String valueType;
054  private Integer samples;
055
056  /**
057   * @return the DescriptiveStats for the given request.
058   */
059  public DescriptiveStats doRetrieve() {
060    getLogger().log(
061        Level.INFO,
062        "GET /wattdepot/{" + orgId + "}/" + Labels.DEPOSITORY + "/{" + depositoryId + "}/" + Labels.DESCRIPTIVE_STATS + "/"
063            + hourlyDailyChoice + "/?" + Labels.SENSOR + "={" + sensorId + "}&" + Labels.TIMESTAMP + "={"
064            + timestamp + "}&" + Labels.VALUE_TYPE + "={" + valueType + "}&" + Labels.SAMPLES + "={" + samples + "}");
065    if (isInRole(orgId)) {
066      DescriptiveStats ret = new DescriptiveStats();
067      try {
068        Depository depository = depot.getDepository(depositoryId, orgId, true);
069        if (depository != null) {
070          XMLGregorianCalendar time = DateConvert.parseCalString(timestamp);
071          XMLGregorianCalendar begin = DateConvert.getBeginning(time, hourlyDailyChoice);
072          XMLGregorianCalendar end = DateConvert.getEnding(time, hourlyDailyChoice);
073          Double minimum = Double.MAX_VALUE; // for this time period
074          Double maximum = Double.MIN_VALUE; // for this time period
075          Double average = 0.0;
076          Double upQuartile = 0.0;
077          Double lowQuartile = 0.0;
078          DescriptiveStatistics statistics = new DescriptiveStatistics();
079          for (int i = 0; i < samples; i++) {
080            begin = Tstamp.incrementDays(begin, -7); // back it up a week
081            end = Tstamp.incrementDays(end, -7);
082            Sensor sensor = depot.getSensor(sensorId, orgId, false);
083            if (valueType.equals(Labels.POINT)) { // since it is point data must calculate the average of the measurements
084              if (sensor != null) {
085                ret.addDefinedSensor(sensorId);
086                statistics = updateStatistics(depositoryId, sensorId, begin, end, statistics);
087                ret.addReportingSensor(sensorId);
088                if (statistics.getMin() < minimum) {
089                  minimum = statistics.getMin();
090                }
091                if (statistics.getMax() > maximum) {
092                  maximum = statistics.getMax();
093                }
094                average += statistics.getMean();
095                upQuartile += statistics.getPercentile(75.0);
096                lowQuartile += statistics.getPercentile(25.0);
097              }
098              else {
099                SensorGroup group = depot.getSensorGroup(sensorId, orgId, false);
100                if (group != null) {
101                  Double groupMin = 0.0;
102                  Double groupMax = 0.0;
103                  Double groupAve = 0.0;
104                  Double lowerQuartile = 0.0;
105                  Double upperQuartile = 0.0;
106                  for (String s : group.getSensors()) {
107                    ret.addDefinedSensor(s);
108                    try {
109                      statistics = getStatistics(depositoryId, s, begin, end);
110                      ret.addReportingSensor(s);
111                      groupMin += statistics.getMin();
112                      groupMax += statistics.getMax();
113                      groupAve += statistics.getMean();
114                      lowerQuartile += statistics.getPercentile(0.25);
115                      upperQuartile += statistics.getPercentile(0.75);
116                    }
117                    catch (NoMeasurementException nme) {
118                      // skip this sensor
119                      ret.addMissingSensor(s);
120                    }
121                  }
122                  if (maximum < groupMax) {
123                    maximum = groupMax;
124                  }
125                  if (minimum > groupMin) {
126                    minimum = groupMin;
127                  }
128                  average += groupAve;
129                  upQuartile += upperQuartile;
130                  lowQuartile += lowerQuartile;
131                }
132              }
133            }
134            else { // difference values
135              if (sensor != null) {
136                ret.addDefinedSensor(sensorId);
137                statistics.addValue(depot.getValue(depositoryId, orgId, sensorId, DateConvert.convertXMLCal(begin), DateConvert.convertXMLCal(end), false));
138                ret.addReportingSensor(sensorId);
139              }
140              else {
141                SensorGroup group = depot.getSensorGroup(sensorId, orgId, false);
142                if (group != null) {
143                  Double d = 0.0;
144                  for (String s : group.getSensors()) {
145                    try {
146                      ret.addDefinedSensor(s);
147                      d += depot.getValue(depositoryId, orgId, s, DateConvert.convertXMLCal(begin), DateConvert.convertXMLCal(end), false);
148                      ret.addReportingSensor(s);
149                    }
150                    catch (NoMeasurementException nme) {
151                      // just skip this sensor.
152                      ret.addMissingSensor(s);
153                    }
154                  }
155                  statistics.addValue(d);
156                }
157              }
158            }
159          }
160          if (valueType.equals(Labels.POINT)) {
161            average /= samples;
162            lowQuartile /= samples;
163            upQuartile /= samples;
164          }
165          else {
166            average = statistics.getMean();
167            minimum = statistics.getMin();
168            maximum = statistics.getMax();
169            lowQuartile = statistics.getPercentile(25.0);
170            upQuartile = statistics.getPercentile(75.0);
171          }
172          ret.setDepositoryId(depositoryId);
173          ret.setSensorId(sensorId);
174          ret.setAverage(average);
175          ret.setMaximum(maximum);
176          ret.setMinimum(minimum);
177          ret.setLowerQuartile(lowQuartile);
178          ret.setUpperQuartile(upQuartile);
179          ret.setNumSamples(samples);
180          ret.setWindowWidth(hourlyDailyChoice);
181          ret.setValueType(valueType);
182          ret.setTimestamp(DateConvert.parseCalStringToDate(timestamp));
183          return ret;
184        }
185      }
186      catch (IdNotFoundException e) {
187        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
188        return null;
189      }
190      catch (DatatypeConfigurationException e) {
191        setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
192        return null;
193      }
194      catch (ParseException e) {
195        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
196        return null;
197      }
198      catch (MisMatchedOwnerException e) {
199        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
200        return null;
201      }
202      catch (NoMeasurementException e) {
203        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
204        return null;
205      }
206      return null;
207    }
208    return null;
209  }
210
211  /*
212   * (non-Javadoc)
213   *
214   * @see org.restlet.resource.Resource#doInit()
215   */
216  @Override
217  protected void doInit() throws ResourceException {
218    super.doInit();
219    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
220    this.hourlyDailyChoice = getAttribute(Labels.HOURLY_DAILY);
221    this.sensorId = getQuery().getValues(Labels.SENSOR);
222    this.timestamp = getQuery().getValues(Labels.TIMESTAMP);
223    this.valueType = getQuery().getValues(Labels.VALUE_TYPE);
224    this.samples = Integer.parseInt(getQuery().getValues(Labels.SAMPLES));
225  }
226
227  /**
228   * @param depositoryId the depository id.
229   * @param sensorId     the sensor id.
230   * @param begin        the begin time.
231   * @param end          the end time.
232   * @return DescriptiveStatistics of the measurements for the peroiod.
233   * @throws IdNotFoundException    if there is a problem with the sensorId.
234   * @throws NoMeasurementException if there are no measurements during the period.
235   */
236  private DescriptiveStatistics getStatistics(String depositoryId, String sensorId, XMLGregorianCalendar begin, XMLGregorianCalendar end) throws IdNotFoundException, NoMeasurementException {
237    List<Measurement> measurements = depot.getMeasurements(depositoryId, orgId, sensorId, DateConvert.convertXMLCal(begin), DateConvert.convertXMLCal(end), false);
238    if (measurements == null || measurements.size() == 0) {
239      throw new NoMeasurementException("No measurements for " + sensorId + " from " + begin.toString() + " to " + end.toString());
240    }
241    DescriptiveStatistics ret = new DescriptiveStatistics();
242    for (Measurement m : measurements) {
243      ret.addValue(m.getValue());
244    }
245    return ret;
246  }
247
248  /**
249   * @param depositoryId the depository id.
250   * @param sensorId     the sensor id.
251   * @param begin        the begin time.
252   * @param end          the end time.
253   * @param statistics   the statistics object to update.
254   * @return the updated object.
255   * @throws IdNotFoundException if there is a problem with the sensorId.
256   */
257  private DescriptiveStatistics updateStatistics(String depositoryId, String sensorId, XMLGregorianCalendar begin, XMLGregorianCalendar end, DescriptiveStatistics statistics) throws IdNotFoundException {
258    List<Measurement> measurements = depot.getMeasurements(depositoryId, orgId, sensorId, DateConvert.convertXMLCal(begin), DateConvert.convertXMLCal(end), false);
259    for (Measurement m : measurements) {
260      statistics.addValue(m.getValue());
261    }
262    return statistics;
263  }
