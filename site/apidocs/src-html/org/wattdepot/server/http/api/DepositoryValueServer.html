042/**
043 * DepositoryValueServerResource - ServerResouce that handles the GET
044 * /wattdepot/{org-id}/depository/{depository-id}/value/ response.
045 * 
046 * @author Cam Moore
047 * 
048 */
049public class DepositoryValueServer extends WattDepotServerResource {
050  private String depositoryId;
051  private String sensorId;
052  private String start;
053  private String end;
054  private String timestamp;
055  private String latest;
056  private String earliest;
057  private String gapSeconds;
058  
059  private static DescriptiveStatistics averageGetTime = new DescriptiveStatistics();
060
061  /*
062   * (non-Javadoc)
063   * 
064   * @see org.restlet.resource.Resource#doInit()
065   */
066  @Override
067  protected void doInit() throws ResourceException {
068    super.doInit();
069    this.sensorId = getQuery().getValues(Labels.SENSOR);
070    this.start = getQuery().getValues(Labels.START);
071    this.end = getQuery().getValues(Labels.END);
072    this.timestamp = getQuery().getValues(Labels.TIMESTAMP);
073    this.latest = getQuery().getValues(Labels.LATEST);
074    this.earliest = getQuery().getValues(Labels.EARLIEST);
075    this.orgId = getAttribute(Labels.ORGANIZATION_ID);
076    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
077    this.gapSeconds = getQuery().getValues(Labels.GAP);
078  }
079
080  /**
081   * retrieve the depository value for a sensor.
082   * 
083   * @return measured value.
084   */
085  public InterpolatedValue doRetrieve() {
086    getLogger().log(
087        Level.INFO,
088        "GET /wattdepot/{" + orgId + "}/depository/{" + depositoryId + "}/value/?sensor={"
089            + sensorId + "}&start={" + start + "}&end={" + end + "}&timestamp={" + timestamp
090            + "}&latest={" + latest + "}&earliest={" + earliest + "}&gap={" + gapSeconds + "}");
091    Long startTime = null;
092    if (depot.getServerProperties().get(ServerProperties.SERVER_TIMING_KEY)
093        .equals(ServerProperties.TRUE)) {
094      startTime = System.nanoTime();
095    }
096    if (isInRole(orgId)) {
097      try {
098        Depository deposit = depot.getDepository(depositoryId, orgId, true);
099        if (deposit != null) {
100          try {
101            depot.getSensor(sensorId, orgId, true);
102            InterpolatedValue val =  calculateValue(sensorId, start, end, timestamp, earliest, latest);
103            if (depot.getServerProperties().get(ServerProperties.SERVER_TIMING_KEY)
104                .equals(ServerProperties.TRUE)) {
105              Long endTime = System.nanoTime();
106              Long diff = endTime - startTime;
107              averageGetTime.addValue(diff / 1E9);
108              getLogger().log(
109                  Level.SEVERE,
110                  "GET /wattdepot/{" + orgId + "}/depository/{" + depositoryId
111                      + "}/value/?sensor={" + sensorId + "}&start={" + start + "}&end={" + end
112                      + "}&timestamp={" + timestamp + "}&latest={" + latest + "}&earliest={"
113                      + earliest + "}&gap={" + gapSeconds + "} took "
114                      + (diff / 1E9) + " seconds. Running average is " + averageGetTime.getMean());
115            }            
116            return val;
117          }
118          catch (IdNotFoundException inf) {
119            try {
120              SensorGroup group = depot.getSensorGroup(sensorId, orgId, true);
121              InterpolatedValue val = null;
122              // this wont work for earliest or latest.
123              if (earliest != null) {
124                // find the last 'earliest' time
125                Date time = null;
126                for (String s : group.getSensors()) {
127                  InterpolatedValue v = calculateValue(s, start, end, timestamp, earliest, latest);
128                  if (time == null) {
129                    time = v.getStart();
130                  }
131                  else if (time.before(v.getStart())) {
132                    time = v.getStart();
133                  }
134                }
135                // have the time get the value at time
136                for (String s : group.getSensors()) {
137                  if (val == null) {
138                    val = calculateValue(s, null, null, DateConvert.convertDate(time).toString(),
139                        null, null);
140                  }
141                  else {
142                    val.setValue(val.getValue()
143                        + calculateValue(s, null, null, DateConvert.convertDate(time).toString(),
144                            null, null).getValue());
145                  }
146                }
147              }
148              else if (latest != null) {
149                // find the first 'latest' time
150                Date time = null;
151                for (String s : group.getSensors()) {
152                  InterpolatedValue v = calculateValue(s, start, end, timestamp, earliest, latest);
153                  if (time == null) {
154                    time = v.getEnd();
155                  }
156                  else if (time.after(v.getEnd())) {
157                    time = v.getEnd();
158                  }
159                }
160                // have the time get the value at time
161                for (String s : group.getSensors()) {
162                  if (val == null) {
163                    val = calculateValue(s, null, null, DateConvert.convertDate(time).toString(),
164                        null, null);
165                  }
166                  else {
167                    val.setValue(val.getValue()
168                        + calculateValue(s, null, null, DateConvert.convertDate(time).toString(),
169                            null, null).getValue());
170                  }
171                }
172              }
173              else {
174                for (String s : group.getSensors()) {
175                  if (val == null) {
176                    val = calculateValue(s, start, end, timestamp, earliest, latest);
177                  }
178                  else {
179                    val.setValue(val.getValue()
180                        + calculateValue(s, start, end, timestamp, earliest, latest).getValue());
181                  }
182                }
183              }
184              if (depot.getServerProperties().get(ServerProperties.SERVER_TIMING_KEY)
185                  .equals(ServerProperties.TRUE)) {
186                Long endTime = System.nanoTime();
187                getLogger().log(
188                    Level.SEVERE,
189                    "GET /wattdepot/{" + orgId + "}/depository/{" + depositoryId
190                        + "}/value/?sensor={" + sensorId + "}&start={" + start + "}&end={" + end
191                        + "}&timestamp={" + timestamp + "}&latest={" + latest + "}&earliest={"
192                        + earliest + "}&gap={" + gapSeconds + "} took "
193                        + ((endTime - startTime) / 1E9) + " seconds.");
194              }
195
196              return val;
197            }
198            catch (IdNotFoundException inf1) {
199              setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Could not find sensor " + sensorId);
200            }
201          }
202        }
203        else {
204          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Could not find depository " + depositoryId);
205        }
206      }
207      catch (MisMatchedOwnerException e) {
208        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
209      }
210      catch (NoMeasurementException e1) {
211        setStatus(Status.SERVER_ERROR_INTERNAL, e1.getMessage());
212      }
213      catch (NumberFormatException e) {
214        setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
215      }
216      catch (MeasurementGapException e) {
217        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
218      }
219      catch (ParseException e) {
220        setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
221      }
222      catch (DatatypeConfigurationException e) {
223        setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
224      }
225      catch (IdNotFoundException e) {
226        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
227      }
228      return null;
229    }
230    else {
231      setStatus(Status.CLIENT_ERROR_FORBIDDEN, "Bad credentials.");
232      return null;
233    }
234  }

236  /**
237   * @param sensorId the Sensor's id.
238   * @param start The start time. Optional, must have end if not null.
239   * @param end The end time. Optional, must exist if start not null.
240   * @param timestamp The time of the value. Optional, must be null if start or
241   *        end.
242   * @param earliest if not null will get earliest value. Optional, must be null
243   *        if start or end or latest or timestamp.
244   * @param latest if not null will get latest value. Optional, must be null if
245   *        start or end or earliest or timestamp.
246   * @return The interpolated value for the sensorId and time(s).
247   * @throws IdNotFoundException if sensorId is not defined.
248   * @throws MisMatchedOwnerException if sensorId is not in orgId.
249   * @throws NoMeasurementException if there aren't measurements around the
250   *         time(s).
251   * @throws ParseException if the times are not valid Date strings.
252   * @throws DatatypeConfigurationException if there is a server problem.
253   * @throws NumberFormatException if there is a problem.
254   * @throws MeasurementGapException if the measurements are too far apart.
255   */
256  private InterpolatedValue calculateValue(String sensorId, String start, String end,
257      String timestamp, String earliest, String latest) throws IdNotFoundException,
258      MisMatchedOwnerException, NoMeasurementException, ParseException,
259      DatatypeConfigurationException, NumberFormatException, MeasurementGapException {
260    Depository deposit = depot.getDepository(depositoryId, orgId, true);
261    Sensor sensor = depot.getSensor(sensorId, orgId, true);
262    Double value = null;
263    Date startDate = null;
264    Date endDate = null;
265    Date time = null;
266    InterpolatedValue val = null;
267
268    if (earliest != null) {
269      val = depot.getEarliestMeasuredValue(depositoryId, orgId, sensorId, true);
270      val.addDefinedSensor(sensorId);
271      val.addReportingSensor(sensorId);
272      return val;
273    }
274    else if (latest != null) {
275      val = depot.getLatestMeasuredValue(depositoryId, orgId, sensorId, true);
276      val.addDefinedSensor(sensorId);
277      val.addReportingSensor(sensorId);
278      return val;
279    }
280    else if (timestamp != null) {
281      time = DateConvert.parseCalStringToDate(timestamp);
282      if (gapSeconds != null) {
283        value = depot.getValue(depositoryId, orgId, sensor.getId(), time,
284            Long.parseLong(gapSeconds), true);
285      }
286      else {
287        value = depot.getValue(depositoryId, orgId, sensor.getId(), time, true);
288      }
289      val = new InterpolatedValue(sensorId, value, deposit.getMeasurementType(), time);
290    }
291    else if (start != null && end != null) {
292      startDate = DateConvert.parseCalStringToDate(start);
293      endDate = DateConvert.parseCalStringToDate(end);
294      if (gapSeconds != null) {
295        value = depot.getValue(depositoryId, orgId, sensor.getId(), startDate, endDate,
296            Long.parseLong(gapSeconds), true);
297      }
298      else {
299        value = depot.getValue(depositoryId, orgId, sensor.getId(), startDate, endDate, true);
300      }
301      val = new InterpolatedValue(sensorId, value, deposit.getMeasurementType(), startDate, endDate);
302    }
303    if (val != null) {
304      val.addDefinedSensor(sensorId);
305      if (value != null) {
306        val.addReportingSensor(sensorId);
307      }
308    }
309    return val;
310  }
