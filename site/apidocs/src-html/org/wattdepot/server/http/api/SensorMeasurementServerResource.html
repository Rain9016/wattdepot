019package org.wattdepot.server.http.api;
020
021import java.text.ParseException;
022import java.util.Date;
023import java.util.logging.Level;
024
025import javax.xml.datatype.DatatypeConfigurationException;
026
027import org.restlet.data.Status;
028import org.restlet.resource.ResourceException;
029import org.wattdepot.common.domainmodel.Depository;
030import org.wattdepot.common.domainmodel.Labels;
031import org.wattdepot.common.domainmodel.Measurement;
032import org.wattdepot.common.domainmodel.MeasurementList;
033import org.wattdepot.common.domainmodel.Sensor;
034import org.wattdepot.common.domainmodel.SensorGroup;
035import org.wattdepot.common.domainmodel.SensorMeasurementSummary;
036import org.wattdepot.common.exception.IdNotFoundException;
037import org.wattdepot.common.exception.MisMatchedOwnerException;
038import org.wattdepot.common.http.api.SensorMeasurementResource;
039import org.wattdepot.common.util.DateConvert;
040
041/**
042 * SensorMeasurementServerResource - Handles the SensorMeasurementSummary HTTP
043 * API (
044 * "/wattdepot/{org-id}/depository/{depository-id}/?sensor={sensor-id}&amp;start={start}&amp;end=
045 * { e n d }).
046 * 
047 * @author Cam Moore
048 * 
049 */
050public class SensorMeasurementServerResource extends WattDepotServerResource implements
051    SensorMeasurementResource {
052  /** The depository id. */
053  private String depositoryId;
054  /** The sensor id. */
055  private String sensorId;
056  /** The start time. */
057  private String start;
058  /** The end time. */
059  private String end;
060
061  /*
062   * (non-Javadoc)
063   * 
064   * @see org.restlet.resource.Resource#doInit()
065   */
066  @Override
067  protected void doInit() throws ResourceException {
068    super.doInit();
069    this.sensorId = getQuery().getValues(Labels.SENSOR);
070    this.start = getQuery().getValues(Labels.START);
071    this.end = getQuery().getValues(Labels.END);
072    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
073
074  }
075
076  /*
077   * (non-Javadoc)
078   * 
079   * @see org.wattdepot.common.http.api.SensorMeasurementResource#retrieve()
080   */
081  @Override
082  public SensorMeasurementSummary retrieve() {
083    getLogger().log(
084        Level.INFO,
085        "GET /wattdepot/{" + orgId + "}/depository/{" + depositoryId + "}/summary/?sensor={"
086            + sensorId + "}&start={" + start + "}&end={" + end + "}");
087    SensorMeasurementSummary ret = null;
088    if (isInRole(orgId)) {
089      MeasurementList meas = getMeasurements();
090      if (meas != null) {
091        try {
092          ret = new SensorMeasurementSummary(sensorId, depositoryId,
093              DateConvert.parseCalStringToDate(start), DateConvert.parseCalStringToDate(end), meas
094                  .getMeasurements().size());
095        }
096        catch (ParseException e) {
097          e.printStackTrace();
098          setStatus(Status.SERVER_ERROR_INTERNAL);
099        }
100        catch (DatatypeConfigurationException e) {
101          e.printStackTrace();
102          setStatus(Status.SERVER_ERROR_INTERNAL);
103        }
104      }
105    }
106    return ret;
107  }
108
109  /**
110   * @return The measurements between start and end for the given sensor and
111   *         depository.
112   */
113  private MeasurementList getMeasurements() {
114    if (start != null && end != null) {
115      MeasurementList ret = new MeasurementList();
116      try {
117        Depository depository = depot.getDepository(depositoryId, orgId, true);
118        if (depository != null) {
119          Date startDate = DateConvert.parseCalStringToDate(start);
120          Date endDate = DateConvert.parseCalStringToDate(end);
121          if (startDate != null && endDate != null) {
122            try {
123              Sensor sensor = depot.getSensor(sensorId, orgId, true);
124              for (Measurement meas : depot.getMeasurements(depositoryId, orgId, sensor.getId(),
125                  startDate, endDate, true)) {
126                ret.getMeasurements().add(meas);
127              }
128            }
129            catch (IdNotFoundException nf) {
130              try {
131                SensorGroup group = depot.getSensorGroup(sensorId, orgId, true);
132                for (String s : group.getSensors()) {
133                  Sensor sensor = depot.getSensor(s, orgId, true);
134                  for (Measurement meas : depot.getMeasurements(depositoryId, orgId,
135                      sensor.getId(), startDate, endDate, true)) {
136                    ret.getMeasurements().add(meas);
137                  }
138                }
139              }
140              catch (IdNotFoundException nf1) {
141                setStatus(Status.CLIENT_ERROR_BAD_REQUEST, sensorId + " is not defined");
142              }
143            }
144          }
145          else {
146            setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Start date and/or end date missing.");
147          }
148        }
149        else {
150          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not defined.");
151        }
152      }
153      catch (IdNotFoundException e) {
154        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
155      }
156      catch (ParseException e) {
157        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
158      }
159      catch (DatatypeConfigurationException e) {
160        setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
161      }
162      catch (MisMatchedOwnerException e) {
163        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
164      }
165      return ret;
166    }
167    else {
168      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "No start and/or end dates.");
169      return null;
170    }
171  }
172}
