001package org.wattdepot.server.http.api;
002
003import org.restlet.data.Status;
004import org.restlet.resource.ResourceException;
005import org.wattdepot.common.domainmodel.Depository;
006import org.wattdepot.common.domainmodel.InterpolatedValue;
007import org.wattdepot.common.domainmodel.InterpolatedValueList;
008import org.wattdepot.common.domainmodel.Labels;
009import org.wattdepot.common.domainmodel.Sensor;
010import org.wattdepot.common.domainmodel.SensorGroup;
011import org.wattdepot.common.exception.IdNotFoundException;
012import org.wattdepot.common.exception.MisMatchedOwnerException;
013import org.wattdepot.common.exception.NoMeasurementException;
014import org.wattdepot.common.http.api.DepositoryLatestValuesResource;
015
016import java.util.Date;
017import java.util.logging.Level;
018
019/**
020 * DepositoryLatestValuesServerResource - Returns the latest value for the sensor or sensor group stored in the
021 * depository as an InterpolatedValueList.
022 *
023 * @author Cam Moore
024 */
025
026public class DepositoryLatestValuesServerResource extends WattDepotServerResource implements DepositoryLatestValuesResource {
027  private String depositoryId;
028  private String sensorId;
029
030  /*
031 * (non-Javadoc)
032 *
033 * @see org.restlet.resource.Resource#doInit()
034 */
035  @Override
036  protected void doInit() throws ResourceException {
037    super.doInit();
038    this.sensorId = getQuery().getValues(Labels.SENSOR);
039    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
040  }
041
042  @Override
043  public InterpolatedValueList retrieve() {
044    getLogger().log(
045        Level.INFO,
046        "GET /wattdepot/{" + orgId + "}/" + Labels.DEPOSITORY + "/{" + depositoryId + "}/"
047            + Labels.VALUES + "/" + Labels.LATEST + "/?" + Labels.SENSOR + "={" + sensorId + "}");
048    if (isInRole(orgId)) {
049      try {
050        InterpolatedValueList list = new InterpolatedValueList();
051        Depository depository = depot.getDepository(depositoryId, orgId, true);
052        Sensor sensor = depot.getSensor(sensorId, orgId, false);
053        if (sensor != null) {
054          InterpolatedValue value = null;
055          try {
056            value = depot.getLatestMeasuredValue(depositoryId, orgId, sensorId, false);
057            list.getInterpolatedValues().add(value);
058          }
059          catch (NoMeasurementException e) {
060            value = new InterpolatedValue(sensorId, Double.NaN, depository.getMeasurementType(), new Date());
061            list.getInterpolatedValues().add(value); // adding null to the list?
062          }
063        }
064        else {
065          SensorGroup group = depot.getSensorGroup(sensorId, orgId, false);
066          if (group != null) {
067            for (String s : group.getSensors()) {
068              sensor = depot.getSensor(s, orgId, false);
069              if (sensor != null) {
070                InterpolatedValue value = null;
071                try {
072                  value = depot.getLatestMeasuredValue(depositoryId, orgId, s, false);
073                  list.getInterpolatedValues().add(value);
074                }
075                catch (NoMeasurementException e) {
076                  value = new InterpolatedValue(s, Double.NaN, depository.getMeasurementType(), new Date());
077                  list.getInterpolatedValues().add(value); // adding null to the list?
078                }
079              }
080            }
081          }
082        }
083        return list;
084      }
085      catch (IdNotFoundException e) {
086        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not a Depository id.");
087        return null;
088      }
089      catch (MisMatchedOwnerException e) {
090        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not in organization " + orgId + ".");
091        return null;
092      }
093
094    }
095    else {
096      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
097      return null;
098    }
099  }
100}
