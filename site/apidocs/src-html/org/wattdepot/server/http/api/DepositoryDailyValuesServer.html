044/**
045 * DepositoryDailyValuesServer - Base class for handling daily sample HTTP requests.
046 *
047 * @author Cam Moore
048 */
049public class DepositoryDailyValuesServer extends WattDepotServerResource {
050  private static final Integer DAY_MINUTES = 24 * 60;
051  private String depositoryId;
052  private String sensorId;
053  private String start;
054  private String end;
055  private String dataType;
056
057  /*
058   * (non-Javadoc)
059   *
060   * @see org.restlet.resource.Resource#doInit()
061   */
062  @Override
063  protected void doInit() throws ResourceException {
064    super.doInit();
065    this.sensorId = getQuery().getValues(Labels.SENSOR);
066    this.start = getQuery().getValues(Labels.START);
067    this.end = getQuery().getValues(Labels.END);
068    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
069    this.dataType = getQuery().getValues(Labels.VALUE_TYPE);
070  }
071
072  /**
073   * retrieve the daily depository value list for a sensor.
074   *
075   * @return measurement list.
076   */
077  public InterpolatedValueList doRetrieve() {
078    getLogger().log(
079        Level.INFO,
080        "GET /wattdepot/{" + orgId + "}/" + Labels.DEPOSITORY + "/{" + depositoryId + "}/" + Labels.DAILY + "/"
081            + Labels.VALUES + "/?" + Labels.SENSOR + "={" + sensorId + "}&" + Labels.START + "={"
082            + start + "}&" + Labels.END + "={" + end + "}&" + Labels.VALUE_TYPE + "={" + dataType + "}");
083    if (isInRole(orgId)) {
084      if (start != null && end != null && dataType != null) {
085        InterpolatedValueList ret = new InterpolatedValueList();
086        try {
087          Depository depository = depot.getDepository(depositoryId, orgId, true);
088          if (depository != null) {
089            XMLGregorianCalendar startTime = DateConvert.parseCalString(start);
090            // set start time to beginning of day.
091            startTime.setTime(0, 0, 0, 0);
092            XMLGregorianCalendar endTime = DateConvert.parseCalString(end);
093            // set end time to beginning of day.
094            endTime.setTime(0, 0, 0, 0);
095            List<XMLGregorianCalendar> times = Tstamp.getTimestampList(startTime, endTime, DAY_MINUTES);
096            if (times != null) {
097              for (int i = 1; i < times.size(); i++) {
098                XMLGregorianCalendar begin = times.get(i - 1);
099                Date beginDate = begin.toGregorianCalendar().getTime();
100                XMLGregorianCalendar end = times.get(i);
101                Date endDate = end.toGregorianCalendar().getTime();
102                Sensor sensor = depot.getSensor(sensorId, orgId, false);
103                Double val = 0.0;
104                InterpolatedValue value = new InterpolatedValue(sensorId, val, depository.getMeasurementType(), beginDate, endDate);
105                value.addDefinedSensor(sensorId);
106                if (sensor != null) {
107                  try {
108                    val = getValueForSensor(depositoryId, orgId, sensorId, beginDate, endDate, dataType);
109                    value.addReportingSensor(sensorId);
110                    value.setValue(val);
111                  }
112                  catch (NoMeasurementException e) {
113                    val = Double.NaN;
114                    value.setValue(val);
115                  }
116
117                  ret.getInterpolatedValues().add(value);
118                }
119                else { // try SensorGroup.
120                  SensorGroup group = depot.getSensorGroup(sensorId, orgId, false);
121                  if (group != null) {
122                    value = new InterpolatedValue(sensorId, val, depository.getMeasurementType(), beginDate, endDate);
123                    for (String s : group.getSensors()) {
124                      value.addDefinedSensor(s);
125                      sensor = depot.getSensor(s, orgId, false);
126                      if (sensor != null) {
127                        try {
128                          value.setValue(value.getValue() + getValueForSensor(depositoryId, orgId, s, beginDate, endDate, dataType));
129                          value.addReportingSensor(s);
130                        }
131                        catch (IdNotFoundException e) { // NOPMD
132                        }
133                        catch (NoMeasurementException e) { // NOPMD
134                        }
135                      }
136                    }
137                    ret.getInterpolatedValues().add(value);
138                  }
139                }
140              }
141            }
142          }
143        }
144        catch (ParseException e) {
145          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
146          return null;
147        }
148        catch (DatatypeConfigurationException e) {
149          setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
150          return null;
151        }
152        catch (IdNotFoundException e) {
153          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
154          return null;
155        }
156        catch (MisMatchedOwnerException e) {
157          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
158          return null;
159        }
160        return ret;
161      }
162      else {
163        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Missing start and/or end times or value type.");
164        return null;
165      }
166    }
167    else {
168      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
169      return null;
170    }
171  }
172
173  /**
174   * Gets the value for the Sensor or SensorGroup.
175   * @param depositoryId The depository id.
176   * @param orgId The organization id.
177   * @param sensorId The sensor or sensor group id.
178   * @param beginDate The beginning date.
179   * @param endDate The ending date.
180   * @param dataType Either 'point' or difference.
181   * @return The average for point values or the difference for difference values.
182   * @throws IdNotFoundException If the sensor or depository are not defined.
183   * @throws NoMeasurementException If there are no measurements for the time period.
184   */
185  private Double getValueForSensor(String depositoryId, String orgId, String sensorId, Date beginDate, Date endDate, String dataType) throws IdNotFoundException, NoMeasurementException {
186    Double val = 0.0;
187    if (dataType.equals("point")) {  // need to calculate the average value for the hourly intervals
188      List<Measurement> measurements = depot.getMeasurements(depositoryId, orgId, sensorId, beginDate, endDate, false);
189      if (measurements.size() > 0) {
190        for (Measurement m : measurements) {
191          val += m.getValue();
192        }
193        val = val / measurements.size();
194      }
195      else {
196        val = Double.NaN;
197      }
198    }
199    else {  // calculate the difference
200      val = depot.getValue(depositoryId, orgId, sensorId, beginDate, endDate, false);
201    }
202    return val;
203  }
