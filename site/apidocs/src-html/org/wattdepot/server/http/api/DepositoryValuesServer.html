019package org.wattdepot.server.http.api;

021import java.text.ParseException;
022import java.util.Date;
023import java.util.List;
024import java.util.logging.Level;
025
026import javax.xml.datatype.DatatypeConfigurationException;
027import javax.xml.datatype.XMLGregorianCalendar;
028
029import org.restlet.data.Status;
030import org.restlet.resource.ResourceException;
031import org.wattdepot.common.domainmodel.Depository;
032import org.wattdepot.common.domainmodel.Labels;
033import org.wattdepot.common.domainmodel.InterpolatedValue;
034import org.wattdepot.common.domainmodel.InterpolatedValueList;
035import org.wattdepot.common.domainmodel.Sensor;
036import org.wattdepot.common.domainmodel.SensorGroup;
037import org.wattdepot.common.exception.IdNotFoundException;
038import org.wattdepot.common.exception.MisMatchedOwnerException;
039import org.wattdepot.common.exception.NoMeasurementException;
040import org.wattdepot.common.util.DateConvert;
041import org.wattdepot.common.util.tstamp.Tstamp;

043/**
044 * DepositoryValuesServer - Base class for handling the Depository values HTTP
045 * API ("/wattdepot/{org-id}/depository/{depository-id}/values/").
046 * 
047 * @author Yongwen Xu
048 * 
049 */
050public class DepositoryValuesServer extends WattDepotServerResource {
051  protected String depositoryId;
052  protected String sensorId;
053  protected String start;
054  protected String end;
055  protected String interval;
056  protected String dataType;
057
058  /*
059   * (non-Javadoc)
060   * 
061   * @see org.restlet.resource.Resource#doInit()
062   */
063  @Override
064  protected void doInit() throws ResourceException {
065    super.doInit();
066    this.sensorId = getQuery().getValues(Labels.SENSOR);
067    this.start = getQuery().getValues(Labels.START);
068    this.end = getQuery().getValues(Labels.END);
069    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
070    this.interval = getQuery().getValues(Labels.INTERVAL);
071    this.dataType = getQuery().getValues(Labels.VALUE_TYPE);
072  }
073
074  /**
075   * retrieve the depository measurement list for a sensor.
076   * 
077   * @return measurement list.
078   */
079  public InterpolatedValueList doRetrieve() {
080    getLogger().log(
081        Level.INFO,
082        "GET /wattdepot/{" + orgId + "}/" + Labels.DEPOSITORY + "/{" + depositoryId + "}/"
083            + Labels.VALUES + "/?" + Labels.SENSOR + "={" + sensorId + "}&" + Labels.START + "={"
084            + start + "}&" + Labels.END + "={" + end + "}&" + Labels.INTERVAL + "={" + interval
085            + "}&" + Labels.VALUE_TYPE + "={" + dataType + "}");
086    if (isInRole(orgId)) {
087      if (start != null && end != null && interval != null) {
088        InterpolatedValueList ret = new InterpolatedValueList();
089        try {
090          Depository depository = depot.getDepository(depositoryId, orgId, true);
091          if (depository != null) {
092            XMLGregorianCalendar startTime = DateConvert.parseCalString(start);
093            XMLGregorianCalendar endTime = DateConvert.parseCalString(end);
094
095            // interval is in minute
096            int intervalMinutes = Integer.parseInt(interval);
097
098            // Build list of timestamps, starting with startTime, separated by
099            // intervalMinutes
100            List<XMLGregorianCalendar> timestampList = Tstamp.getTimestampList(startTime, endTime,
101                intervalMinutes);
102            if (timestampList != null) {
103              Sensor sensor = depot.getSensor(sensorId, orgId, false);
104              if (sensor != null) {
105                Date previous = null;
106                for (int i = 0; i < timestampList.size(); i++) {
107                  InterpolatedValue mValue;
108                  Date timestamp = DateConvert.convertXMLCal(timestampList.get(i));
109                  Double value = new Double(0);
110                  if ("point".equals(dataType)) {
111                    try {
112                      value = depot.getValue(depositoryId, orgId, sensor.getId(), timestamp, true);
113                    }
114                    catch (NoMeasurementException e) { // NOPMD
115                      // no measurements around the time so return 0?
116                    }
117                    mValue = new InterpolatedValue(sensor.getId(), value,
118                        depository.getMeasurementType(), timestamp);
119                    ret.getInterpolatedValues().add(mValue);
120                  }
121                  else {
122                    if (previous != null) {
123                      try {
124                        value = depot.getValue(depositoryId, orgId, sensor.getId(), previous, timestamp, true);
125                      }
126                      catch (NoMeasurementException e) { // NOPMD
127                        // No measurements so return 0,
128                      }
129                      mValue = new InterpolatedValue(sensor.getId(), value, depository.getMeasurementType(), previous, timestamp);
130                      ret.getInterpolatedValues().add(mValue);
131                    }
132                    previous = timestamp;
133                  }
134                }
135              }
136              else {
137                SensorGroup group = depot.getSensorGroup(sensorId, orgId, false);
138                if (group != null) {
139                  Date previous = null;
140                  for (int i = 0; i < timestampList.size(); i++) {
141                    Date timestamp = DateConvert.convertXMLCal(timestampList.get(i));
142                    Double value = new Double(0);
143                    for (String s : group.getSensors()) {
144                      Sensor sens = depot.getSensor(s, orgId, true);
145                      if (sens != null) {
146                        if ("point".equals(dataType)) {
147                          try {
148                            value += depot.getValue(depositoryId, orgId, sens.getId(), timestamp,
149                                true);
150                          }
151                          catch (NoMeasurementException e) { // NOPMD
152                            // No measurements so return 0;
153                          }
154                        }
155                        else {
156                          if (previous != null) {
157                            try {
158                              value += depot.getValue(depositoryId, orgId, sens.getId(),
159                                  previous, timestamp, true);
160                            }
161                            catch (NoMeasurementException e) { // NOPMD
162                              // No measurements so return 0,
163                            }
164                          }
165                        }
166                      }
167                    }
168                    if ("point".equals(dataType)) {
169                      InterpolatedValue mValue = new InterpolatedValue(group.getId(), value,
170                          depository.getMeasurementType(), timestamp);
171                      ret.getInterpolatedValues().add(mValue);
172                    }
173                    else {
174                      if (previous != null) {
175                        InterpolatedValue mValue = new InterpolatedValue(group.getId(), value,
176                            depository.getMeasurementType(), previous, timestamp);
177                        ret.getInterpolatedValues().add(mValue);
178                      }
179                    }
180                    previous = timestamp;
181                  }
182                }
183              }
184            }
185            else {
186              setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad start, end, or interval.");
187              return null;
188            }
189          }
190          else {
191            setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not defined.");
192            return null;
193          }
194        }
195        catch (MisMatchedOwnerException e) {
196          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
197          return null;
198        }
199        catch (ParseException e) {
200          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
201          return null;
202        }
203        catch (DatatypeConfigurationException e) {
204          setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
205          return null;
206        }
207        catch (IdNotFoundException e) {
208          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
209          return null;
210        }
211        getLogger().info(ret.toString());
212        return ret;
213      }
214      else {
215        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Missing start and/or end times or interval.");
216        return null;
217      }
218    }
219    else {
220      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
221      return null;
222    }
223  }
224}
