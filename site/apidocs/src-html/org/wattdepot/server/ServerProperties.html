137  /**
138   * Sets the properties to their "test" equivalents and updates the system
139   * properties.
140   */
141  public void setTestProperties() {
142    properties.setProperty(PORT_KEY, properties.getProperty(TEST_PORT_KEY));
143    properties.setProperty(WATT_DEPOT_IMPL_KEY, properties.getProperty(TEST_WATT_DEPOT_IMPL_KEY));
144    // turn off logging during testing.
145    properties.setProperty(LOGGING_LEVEL_KEY, "SEVERE");
146    trimProperties(properties);
147    // update the system properties object to reflect these new values.
148    Properties systemProperties = System.getProperties();
149    systemProperties.putAll(properties);
150    System.setProperties(systemProperties);
151  }

153  /**
154   * Returns a string containing all current properties in alphabetical order.
155   * Properties with the word "password" in their key list "((hidden))" rather
156   * than their actual value for security reasons. This is not super
157   * sophisticated - other properties such as database URLs might benefit from
158   * being hidden too - but it should work for now.
159   * 
160   * @return A string with the properties.
161   */
162  public String echoProperties() {
163    String cr = System.getProperty("line.separator");
164    String eq = " = ";
165    String pad = "                ";
166    // Adding them to a treemap has the effect of alphabetizing them.
167    TreeMap<String, String> alphaProps = new TreeMap<String, String>();
168    for (Map.Entry<Object, Object> entry : this.properties.entrySet()) {
169      String propName = (String) entry.getKey();
170      String propValue = (String) entry.getValue();
171      alphaProps.put(propName, propValue);
172    }
173    StringBuffer buff = new StringBuffer(25);
174    buff.append("Server Properties:").append(cr);
175    for (String key : alphaProps.keySet()) {
176      if (key.contains("password")) {
177        buff.append(pad).append(key).append(eq).append("((hidden))").append(cr);
178      }
179      else {
180        buff.append(pad).append(key).append(eq).append(get(key)).append(cr);
181      }
182    }
183    return buff.toString();
184  }

205  /**
206   * Ensures that the there is no leading or trailing whitespace in the property
207   * values. The fact that we need to do this indicates a bug in Java's
208   * Properties implementation to me.
209   * 
210   * @param properties The properties.
211   */
212  private void trimProperties(Properties properties) {
213    // Have to do this iteration in a Java 5 compatible manner. no
214    // stringPropertyNames().
215    for (Map.Entry<Object, Object> entry : properties.entrySet()) {
216      String propName = (String) entry.getKey();
217      properties.setProperty(propName, properties.getProperty(propName).trim());
218    }
219  }

221  /**
222   * Reads in the properties in ~/.wattdepot3/server/wattdepot-server.properties
223   * if this file exists, and provides default values for all properties not
224   * mentioned in this file. Will also add any pre-existing System properties
225   * that start with "wattdepot-server.".
226   * 
227   * @param serverSubdir The name of the subdirectory used to store all files
228   *        for this server.
229   * @throws Exception if errors occur.
230   */
231  private void initializeProperties(String serverSubdir) throws Exception {
232    Logger logger = Logger.getLogger("org.wattdepot.properties");
233    String userHome = UserHome.getHomeString();
234    String wattDepot3Home = userHome + "/.wattdepot3/";
235    String serverHome = null;
236    if (serverSubdir == null) {
237      serverHome = wattDepot3Home + "server";
238    }
239    else {
240      serverHome = wattDepot3Home + serverSubdir;
241    }
242    String propFileName = serverHome + "/wattdepot-server.properties";
243    this.properties = new Properties();
244    // Use properties from file, if they exist.
245    FileInputStream stream = null;
246    try {
247      stream = new FileInputStream(propFileName);
248      properties.load(stream);
249      logger.info("Loading Server properties from: " + propFileName);
250
251    }
252    catch (IOException e) {
253      logger.info(propFileName + " not found. Using default server properties.");
254    }
255    finally {
256      if (stream != null) {
257        stream.close();
258      }
259    }
260    processDatabaseURL(properties.getProperty(DATABASE_URL));
261    // grab all of the properties in the environment
262    Map<String, String> systemProps = System.getenv();
263    for (Map.Entry<String, String> prop : systemProps.entrySet()) {
264      if (prop.getKey().startsWith(ADMIN_USER_NAME_ENV)) {
265        properties.setProperty(ADMIN_USER_NAME, prop.getValue());
266      }
267      if (prop.getKey().startsWith(ADMIN_USER_PASSWORD_ENV)) {
268        properties.setProperty(ADMIN_USER_PASSWORD, prop.getValue());
269      }
270      if (prop.getKey().startsWith(DB_CONNECTION_URL_ENV)) {
271        processDatabaseURL(prop.getValue());
272      }
273      if (prop.getKey().startsWith("DATABASE_URL")) {
274        processDatabaseURL(prop.getValue());
275      }
276    }
277    if (!properties.containsKey(ADMIN_USER_NAME) || !properties.containsKey(ADMIN_USER_PASSWORD) || !properties.containsKey(DB_CONNECTION_URL)) {
278      StringBuilder sb = new StringBuilder();
279      if (!properties.containsKey(ADMIN_USER_NAME)) {
280        sb.append("WattDepot Admin name not set. ");
281      }
282      if (!properties.containsKey(ADMIN_USER_PASSWORD)) {
283        sb.append("WattDepot Admin password not set. ");
284      }
285      if (!(properties.containsKey(DB_CONNECTION_URL_ENV) || properties.containsKey("DATABASE_URL"))) {
286        sb.append("WattDepot database url not set. ");
287      }
288      throw new SecurityException(sb.toString());
289    }
290    UserInfo.ROOT.setUid(properties.getProperty(ADMIN_USER_NAME));
291    UserInfo.ROOT.setPassword(properties.getProperty(ADMIN_USER_PASSWORD));
292    UserPassword.ROOT.setUid(properties.getProperty(ADMIN_USER_NAME));
293    UserPassword.ROOT.setPassword(properties.getProperty(ADMIN_USER_PASSWORD));
294    String defaultWattDepotImpl = "org.wattdepot.server.depository.impl.hibernate.WattDepotPersistenceImpl";
295    // Set default values if not set
296    if (!properties.containsKey(SERVER_HOME_DIR)) {
297      properties.setProperty(SERVER_HOME_DIR, serverHome);
298    }
299    if (!properties.containsKey(WATT_DEPOT_IMPL_KEY)) {
300      properties.setProperty(WATT_DEPOT_IMPL_KEY, defaultWattDepotImpl);
301    }
302    if (!properties.containsKey(HOSTNAME_KEY)) {
303      properties.setProperty(HOSTNAME_KEY, "localhost");
304    }
305    if (!properties.containsKey(PORT_KEY)) {
306      properties.setProperty(PORT_KEY, "8192");
307    }
308    if (!properties.containsKey(DB_CONNECTION_DRIVER)) {
309      properties.setProperty(DB_CONNECTION_DRIVER, "org.postgresql.Driver");
310    }
311    if (!properties.containsKey(DB_CONNECTION_URL)) {
312      properties.setProperty(DB_CONNECTION_URL, "jdbc:postgresql://localhost:5432/wattdepot");
313    }
314    if (!properties.containsKey(DB_SHOW_SQL)) {
315      properties.setProperty(DB_SHOW_SQL, FALSE);
316    }
317    if (!properties.containsKey(DB_TABLE_UPDATE)) {
318      properties.setProperty(DB_TABLE_UPDATE, "update");
319    }
320    if (!properties.containsKey(ENABLE_LOGGING_KEY)) {
321      properties.setProperty(ENABLE_LOGGING_KEY, TRUE);
322    }
323    if (!properties.containsKey(CHECK_SESSIONS)) {
324      properties.setProperty(CHECK_SESSIONS, FALSE);
325    }
326    if (!properties.containsKey(LOGGING_LEVEL_KEY)) {
327      properties.setProperty(LOGGING_LEVEL_KEY, "INFO");
328    }
329    if (!properties.containsKey(CONTEXT_ROOT_KEY)) {
330      properties.setProperty(CONTEXT_ROOT_KEY, "wattdepot");
331    }
332    if (!properties.containsKey(SERVER_TIMING_KEY)) {
333      properties.setProperty(SERVER_TIMING_KEY, FALSE);
334    }
335    if (!properties.containsKey(TEST_PORT_KEY)) {
336      properties.setProperty(TEST_PORT_KEY, "8194");
337    }
338    if (!properties.containsKey(TEST_WATT_DEPOT_IMPL_KEY)) {
339      properties.setProperty(TEST_WATT_DEPOT_IMPL_KEY, defaultWattDepotImpl);
340    }
341    if (!properties.containsKey(USE_HEROKU_KEY)) {
342      properties.setProperty(USE_HEROKU_KEY, FALSE);
343    }
344    if (!properties.containsKey(TEST_HEROKU_KEY)) {
345      properties.setProperty(TEST_HEROKU_KEY, FALSE);
346    }
347    logger.finest(echoProperties());
348    trimProperties(properties);
349    logger.finest(echoProperties());
350
351    // get PORT and DATABASE_URL for heroku
352    String webPort = System.getenv("PORT");
353    if (webPort != null && !webPort.isEmpty()) {
354      properties.setProperty(PORT_KEY, webPort);
355    }
356
357    String databaseURL = System.getenv("DATABASE_URL");
358    if (databaseURL != null && !databaseURL.isEmpty()) {
359      URI dbUri = new URI(databaseURL);
360      String username = dbUri.getUserInfo().split(":")[0];
361      String password = dbUri.getUserInfo().split(":")[1];
362      String dbUrl = "jdbc:postgresql://" + dbUri.getHost() +  dbUri.getPath();
363      properties.setProperty(DB_USER_NAME, username);
364      properties.setProperty(DB_PASSWORD, password);
365      properties.setProperty(DB_CONNECTION_URL, dbUrl);
366    }
367
368  }

370  /**
371   * Parses the given database url to set the database username, password, and connection url.
372   * @param url the database url.
373   * @throws java.net.URISyntaxException if there is a problem with the url.
374   */
375  private void processDatabaseURL(String url) throws URISyntaxException {
376    if (url != null && !url.isEmpty()) {
377      URI dbUri = new URI(url);
378      String username = dbUri.getUserInfo().split(":")[0];
379      String password = dbUri.getUserInfo().split(":")[1];
380      String dbUrl = "jdbc:postgresql://" + dbUri.getHost() + ":" + dbUri.getPort() +  dbUri.getPath();
381      properties.setProperty(DB_USER_NAME, username);
382      properties.setProperty(DB_PASSWORD, password);
383      properties.setProperty(DB_CONNECTION_URL, dbUrl);
384    }
385
386  }

388  /**
389   * Returns the fully qualified host name, such as
390   * "http://localhost:9876/wattdepot/". Note, the String will end with "/", so
391   * there is no need to append another if you are constructing a URI.
392   * 
393   * @return The fully qualified host name.
394   */
395  public String getFullHost() {
396    // We have a special case for Heroku here, which leaves out the port number.
397    // This is needed
398    // because Heroku apps listen on a private port on localhost, but remote
399    // connections into
400    // the server always come on port 80. This causes problems because the port
401    // number is used
402    // in at least 3 places: by the Server to decide what port number to bind to
403    // (on Heroku this
404    // is the private port # given by the $PORT environment variable), the
405    // announced URL of the
406    // server to the public (always 80 on Heroku, though usually left out of URI
407    // to default to
408    // 80), and the URI of parent resources such as a Source in a SensorData
409    // resource (should be
410    // the public port on Heroku).
411    if (properties.getProperty(USE_HEROKU_KEY).equals(TRUE)
412        || properties.getProperty(TEST_HEROKU_KEY).equals(TRUE)) {
413      return "http://" + get(HOSTNAME_KEY) + ":80/" + get(CONTEXT_ROOT_KEY) + "/";
414    }
415    else {
416      return "http://" + get(HOSTNAME_KEY) + ":" + get(PORT_KEY) + "/" + get(CONTEXT_ROOT_KEY)
417          + "/";
418    }
419  }
