122  /**
123   * Creates a new WattDepotClient.
124   * 
125   * @param serverUri The URI of the WattDepot server (e.g.
126   *        "http://server.wattdepot.org/")
127   * @param username The name of the user. The user must be defined in the
128   *        WattDepot server.
129   * @param orgId the organization the user is in.
130   * @param password The password for the user.
131   * @throws BadCredentialException If the user or password don't match the
132   *         credentials on the WattDepot server.
133   */
134  public WattDepotClient(String serverUri, String username, String orgId, String password)
135      throws BadCredentialException {
136    // ensure that the UnitsHelper class is loaded.
137    new UnitsHelper();
138    this.properties = new ClientProperties();
139    this.logger = Logger.getLogger("org.wattdepot.client");
140    LoggerUtil.setLoggingLevel(this.logger, properties.get(ClientProperties.LOGGING_LEVEL_KEY));
141    LoggerUtil.useConsoleHandler();
142    logger.finest("Client " + serverUri + ", " + username + ", " + password);
143    this.authentication = new ChallengeResponse(this.scheme, username, password);
144    if (serverUri == null) {
145      throw new IllegalArgumentException("serverUri cannot be null");
146    }
147    if (!serverUri.endsWith("/")) {
148      throw new IllegalArgumentException("serverUri must end with '/'");
149    }
150    this.wattDepotUri = serverUri + Labels.WATTDEPOT + "/";
151
152    ClientResource client = null;
153    client = makeClient(orgId + "/");
154    try {
155      client.head();
156      if (client.getLocationRef() != null) {
157        String path = client.getLocationRef().getPath();
158        path = path.substring(0, path.length() - 1);
159        int lastSlash = path.lastIndexOf('/') + 1;
160        organizationId = path.substring(lastSlash);
161      }
162      else {
163        organizationId = orgId;
164      }
165      client.release();
166    }
167    catch (ResourceException e) {
168      throw new BadCredentialException(e.getMessage() + " username and or password are not corect.");
169    }
170  }
