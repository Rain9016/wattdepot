019package org.wattdepot.client.http.api.performance;
020
021import java.util.Date;
022
023import javax.measure.unit.Unit;
024
025import org.wattdepot.common.domainmodel.Measurement;
026import org.wattdepot.common.exception.BadCredentialException;
027import org.wattdepot.common.exception.BadSensorUriException;
028import org.wattdepot.common.exception.IdNotFoundException;
029import org.wattdepot.common.exception.MeasurementTypeException;
030
031/**
032 * PutTask stores a Measurement using the WattDepotClient. Records
033 * the amount of time the puts take.
034 * 
035 * @author Cam Moore
036 * 
037 */
038public class PutTask extends PerformanceTimedTask {
039
040  /**
041   * Initializes the PutTask.
042   * 
043   * @param serverUri The URI for the WattDepot server.
044   * @param username The name of a user defined in the WattDepot server.
045   * @param orgId the id of the organization the user is in.
046   * @param password The password for the user.
047   * @param debug flag for debugging messages.
048   * @throws BadCredentialException if the user or password don't match the
049   *         credentials in WattDepot.
050   * @throws IdNotFoundException if the processId is not defined.
051   * @throws BadSensorUriException if the Sensor's URI isn't valid.
052   */
053  public PutTask(String serverUri, String username, String orgId, String password,
054      boolean debug) throws BadCredentialException, IdNotFoundException,
055      BadSensorUriException {
056    super(serverUri, username, orgId, password, debug);
057  }
058
059  /**
060   * Initializes the PutTask.
061   * 
062   * @param serverUri The URI for the WattDepot server.
063   * @param username The name of a user defined in the WattDepot server.
064   * @param orgId the id of the organization the user is in.
065   * @param password The password for the user.
066   * @param debug flag for debugging messages.
067   * @param cpd the CollectorProcessDefinition id.
068   * @throws BadCredentialException if the user or password don't match the
069   *         credentials in WattDepot.
070   * @throws IdNotFoundException if the processId is not defined.
071   * @throws BadSensorUriException if the Sensor's URI isn't valid.
072   */
073  public PutTask(String serverUri, String username, String orgId, String password,
074      boolean debug, String cpd) throws BadCredentialException, IdNotFoundException,
075      BadSensorUriException {
076    super(serverUri, username, orgId, password, debug, cpd);
077  }
078
079  /**
080   * @return Generates a fake Measurement with the current time.
081   */
082  private Measurement generateFakeMeasurement() {
083    Date now = new Date();
084    Unit<?> unit = Unit.valueOf(this.depository.getMeasurementType().getUnits());
085    Measurement ret = new Measurement(this.definition.getSensorId(), now,
086        25.0 - 50 * Math.random() + 1000.0, unit);
087    return ret;
088  }
089
090  /* (non-Javadoc)
091   * @see org.wattdepot.client.http.api.performance.PerformanceTimedTask#clientTask()
092   */
093  @Override
094  public void clientTask() {
095    Measurement meas = generateFakeMeasurement();
096    try {
097      this.client.putMeasurement(depository, meas);
098    }
099    catch (MeasurementTypeException e) { // NOPMD
100      // shouldn't happen
101      e.printStackTrace();
102    }
103  }
104
105}
