019package org.wattdepot.client.http.api.performance;
020
021import java.util.Timer;
022
023import org.apache.commons.cli.CommandLine;
024import org.apache.commons.cli.CommandLineParser;
025import org.apache.commons.cli.HelpFormatter;
026import org.apache.commons.cli.Options;
027import org.apache.commons.cli.ParseException;
028import org.apache.commons.cli.PosixParser;
029import org.wattdepot.common.exception.BadCredentialException;
030import org.wattdepot.common.exception.BadSensorUriException;
031import org.wattdepot.common.exception.IdNotFoundException;
032
033/**
034 * PutRate - Starts up a process that sends Measurements to the
035 * WattDepot server at the given rate of measurements per second.
036 * 
037 * @author Cam Moore
038 * 
039 */
040public class PutRate {
041
042  /**
043   * @param args command line arguments -s <server uri> -u <username> -p
044   *        <password> -o <orgId> -mps <measurementsPerSecond> [-d].
045   * @throws BadSensorUriException if there is a problem with the WattDepot
046   *         sensor definition.
047   * @throws IdNotFoundException if there is a problem with the organization id.
048   * @throws BadCredentialException if the credentials are not valid.
049   */
050  public static void main(String[] args) throws BadCredentialException, IdNotFoundException,
051      BadSensorUriException {
052    Options options = new Options();
053    CommandLine cmd = null;
054    String serverUri = null;
055    String username = null;
056    String organizationId = null;
057    String password = null;
058    Integer measPerSec = null;
059    boolean debug = false;
060
061    options.addOption("h", false, "Usage: PutRate -s <server uri> -u <username>"
062        + " -p <password> -o <orgId> -mps <measurementsPerSecond> [-d]");
063    options.addOption("s", "server", true, "WattDepot Server URI. (http://server.wattdepot.org)");
064    options.addOption("u", "username", true, "Username");
065    options.addOption("o", "organizationId", true, "User's Organization id.");
066    options.addOption("p", "password", true, "Password");
067    options.addOption("mps", "measurementsPerSecond", true,
068        "Number of measurements to put per second.");
069    options.addOption("d", "debug", false, "Displays statistics as the Measurements are stored.");
070    CommandLineParser parser = new PosixParser();
071    HelpFormatter formatter = new HelpFormatter();
072    try {
073      cmd = parser.parse(options, args);
074    }
075    catch (ParseException e) {
076      System.err.println("Command line parsing failed. Reason: " + e.getMessage() + ". Exiting.");
077      System.exit(1);
078    }
079    if (cmd.hasOption("h")) {
080      formatter.printHelp("PutRate", options);
081      System.exit(0);
082    }
083    if (cmd.hasOption("s")) {
084      serverUri = cmd.getOptionValue("s");
085    }
086    else {
087      serverUri = "http://server.wattdepot.org/";
088    }
089    if (cmd.hasOption("u")) {
090      username = cmd.getOptionValue("u");
091    }
092    else {
093      username = "user";
094    }
095    if (cmd.hasOption("p")) {
096      password = cmd.getOptionValue("p");
097    }
098    else {
099      password = "default";
100    }
101    if (cmd.hasOption("o")) {
102      organizationId = cmd.getOptionValue("o");
103    }
104    else {
105      organizationId = "organization";
106    }
107    if (cmd.hasOption("mps")) {
108      measPerSec = Integer.parseInt(cmd.getOptionValue("mps"));
109    }
110    else {
111      measPerSec = 13;
112    }
113    debug = cmd.hasOption("d");
114    if (debug) {
115      System.out.println("Measurement Put Rate:");
116      System.out.println("    WattDepotServer: " + serverUri);
117      System.out.println("    Username: " + username);
118      System.out.println("    OrganizationId: " + organizationId);
119      System.out.println("    Password: ********");
120      System.out.println("    Meas/Sec: " + measPerSec);
121    }
122
123    Timer t = new Timer("monitoring");
124    for (int i = 0; i < measPerSec; i++) {
125      t.schedule(new PutTask(serverUri, username, organizationId, password, debug), 0,
126          1000);
127    }
128  }
129
130}
