019package org.wattdepot.client.http.api.collector;
020
021import org.apache.commons.cli.CommandLine;
022import org.apache.commons.cli.CommandLineParser;
023import org.apache.commons.cli.HelpFormatter;
024import org.apache.commons.cli.Options;
025import org.apache.commons.cli.ParseException;
026import org.apache.commons.cli.PosixParser;
027import org.wattdepot.client.http.api.WattDepotClient;
028import org.wattdepot.common.domainmodel.CollectorProcessDefinition;
029import org.wattdepot.common.exception.BadCredentialException;
030
031/**
032 * AllOrganizationCollectors attempts to start all the collectors defined for an
033 * Organization.
034 * 
035 * @author Cam Moore
036 * 
037 */
038public class AllOrganizationCollectors {
039
040  /**
041   * Queries the WattDepot server to find all the CollectorProcessDefinitions
042   * defined for the Organization. Then tries to start Collectors for each
043   * definition.
044   * 
045   * @param args command line arguments -s <server uri> -u <username> -p
046   *        <password> -o <orgId> [-d].
047   * @throws BadCredentialException if the credentials are not defined on the
048   *         WattDepot server.
049   */
050  public static void main(String[] args) throws BadCredentialException {
051    Options options = new Options();
052    CommandLine cmd = null;
053    String serverUri = null;
054    String username = null;
055    String organizationId = null;
056    String password = null;
057    boolean debug = false;
058
059    options.addOption("h", false, "Usage: AllOrganizationCollectors -s <server uri> -u <username>"
060        + " -p <password> -o <orgId> [-d]");
061    options.addOption("s", "server", true, "WattDepot Server URI. (http://server.wattdepot.org)");
062    options.addOption("u", "username", true, "Username");
063    options.addOption("o", "organizationId", true, "User's Organization id.");
064    options.addOption("p", "password", true, "Password");
065    options.addOption("d", "debug", false, "Displays debugging information.");
066    CommandLineParser parser = new PosixParser();
067    HelpFormatter formatter = new HelpFormatter();
068    try {
069      cmd = parser.parse(options, args);
070    }
071    catch (ParseException e) {
072      System.err.println("Command line parsing failed. Reason: " + e.getMessage() + ". Exiting.");
073      System.exit(1);
074    }
075    if (cmd.hasOption("h")) {
076      formatter.printHelp("AllOrganizationCollectors", options);
077      System.exit(0);
078    }
079    if (cmd.hasOption("s")) {
080      serverUri = cmd.getOptionValue("s");
081    }
082    else {
083      serverUri = "http://server.wattdepot.org/";
084    }
085    if (cmd.hasOption("u")) {
086      username = cmd.getOptionValue("u");
087    }
088    else {
089      username = "user";
090    }
091    if (cmd.hasOption("p")) {
092      password = cmd.getOptionValue("p");
093    }
094    else {
095      password = "default";
096    }
097    if (cmd.hasOption("o")) {
098      organizationId = cmd.getOptionValue("o");
099    }
100    else {
101      organizationId = "organization";
102    }
103    debug = cmd.hasOption("d");
104    if (debug) {
105      System.out.println("All Organization Collectors:");
106      System.out.println("    WattDepotServer: " + serverUri);
107      System.out.println("    Username: " + username);
108      System.out.println("    OrganizationId: " + organizationId);
109      System.out.println("    Password: ********");
110    }
111    WattDepotClient client = new WattDepotClient(serverUri, username, organizationId, password);
112    for (CollectorProcessDefinition cpd : client.getCollectorProcessDefinitions().getDefinitions()) {
113      try {
114        if (debug) {
115          System.out.println("Starting " + cpd.getId());
116        }
117        if (!MultiThreadedCollector.start(serverUri, username, organizationId, password, cpd.getId(),
118            debug)) {
119          System.err.println("Problem with collector " + cpd.getId() + ". Skipping it.");
120        }
121      }
122      catch (InterruptedException e) {
123        System.err.println("Couldn't start collector " + cpd.getId() + ". Skipping it.");
124      }
125    }
126  }
127
128}
