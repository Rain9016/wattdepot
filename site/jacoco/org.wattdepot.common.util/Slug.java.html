package org.wattdepot.common.util;

/**
 * Slug - Utility class to slugify strings.
 * 
 * @author Cam Moore
 * 
 */
public class Slug {

  /**
   * Converts to lowercase, removes non-word characters (alphanumerics and
   * underscores) and converts spaces to hyphens. Also strips leading and
   * trailing whitespace.
   * 
   * @param str
   *          The string to slugify.
   * @return The slugified String.
   */
  public static String slugify(String str) {
    String ret = null;
    ret = str.toLowerCase();
    ret = ret.replace(" ", "-");
    ret = ret.replace("!", "");
    ret = ret.replace("@", "");
    ret = ret.replace("#", "");
    ret = ret.replace("$", "");
    ret = ret.replace("%", "");
    ret = ret.replace("^", "");
    ret = ret.replace("&", "");
    ret = ret.replace("*", "");
    ret = ret.replace("(", "");
    ret = ret.replace(")", "");
    ret = ret.replace("=", "");
    ret = ret.replace("+", "");
    ret = ret.replace("`", "");
    ret = ret.replace("~", "");
    ret = ret.replace(",", "");
    ret = ret.replace("<", "");
    ret = ret.replace(".", "");
    ret = ret.replace(">", "");
    ret = ret.replace("/", "");
    ret = ret.replace("?", "");
    ret = ret.replace(";", "");
    ret = ret.replace(":", "");
    ret = ret.replace("[", "");
    ret = ret.replace("{", "");
    ret = ret.replace("]", "");
    ret = ret.replace("}", "");
    ret = ret.replace("\\", "");
    ret = ret.replace("|", "");
    return ret;
  }

  /**
   * Checks to see if a slug is valid.
   * 
   * @param slug
   *          the slug to check.
   * @return True if it is valid, false otherwise.
   */
  public static boolean validateSlug(String slug) {
    if (slug != null) {
      return slug.equals(slugify(slug));
    }
    return false;
  }
}
