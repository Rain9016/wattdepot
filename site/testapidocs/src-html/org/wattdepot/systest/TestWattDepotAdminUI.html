056  /**
057   * Starts up a WattDepotServer to start the testing.
058   *
059   * @throws Exception if there is a problem starting the server.
060   */
061  @BeforeClass
062  public static void setupServer() throws Exception {
063//    LoggerUtil.disableLogging();
064    server = WattDepotServer.newTestInstance();
065    // Set up the test instances.
066    Set<Property> properties = new HashSet<Property>();
067    properties.add(new Property("isAdmin", "no they are not"));
068
069    // Set up the logging and clients.
070    try {
071      ClientProperties props = new ClientProperties();
072      props.setTestProperties();
073      logger = Logger.getLogger("org.wattdepot.client");
074      LoggerUtil.setLoggingLevel(logger, props.get(ClientProperties.LOGGING_LEVEL_KEY));
075      LoggerUtil.useConsoleHandler();
076      serverURL = "http://" + props.get(ClientProperties.WATTDEPOT_SERVER_HOST) + ":"
077        + props.get(ClientProperties.PORT_KEY) + "/";
078      logger.finest("Using server " + serverURL);
079    } catch (Exception e) {
080      e.printStackTrace();
081    }
082  }
083
084  /**
085   * Shuts down the WattDepotServer.
086   *
087   * @throws Exception if there is a problem.
088   */
089  @AfterClass
090  public static void stopServer() throws Exception {
091    logger.finest("tearDown()");
092    logger.finest("Done tearDown()");
093    server.stop();
094  }
