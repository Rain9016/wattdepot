059  /**
060   * Test method for
061   * {@link org.wattdepot.common.domainmodel.MeasurementRateSummary#hashCode()}.
062   */
063  @Test
064  public void testHashCode() {
065    int value = base.hashCode();
066    assertTrue(base.hashCode() == value);
067    MeasurementRateSummary sum = new MeasurementRateSummary();
068    assertFalse(sum.hashCode() == value);
069    sum.setLatestValue(base.getLatestValue());
070    assertFalse(sum.hashCode() == value);
071    sum.setDepositoryId(base.getDepositoryId());
072    assertFalse(sum.hashCode() == value);
073    sum.setOneMinuteCount(base.getOneMinuteCount());
074    assertFalse(sum.hashCode() == value);
075    sum.setOneMinuteRate(base.getOneMinuteRate());
076    assertFalse(sum.hashCode() == value);
077    sum.setSensorId(base.getSensorId());
078    assertFalse(sum.hashCode() == value);
079    sum.setTimestamp(base.getTimestamp());
080    assertFalse(sum.hashCode() == value);
081    sum.setTotalCount(base.getTotalCount());
082    assertTrue(sum.hashCode() == value);
083  }
084
085  /**
086   * Test method for
087   * {@link org.wattdepot.common.domainmodel.MeasurementRateSummary#equals(java.lang.Object)}
088   * .
089   */
090  @Test
091  public void testEqualsObject() {
092    assertFalse(base.equals(null));
093    assertTrue(base.equals(base));
094    assertFalse(base.equals(""));
095    MeasurementRateSummary sum = new MeasurementRateSummary();
096    assertFalse(sum.equals(base));
097    sum.setLatestValue(-1.0);
098    assertFalse(sum.equals(base));
099    sum.setLatestValue(base.getLatestValue());
100    assertFalse(sum.equals(base));
101    sum.setDepositoryId("foo");
102    assertFalse(sum.equals(base));
103    sum.setDepositoryId(base.getDepositoryId());
104    assertFalse(sum.equals(base));
105    sum.setOneMinuteCount(-100l);
106    assertFalse(sum.equals(base));
107    sum.setOneMinuteCount(base.getOneMinuteCount());
108    assertFalse(sum.equals(base));
109    sum.setOneMinuteRate(-10.0);
110    assertFalse(sum.equals(base));
111    sum.setOneMinuteRate(base.getOneMinuteRate());
112    assertFalse(sum.equals(base));
113    sum.setSensorId("foo");
114    assertFalse(sum.equals(base));
115    sum.setSensorId(base.getSensorId());
116    assertFalse(sum.equals(base));
117    sum.setTimestamp(new Date());
118    assertFalse(sum.equals(base));
119    sum.setTimestamp(base.getTimestamp());
120    assertFalse(sum.equals(base));
121    sum.setTotalCount(-392l);
122    assertFalse(sum.equals(base));
123    sum.setTotalCount(base.getTotalCount());
124    assertTrue(sum.equals(base));
125    assertTrue(base.equals(InstanceFactory.getMeasurementRateSummary()));
126  }
