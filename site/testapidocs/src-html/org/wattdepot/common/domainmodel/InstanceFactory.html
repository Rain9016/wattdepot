080  /**
081   * @return A MeasurementRateSummary for testing.
082   */
083  public static MeasurementRateSummary getMeasurementRateSummary() {
084    MeasurementRateSummary ret = new MeasurementRateSummary();
085    ret.setDepositoryId(getDepository().getId());
086    ret.setLatestValue(LATEST_VALUE);
087    ret.setOneMinuteCount(ONE_MINUTE_COUNT);
088    ret.setOneMinuteRate(ONE_MINUTE_RATE);
089    ret.setSensorId(getSensor().getId());
090    ret.setTimestamp(getTimeBeforeM1());
091    ret.setTotalCount(TOTAL_COUNT);
092    ret.setType(getMeasurementType());
093    return ret;
094  }
