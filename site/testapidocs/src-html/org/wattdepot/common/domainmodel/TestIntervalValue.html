020package org.wattdepot.common.domainmodel;
021
022import org.junit.Before;
023import org.junit.Test;
024
025import java.util.Date;
026
027import static org.junit.Assert.*;
028import static org.junit.Assert.assertEquals;
029
030public class TestIntervalValue {
031  private String sensorId = "sensorId";
032  private Double dVal = 10.1;
033  private MeasurementType type = InstanceFactory.getMeasurementType();
034  private Date start = InstanceFactory.getTimeBetweenM1andM2();
035  private Date end = InstanceFactory.getTimeBetweenM1andM3();
036  private IntervalValue val1;
037  private IntervalValue val2;
038
039  /**
040   * Sets up the private IntervalValue instances.
041   */
042  @Before
043  public void setUp() {
044    val1 = new IntervalValue(sensorId, dVal, type, start, end);
045    val2 = new IntervalValue();
046
047  }
048
049  /**
050   * Tests the constructor.
051   * @throws Exception if there is a problem.
052   */
053  @Test
054  public void testConstructor() throws Exception {
055    assertNotNull(val1);
056  }
057
058  @Test
059  public void testEquals() throws Exception {
060    assertFalse(val2.equals(val1));
061    assertTrue(val1.equals(val1));
062  }
063
064  @Test
065  public void testGetEnd() throws Exception {
066    assertEquals("Wrong end", end, val1.getEnd());
067    val1.setEnd(start);
068    assertFalse(val1.getEnd().equals(end));
069  }
070
071  @Test
072  public void testGetMeasurementType() throws Exception {
073    assertEquals("Wrong MeasurementType", type, val1.getMeasurementType());
074    val1.setMeasurementType(InstanceFactory.getMeasurementType2());
075    assertFalse(val1.getMeasurementType().equals(type));
076  }
077
078  @Test
079  public void testGetSensorId() throws Exception {
080    assertEquals("Wrong sensorId", sensorId, val1.getSensorId());
081    val1.setSensorId("foo");
082    assertFalse("Failed setSensorId", val1.getSensorId().equals(sensorId));
083  }
084
085  @Test
086  public void testGetStart() throws Exception {
087    assertEquals("Wrong start", start, val1.getStart());
088    val1.setStart(end);
089    assertFalse("Failed setStart", val1.getStart().equals(start));
090  }
091
092  @Test
093  public void testGetValue() throws Exception {
094    assertEquals("Wrong value", dVal, val1.getValue(), 0.000001);
095    val1.setValue(dVal + 2);
096    assertNotEquals("failed setValue", dVal, val1.getValue());
097  }
098
099  @Test
100  public void testHashCode() throws Exception {
101    int hashCode1 = val1.hashCode();
102    int hashCode2 = val2.hashCode();
103    assertNotEquals(hashCode1, hashCode2);
104  }
105
106  @Test
107  public void testToString() throws Exception {
108    String s1 = val1.toString();
109    String s2 = val2.toString();
110    assertNotEquals(s1, s2);
111  }
112}
