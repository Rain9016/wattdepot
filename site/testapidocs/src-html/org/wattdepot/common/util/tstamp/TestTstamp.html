026  @Test
027  public void testTstampArithmetic() throws Exception {
028    XMLGregorianCalendar date1 = Tstamp.makeTimestamp("2007-08-01");
029    XMLGregorianCalendar date2 = Tstamp.makeTimestamp("2007-08-02");
030    XMLGregorianCalendar date3 = Tstamp.makeTimestamp("2007-08-03");
031    assertTrue("Test equal", Tstamp.equal(date1, date1));
032    assertTrue("Test increment days", Tstamp.equal(date2, Tstamp.incrementDays(date1, 1)));
033
034    XMLGregorianCalendar date4 = Tstamp.makeTimestamp("2007-08-01T01:00:00");
035    assertTrue("Test increment hours", Tstamp.equal(date4, Tstamp.incrementHours(date1, 1)));
036
037    XMLGregorianCalendar date5 = Tstamp.makeTimestamp("2007-08-01T00:01:00");
038    assertTrue("Test increment mins", Tstamp.equal(date5, Tstamp.incrementMinutes(date1, 1)));
039
040    XMLGregorianCalendar date6 = Tstamp.makeTimestamp("2007-08-01T00:00:01");
041    assertTrue("Test increment secs", Tstamp.equal(date6, Tstamp.incrementSeconds(date1, 1)));
042
043    assertTrue("Test greater than 1", Tstamp.greaterThan(date2, date1));
044    assertFalse("Test greater than 2", Tstamp.greaterThan(date1, date2));
045    assertFalse("Test greater than 3", Tstamp.greaterThan(date1, date1));
046
047    assertTrue("Test less than 1", Tstamp.lessThan(date1, date2));
048    assertFalse("Test less than 2", Tstamp.lessThan(date2, date1));
049    assertFalse("Test less than 3", Tstamp.lessThan(date1, date1));
050
051    assertTrue("Test inbetween 1", Tstamp.inBetween(date1, date2, date3));
052    assertTrue("Test inbetween 2", Tstamp.inBetween(date1, date1, date2));
053    assertTrue("Test inbetween 3", Tstamp.inBetween(date1, date2, date2));
054    assertFalse("Test inbetween 4", Tstamp.inBetween(date1, date3, date2));
055  }

071  /**
072   * Test to make sure that the is* methods work OK.
073   * 
074   * @throws Exception If problems occur.
075   */
076  @Test
077  public void testOrLater() throws Exception {
078    long millisInADay = 1000 * 60 * 60 * 24;
079    XMLGregorianCalendar today = Tstamp.makeTimestamp();
080    long todayInMillis = today.toGregorianCalendar().getTimeInMillis();
081    long yesterdayInMillis = todayInMillis - millisInADay;
082    XMLGregorianCalendar yesterday = Tstamp.makeTimestamp(yesterdayInMillis);
083    long tomorrowInMillis = todayInMillis + millisInADay;
084    XMLGregorianCalendar tomorrow = Tstamp.makeTimestamp(tomorrowInMillis);
085    long lastWeekInMillis = todayInMillis - (millisInADay * 7);
086    XMLGregorianCalendar lastWeek = Tstamp.makeTimestamp(lastWeekInMillis);
087    assertTrue("Testing todayOrLater 1", Tstamp.isTodayOrLater(today));
088    assertTrue("Testing todayOrLater 2", Tstamp.isTodayOrLater(tomorrow));
089    assertFalse("Testing todayOrLater 3", Tstamp.isTodayOrLater(yesterday));
090    assertTrue("Testing yesterdayOrLater 1", Tstamp.isYesterdayOrLater(today));
091    assertTrue("Testing yesterdayOrLater 2", Tstamp.isYesterdayOrLater(tomorrow));
092    assertTrue("Testing yesterdayOrLater 3", Tstamp.isYesterdayOrLater(yesterday));
093    assertFalse("Testing yesterdayOrLater 4", Tstamp.isYesterdayOrLater(lastWeek));
094  }

096  /**
097   * Test to make sure that the daysBetween method works correctly.
098   * 
099   * @throws Exception If problems occur.
100   */
101  @Test
102  public void testDaysBetween() throws Exception {
103    XMLGregorianCalendar today = Tstamp.makeTimestamp();
104    assertEquals("Test daysBetween 1", 0, Tstamp.daysBetween(today, today));
105    XMLGregorianCalendar tomorrow = Tstamp.incrementDays(today, 1);
106    assertEquals("Test daysBetween 2", 1, Tstamp.daysBetween(today, tomorrow));
107    XMLGregorianCalendar nextWeek = Tstamp.incrementDays(today, 7);
108    assertEquals("Test daysBetween 3", 7, Tstamp.daysBetween(today, nextWeek));
109    // Test a whole year, which might find DST issues.
110    for (int i = 1; i <= 365; i++) {
111      XMLGregorianCalendar newDay = Tstamp.incrementDays(today, i);
112      assertEquals("Test daysBetween 4", i, Tstamp.daysBetween(today, newDay));
113    }
114  }

116  /**
117   * Tests that the sort() method works.
118   * 
119   * @throws Exception If problems occur.
120   */
121  @Test
122  public void testSorting() throws Exception {
123    XMLGregorianCalendar tstamp1 = Tstamp.makeTimestamp();
124    XMLGregorianCalendar tstamp2 = Tstamp.incrementSeconds(tstamp1, 1);
125    XMLGregorianCalendar tstamp3 = Tstamp.incrementSeconds(tstamp1, 2);
126    XMLGregorianCalendar tstamp4 = Tstamp.incrementSeconds(tstamp1, 3);
127    List<XMLGregorianCalendar> tstamps = new ArrayList<XMLGregorianCalendar>();
128    tstamps.add(tstamp2);
129    tstamps.add(tstamp1);
130    tstamps.add(tstamp4);
131    tstamps.add(tstamp3);
132    List<XMLGregorianCalendar> sortedList = Tstamp.sort(tstamps);
133    assertEquals("Test sort1", tstamp1, sortedList.get(0));
134    assertEquals("Test sort2", tstamp2, sortedList.get(1));
135    assertEquals("Test sort3", tstamp3, sortedList.get(2));
136    assertEquals("Test sort4", tstamp4, sortedList.get(3));
137  }

139  /**
140   * Tests getTimestampList method.
141   *
142   * @throws Exception If there are problems.
143   */
144  @Test
145  @SuppressWarnings("PMD.AvoidDuplicateLiterals")
146  public void testGetTimestampList() throws Exception {
147    XMLGregorianCalendar startTime, endTime;
148    int interval;
149    List<XMLGregorianCalendar> timestampList = new ArrayList<XMLGregorianCalendar>(100);
150    startTime = Tstamp.makeTimestamp("2009-12-11T00:00:00.000");
151    endTime = Tstamp.makeTimestamp("2009-12-12T00:00:00.000");
152    interval = -30;
153    assertNull("getTimestampList worked with negative interval", Tstamp.getTimestampList(startTime,
154        endTime, interval));
155    interval = 240;
156    assertNull("getTimestampList worked with bad range", Tstamp.getTimestampList(endTime,
157        startTime, interval));
158    interval = 1441;
159    assertNull("getTimestampList worked with interval too large", Tstamp.getTimestampList(endTime,
160        startTime, interval));
161    interval = 240;
162    assertNull("getTimestampList worked with degenerate range", Tstamp.getTimestampList(startTime,
163        startTime, interval));
164    // Try a simple one: one day at 4 hour intervals
165    timestampList.add(Tstamp.makeTimestamp("2009-12-11T00:00:00.000"));
166    timestampList.add(Tstamp.makeTimestamp("2009-12-11T04:00:00.000"));
167    timestampList.add(Tstamp.makeTimestamp("2009-12-11T08:00:00.000"));
168    timestampList.add(Tstamp.makeTimestamp("2009-12-11T12:00:00.000"));
169    timestampList.add(Tstamp.makeTimestamp("2009-12-11T16:00:00.000"));
170    timestampList.add(Tstamp.makeTimestamp("2009-12-11T20:00:00.000"));
171    timestampList.add(Tstamp.makeTimestamp("2009-12-12T00:00:00.000"));
172    assertEquals("getTimestampList didn't return expected list", timestampList, Tstamp
173        .getTimestampList(startTime, endTime, interval));
174    timestampList.clear();
175    // one day at 144 minute intervals, the value from interval = 0
176    interval = 0;
177    timestampList.add(Tstamp.makeTimestamp("2009-12-11T00:00:00.000"));
178    timestampList.add(Tstamp.makeTimestamp("2009-12-11T02:24:00.000"));
179    timestampList.add(Tstamp.makeTimestamp("2009-12-11T04:48:00.000"));
180    timestampList.add(Tstamp.makeTimestamp("2009-12-11T07:12:00.000"));
181    timestampList.add(Tstamp.makeTimestamp("2009-12-11T09:36:00.000"));
182    timestampList.add(Tstamp.makeTimestamp("2009-12-11T12:00:00.000"));
183    timestampList.add(Tstamp.makeTimestamp("2009-12-11T14:24:00.000"));
184    timestampList.add(Tstamp.makeTimestamp("2009-12-11T16:48:00.000"));
185    timestampList.add(Tstamp.makeTimestamp("2009-12-11T19:12:00.000"));
186    timestampList.add(Tstamp.makeTimestamp("2009-12-11T21:36:00.000"));
187    timestampList.add(Tstamp.makeTimestamp("2009-12-12T00:00:00.000"));
188    assertEquals("getTimestampList didn't return expected list", timestampList, Tstamp
189        .getTimestampList(startTime, endTime, interval));
190    // one day at 6:01 intervals = 361 minutes, so range doesn't divide evenly into intervals
191    timestampList.clear();
192    interval = 361;
193    timestampList.add(Tstamp.makeTimestamp("2009-12-11T00:00:00.000"));
194    timestampList.add(Tstamp.makeTimestamp("2009-12-11T06:01:00.000"));
195    timestampList.add(Tstamp.makeTimestamp("2009-12-11T12:02:00.000"));
196    timestampList.add(Tstamp.makeTimestamp("2009-12-11T18:03:00.000"));
197    timestampList.add(Tstamp.makeTimestamp("2009-12-12T00:00:00.000"));
198    assertEquals("getTimestampList didn't return expected list", timestampList, Tstamp
199        .getTimestampList(startTime, endTime, interval));
200  }
