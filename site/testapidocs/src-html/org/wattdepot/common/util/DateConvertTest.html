024import javax.xml.datatype.XMLGregorianCalendar;
025import java.text.ParseException;

030public class DateConvertTest {
031
032  @Test
033  public void testParseCalString() throws Exception {
034    String testTime1 = "2015-02-17T11:51:32.000-1000";
035    String testTime2 = "2015-02-17T11:51:32.000-00:00";
036    String testTime3 = "2015-02-17T11:51:32.000-0900";
037    XMLGregorianCalendar cal1 = DateConvert.parseCalString(testTime1);
038    XMLGregorianCalendar cal2 = DateConvert.parseCalString(testTime2);
039    XMLGregorianCalendar cal3 = DateConvert.parseCalString(testTime3);
040    assertTrue("Got " + cal1 + " == " + cal2, cal1.getHour() == cal2.getHour() + 10);
041    assertTrue("Got " + cal1 + " > " + cal3, cal1.getHour() > cal3.getHour());
042  }
043
044
045  @Test
046  public void testParseCalStringToDate() throws Exception {
047    String testTime1 = "2015-02-17T11:51:32.000-1000";
048    String testTime2 = "2015-02-17T11:51:32.000";
049    String testTime3 = "2015-02-17T11:51:32.000-0900";
050    Date time1 = DateConvert.parseCalStringToDate(testTime1);
051    Date time2 = null;
052    try {
053      time2 = DateConvert.parseCalStringToDate(testTime2);
054    }
055    catch (ParseException pe) {
056      // expected.
057    }
058    Date time3 = DateConvert.parseCalStringToDate(testTime3);
059    assertNotNull(time1);
060    assertNull(time2);
061    assertNotNull(time3);
062    assertTrue(time1.compareTo(time3) > 0);
063  }
