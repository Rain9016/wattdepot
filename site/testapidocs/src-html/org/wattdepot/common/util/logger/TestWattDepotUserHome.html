001package org.wattdepot.common.util.logger;
002
003import static org.junit.Assert.assertEquals;
004
005import java.io.File;
006
007import org.junit.Test;
008import org.wattdepot.common.util.UserHome;
009
010import static org.wattdepot.common.util.UserHome.USER_HOME_PROPERTY;
011
012/**
013 * Tests the WattDepotUserHome class. Portions of this code are adapted from
014 * http://hackystat-utilities.googlecode.com/
015 * 
016 * @author Philip Johnson
017 * @author Robert Brewer
018 */
019
020public class TestWattDepotUserHome {
021
022  /**
023   * Tests the WattDepot user home definition facility.
024   * 
025   */
026  @Test
027  public void testHome() {
028    String wattdepotUserHome = System.getProperty(USER_HOME_PROPERTY);
029    File home = UserHome.getHome();
030
031    if (wattdepotUserHome == null) {
032      File userHomeFile = new File(System.getProperty("user.home"));
033      assertEquals("Checking default home", userHomeFile, home);
034    }
035    else {
036      File userHomeFile = new File(System.getProperty(USER_HOME_PROPERTY));
037      assertEquals("Checking overridden home", userHomeFile, home);
038    }
039  }
040}
