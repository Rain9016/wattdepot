301    int size = list.getDepositories().size();
302    assertTrue(list.getDepositories().size() >= 0);

304      // Put new instance (CREATE)
305      test.putDepository(depo);
306      list = test.getDepositories();

308      assertTrue(list.getDepositories().size() == size + 1);
309      assertTrue(test.isDefinedDepository(depo.getId()));
310    }
311    try {
312      // get instance (READ)
313      Depository ret = test.getDepository(depo.getId());
314      assertEquals(depo, ret);
315      ret.setId("new_slug");
316      // update instance (UPDATE)
317      try {
318        test.updateDepository(ret);
319        fail("Can't update depository.");
320      }
321      catch (Exception e) {
322        // expected.
323      }
324      list = test.getDepositories();
325      assertTrue(list.getDepositories().size() == size + 1);
326      // delete instance (DELETE)
327      test.deleteDepository(depo);
328      try {
329        ret = test.getDepository(depo.getName());
330        assertNull(ret);
331      }
332      catch (IdNotFoundException e) {
333        // this is what we want.
334      }
335    }
336    catch (IdNotFoundException e) {
337      fail("Should have " + depo);
338    }
