205  /**
206   * @throws Exception if there is a problem.
207   */
208  @Test
209  public void testMeasurementPruning() throws Exception {
210    populateMeasurements(2);
211    MeasurementPruner mgc = new MeasurementPruner(properties, gcd.getId(),
212        gcd.getOrganizationId(), false);
213    int numToDel = mgc.getMeasurementsToDelete().size();
214    mgc.run();
215    List<Measurement> toDel = mgc.getMeasurementsToDelete();
216    assertTrue(toDel.size() == 0);
217    WattDepotPersistence p = mgc.getPersistance();
218    List<Measurement> data = p.getMeasurements(gcd.getDepositoryId(), gcd.getOrganizationId(),
219        gcd.getSensorId(), false);
220    assertNotNull(data);
221    assertTrue(data.size() < totalMeasurements);
222    assertTrue(data.size() == totalMeasurements - numToDel);
223  }

331  /**
332   * @param time The time of the measurement.
333   * @return A fake measurement at the given time.
334   */
335  private Measurement makeMeasurement(XMLGregorianCalendar time) {
336    return new Measurement(InstanceFactory.getSensor().getId(), DateConvert.convertXMLCal(time),
337        123.0, Unit.valueOf(InstanceFactory.getMeasurementType().getUnits()));
338  }

340  /**
341   * @param minBetween The number of minutes between measurements.
342   * @throws MeasurementTypeException This shouldn't happen.
343   * @throws IdNotFoundException If there is a problem with the ids. This
344   *         shouldn't happen.
345   */
346  private void populateMeasurements(int minBetween) throws MeasurementTypeException,
347      IdNotFoundException {
348    // Need to add high frequency Measurements to the database.
349    now = Tstamp.makeTimestamp();
350    start = Tstamp.incrementDays(now, -1 * gcd.getIgnoreWindowDays());
351    end = Tstamp.incrementDays(start, -1 * gcd.getCollectWindowDays());
352    List<XMLGregorianCalendar> measTimes = Tstamp.getTimestampList(end, now, minBetween);
353    totalMeasurements = measTimes.size();
354    for (XMLGregorianCalendar cal : measTimes) {
355      impl.putMeasurement(depository.getId(), depository.getOrganizationId(), makeMeasurement(cal));
356    }
357  }
