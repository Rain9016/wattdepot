019package org.wattdepot.server.http.api;
020
021import java.text.ParseException;
022import java.util.Date;
023import java.util.logging.Level;
024
025import javax.xml.datatype.DatatypeConfigurationException;
026
027import org.restlet.data.Status;
028import org.restlet.resource.ResourceException;
029import org.wattdepot.common.domainmodel.Depository;
030import org.wattdepot.common.domainmodel.Labels;
031import org.wattdepot.common.domainmodel.Measurement;
032import org.wattdepot.common.domainmodel.MeasurementList;
033import org.wattdepot.common.domainmodel.Sensor;
034import org.wattdepot.common.exception.MissMatchedOwnerException;
035import org.wattdepot.common.util.DateConvert;
036
037/**
038 * DepositoryMeasurementsServer - Base class for handling the Depository measurements
039 * HTTP API ("/wattdepot/{org-id}/depository/{depository-id}/measurements/").
040 * 
041 * @author Cam Moore
042 *         Yongwen Xu
043 * 
044 */
045public class DepositoryMeasurementsServer extends WattDepotServerResource {
046  private String depositoryId;
047  private String sensorId;
048  private String start;
049  private String end;
050
051  /*
052   * (non-Javadoc)
053   * 
054   * @see org.restlet.resource.Resource#doInit()
055   */
056  @Override
057  protected void doInit() throws ResourceException {
058    super.doInit();
059    this.sensorId = getQuery().getValues(Labels.SENSOR);
060    this.start = getQuery().getValues(Labels.START);
061    this.end = getQuery().getValues(Labels.END);
062    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
063  }
064
065  /**
066   * retrieve the depository measurement list for a sensor.
067   * 
068   * @return measurement list. 
069   */
070  public MeasurementList doRetrieve() {
071    getLogger().log(Level.INFO, "GET /wattdepot/{" + orgId + "}/depository/{" + depositoryId
072        + "}/measurements/?sensor={" + sensorId + "}&start={" + start + "}&end={" + end + "}");
073    
074    if (start != null && end != null) {
075      MeasurementList ret = new MeasurementList();
076      try {
077        Depository depository = depot.getWattDeposiory(depositoryId, orgId);
078        if (depository != null) {
079          Sensor sensor = depot.getSensor(sensorId, orgId);
080          if (sensor != null) {            
081            Date startDate = DateConvert.parseCalStringToDate(start);
082            Date endDate = DateConvert.parseCalStringToDate(end);
083            if (startDate != null && endDate != null) {
084              for (Measurement meas : depository.getMeasurements(sensor, startDate, endDate)) {
085                ret.getMeasurements().add(meas);
086              }              
087            }
088            else {
089              setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Start date and/or end date missing.");
090            }
091          }
092          else {
093            setStatus(Status.CLIENT_ERROR_BAD_REQUEST, sensorId + " is not defined");
094          }
095        }
096        else {
097          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, depositoryId + " is not defined.");
098        }
099      }
100      catch (MissMatchedOwnerException e) {
101        setStatus(Status.CLIENT_ERROR_CONFLICT, e.getMessage());
102      }
103      catch (ParseException e) {
104        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
105      }
106      catch (DatatypeConfigurationException e) {
107        setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
108      }
109      getLogger().info(ret.toString());
110      return ret;
111    }
112    else {
113      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Missing start and/or end times.");
114      return null;
115    }
116  }
117
118}
