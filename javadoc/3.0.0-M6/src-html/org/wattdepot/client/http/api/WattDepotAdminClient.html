019package org.wattdepot.client.http.api;
020
021import org.restlet.resource.ClientResource;
022import org.wattdepot.client.WattDepotAdminInterface;
023import org.wattdepot.common.domainmodel.Labels;
024import org.wattdepot.common.domainmodel.Organization;
025import org.wattdepot.common.domainmodel.UserInfo;
026import org.wattdepot.common.domainmodel.UserPassword;
027import org.wattdepot.common.exception.BadCredentialException;
028import org.wattdepot.common.exception.IdNotFoundException;
029import org.wattdepot.common.http.api.OrganizationResource;
030import org.wattdepot.common.http.api.UserInfoResource;
031import org.wattdepot.common.http.api.UserPasswordResource;
032
033/**
034 * WattDepotAdminClient - Admin level client.
035 * 
036 * @author Cam Moore
037 * 
038 */
039public class WattDepotAdminClient extends WattDepotClient implements
040    WattDepotAdminInterface {
041
042  /**
043   * Creates a new WattDepotAdminClient.
044   * 
045   * @param serverUri
046   *          The URI of the WattDepot server (e.g.
047   *          "http://server.wattdepot.org/")
048   * @param username
049   *          The name of the user. The user must be defined in the WattDepot
050   *          server.
051   * @param password
052   *          The password for the user.
053   * @throws BadCredentialException
054   *           If the user or password don't match the credentials on the
055   *           WattDepot server.
056   */
057  public WattDepotAdminClient(String serverUri, String username, String password)
058      throws BadCredentialException {
059    super(serverUri, username, password);
060    if (!getGroupId().equals("admin")) {
061      throw new BadCredentialException("Wrong group.");
062    }
063  }
064
065  /*
066   * (non-Javadoc)
067   * 
068   * @see
069   * org.wattdepot.client.WattDepotAdminInterface#deleteUser(java.lang.String)
070   */
071  @Override
072  public void deleteUser(String id) throws IdNotFoundException {
073    ClientResource client = makeClient(getGroupId() + "/" + Labels.USER + "/"
074        + id);
075    UserInfoResource resource = client.wrap(UserInfoResource.class);
076    resource.remove();
077    client.release();
078  }
079
080  /*
081   * (non-Javadoc)
082   * 
083   * @see org.wattdepot.client.WattDepotAdminInterface#deleteUserGroup(java.lang
084   * .String)
085   */
086  @Override
087  public void deleteUserGroup(String id) throws IdNotFoundException {
088    ClientResource client = makeClient(getGroupId() + "/" + Labels.ORGANIZATION
089        + "/" + id);
090    OrganizationResource resource = client.wrap(OrganizationResource.class);
091    try {
092      resource.remove();
093    }
094    finally {
095      client.release();
096    }
097  }
098
099  /*
100   * (non-Javadoc)
101   * 
102   * @see
103   * org.wattdepot.client.WattDepotAdminInterface#deleteUserPassword(java.lang
104   * .String)
105   */
106  @Override
107  public void deleteUserPassword(String id) throws IdNotFoundException {
108    ClientResource client = makeClient(getGroupId() + "/"
109        + Labels.USER_PASSWORD + "/" + id);
110    UserPasswordResource resource = client.wrap(UserPasswordResource.class);
111    try {
112      resource.remove();
113    }
114    finally {
115      client.release();
116    }
117  }
118
119  /*
120   * (non-Javadoc)
121   * 
122   * @see
123   * org.wattdepot.client.WattDepotAdminInterface#putUser(org.wattdepot.datamodel
124   * .UserInfo)
125   */
126  @Override
127  public void putUser(UserInfo user) {
128    ClientResource client = makeClient(getGroupId() + "/" + Labels.USER + "/"
129        + user.getId());
130    UserInfoResource resource = client.wrap(UserInfoResource.class);
131    try {
132      resource.store(user);
133    }
134    finally {
135      client.release();
136    }
137  }
138
139  /*
140   * (non-Javadoc)
141   * 
142   * @see
143   * org.wattdepot.client.WattDepotAdminInterface#putUserGroup(org.wattdepot
144   * .datamodel.UserGroup)
145   */
146  @Override
147  public void putUserGroup(Organization group) {
148    ClientResource client = makeClient(getGroupId() + "/" + Labels.ORGANIZATION
149        + "/" + group.getId());
150    OrganizationResource resource = client.wrap(OrganizationResource.class);
151    try {
152      resource.store(group);
153    }
154    finally {
155      client.release();
156    }
157  }
158
159  /*
160   * (non-Javadoc)
161   * 
162   * @see
163   * org.wattdepot.client.WattDepotAdminInterface#putUserPassword(org.wattdepot
164   * .datamodel.UserPassword)
165   */
166  @Override
167  public void putUserPassword(UserPassword password) {
168    ClientResource client = makeClient(getGroupId() + "/"
169        + Labels.USER_PASSWORD + "/" + password.getId());
170    UserPasswordResource resource = client.wrap(UserPasswordResource.class);
171    resource.store(password);
172    client.release();
173  }
174
175}
