019package org.wattdepot.common.domainmodel;
020
021import java.util.Date;
022
023/**
024 * MeasuredValue - represents a measured value. Has sensor id, value and
025 * measurement type.
026 * 
027 * @author Cam Moore
028 * 
029 */
030public class MeasuredValue {
031  /** The id of the sensor making the measurement. */
032  private String sensorId;
033  /** The value of the measurement. */
034  private Double value;
035  /** The type of the measurement. */
036  private MeasurementType measurementType;
037  /** The time of the measured value. */
038  private Date date;
039
040  /**
041   * Hide the default constructor.
042   */
043  protected MeasuredValue() {
044
045  }
046
047  /**
048   * Creates a new MeasuredValue.
049   * 
050   * @param sensorId
051   *          The id of the sensor that made the measurement.
052   * @param value
053   *          The value of the measurement.
054   * @param measurementType
055   *          The type of the measurement.
056   */
057  public MeasuredValue(String sensorId, Double value, MeasurementType measurementType) {
058    this.sensorId = sensorId;
059    this.value = value;
060    this.measurementType = measurementType;
061  }
062
063  /**
064   * @return the sensorId
065   */
066  public String getSensorId() {
067    return sensorId;
068  }
069
070  /**
071   * @return the value
072   */
073  public Double getValue() {
074    return value;
075  }
076
077  /**
078   * @return the measurementType
079   */
080  public MeasurementType getMeasurementType() {
081    return measurementType;
082  }
083
084  /**
085   * @return the date
086   */
087  public Date getDate() {
088    return new Date(date.getTime());
089  }
090
091  /**
092   * @param date the date to set
093   */
094  public void setDate(Date date) {
095    this.date = new Date(date.getTime());
096  }
097
098  /**
099   * @param sensorId the sensorId to set
100   */
101  public void setSensorId(String sensorId) {
102    this.sensorId = sensorId;
103  }
104
105  /**
106   * @param value the value to set
107   */
108  public void setValue(Double value) {
109    this.value = value;
110  }
111
112  /**
113   * @param measurementType the measurementType to set
114   */
115  public void setMeasurementType(MeasurementType measurementType) {
116    this.measurementType = measurementType;
117  }
118
119}
