019package org.wattdepot.server.depository.impl.hibernate;
020
021import org.hibernate.SessionFactory;
022import org.hibernate.cfg.Configuration;
023import org.hibernate.service.ServiceRegistry;
024import org.hibernate.service.ServiceRegistryBuilder;
025import org.wattdepot.server.ServerProperties;
026
027/**
028 * Manager - provides access to the singleton SessionFactory needed to persist
029 * the objects.
030 * 
031 * @author Cam Moore
032 * 
033 */
034public class Manager {
035  private static SessionFactory validateFactory;
036  private static ServiceRegistry validateRegistry;
037  private static SessionFactory sessionFactory;
038  private static ServiceRegistry serviceRegistry;
039
040  /**
041   * @param properties The ServerProperties that contain the database
042   *        configuration.
043   * @return The singleton SessionFactory.
044   */
045  public static SessionFactory getFactory(ServerProperties properties) {
046    if (sessionFactory == null) {
047      Configuration cfg = new Configuration()
048          .addAnnotatedClass(
049              org.wattdepot.server.depository.impl.hibernate.CollectorProcessDefinitionImpl.class)
050          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.DepositoryImpl.class)
051          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.DepositorySensorContribution.class)
052          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.MeasurementPruningDefinitionImpl.class)
053          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.MeasurementImpl.class)
054          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.MeasurementTypeImpl.class)
055          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.PropertyImpl.class)
056          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.SensorImpl.class)
057          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.SensorGroupImpl.class)
058          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.SensorModelImpl.class)
059          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.OrganizationImpl.class)
060          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.UserInfoImpl.class)
061          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.UserPasswordImpl.class)
062          .setProperty("hibernate.connection.driver_class",
063              properties.get(ServerProperties.DB_CONNECTION_DRIVER))
064          .setProperty("hibernate.connection.url",
065              properties.get(ServerProperties.DB_CONNECTION_URL))
066          .setProperty("hibernate.connection.username",
067              properties.get(ServerProperties.DB_USER_NAME))
068          .setProperty("hibernate.connection.password",
069              properties.get(ServerProperties.DB_PASSWORD))
070          .setProperty("hibernate.c3p0.min_size", "5").setProperty("hibernate.c3p0.max_size", "20")
071          .setProperty("hibernate.c3p0.timeout", "1800")
072          .setProperty("hibernate.c3p0.max_statements", "50")
073          .setProperty("hibernate.dialect", "org.hibernate.dialect.PostgreSQLDialect")
074          .setProperty("hibernate.show_sql", properties.get(ServerProperties.DB_SHOW_SQL))
075          .setProperty("hibernate.hbm2ddl.auto", properties.get(ServerProperties.DB_TABLE_UPDATE));
076      serviceRegistry = new ServiceRegistryBuilder().applySettings(cfg.getProperties())
077          .buildServiceRegistry();
078      // A SessionFactory is set up once for an application
079      sessionFactory = cfg.buildSessionFactory(serviceRegistry);
080
081      // // A SessionFactory is set up once for an application
082//       sessionFactory = new Configuration().configure() // configures settings
083//       // from
084//       // hibernate.cfg.xml
085//       .buildSessionFactory();
086
087    }
088    return sessionFactory;
089  }
090
091  /**
092   * @param properties The ServerProperties that contain the database
093   *        configuration.
094   * @return The singleton SessionFactory.
095   */
096  public static SessionFactory getValidateFactory(ServerProperties properties) {
097    if (validateFactory == null) {
098      Configuration cfg = new Configuration()
099          .addAnnotatedClass(
100              org.wattdepot.server.depository.impl.hibernate.CollectorProcessDefinitionImpl.class)
101          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.DepositoryImpl.class)
102          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.MeasurementImpl.class)
103          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.MeasurementTypeImpl.class)
104          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.PropertyImpl.class)
105          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.SensorImpl.class)
106          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.SensorGroupImpl.class)
107          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.SensorModelImpl.class)
108          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.OrganizationImpl.class)
109          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.UserInfoImpl.class)
110          .addAnnotatedClass(org.wattdepot.server.depository.impl.hibernate.UserPasswordImpl.class)
111          .setProperty("hibernate.connection.driver_class",
112              properties.get(ServerProperties.DB_CONNECTION_DRIVER))
113          .setProperty("hibernate.connection.url",
114              properties.get(ServerProperties.DB_CONNECTION_URL))
115          .setProperty("hibernate.connection.username",
116              properties.get(ServerProperties.DB_USER_NAME))
117          .setProperty("hibernate.connection.password",
118              properties.get(ServerProperties.DB_PASSWORD))
119          .setProperty("hibernate.c3p0.min_size", "5").setProperty("hibernate.c3p0.max_size", "20")
120          .setProperty("hibernate.c3p0.timeout", "1800")
121          .setProperty("hibernate.c3p0.max_statements", "50")
122          .setProperty("hibernate.dialect", "org.hibernate.dialect.PostgreSQLDialect")
123          .setProperty("hibernate.show_sql", properties.get(ServerProperties.DB_SHOW_SQL))
124          .setProperty("hibernate.hbm2ddl.auto", "validate");
125      validateRegistry = new ServiceRegistryBuilder().applySettings(cfg.getProperties())
126          .buildServiceRegistry();
127      // A SessionFactory is set up once for an application
128      validateFactory = cfg.buildSessionFactory(validateRegistry);
129
130    }
131    return validateFactory;
132  }
133  
134  /**
135   * Closes the SessionFactory. Should be called when shutting down the system.
136   */
137  public static void closeSession() {
138    if (sessionFactory != null) {
139      sessionFactory.close();
140      sessionFactory = null;
141    }
142    if (validateFactory != null) {
143      validateFactory.close();
144      validateFactory = null;
145    }
146  }
147}
