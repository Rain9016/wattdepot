046/**
047 * AdministratorServerResource - Administrative interface for WattDepot. It
048 * handles the HTTP API ("/wattdepot/{org-id}/").
049 * 
050 * @author Cam Moore
051 * 
052 */
053public class AdminServerResource extends WattDepotServerResource {
054
055  /**
056   * @return The admin user interface as an HTML Representation.
057   */
058  @Get()
059  public Representation toHtml() {
060    getLogger().log(Level.INFO, "GET " + API.BASE_URI + "{" + orgId + "}/");
061    if (!isInRole(orgId) && !isInRole(Organization.ADMIN_GROUP.getId())) {
062      User user = getClientInfo().getUser();
063      // The user isn't in the orgId. Search for the user in other
064      // organizations.
065      boolean foundUser = false;
066      for (UserInfo ui : depot.getUsers()) {
067        if (ui.getUid().equals(user.getIdentifier()) && isInRole(ui.getOrganizationId())) {
068          redirectPermanent("/wattdepot/" + ui.getOrganizationId() + "/");
069          foundUser = true;
070        }
071      }
072      if (!foundUser) {
073        setStatus(Status.CLIENT_ERROR_BAD_REQUEST);
074      }
075    }
076    else {
077      Map<String, Object> dataModel = new HashMap<String, Object>();
078      dataModel.put("orgId", orgId);
079      Representation rep = null;
080      TemplateRepresentation template = null;
081      // decide what to show based upon the orgId.
082      if (orgId.equals(Organization.ADMIN_GROUP.getId())) {
083        // admin can manipulate users and organizations
084        List<UserInfo> users = depot.getUsers();
085        List<Organization> orgs = depot.getOrganizations();
086        dataModel.put("users", users);
087        dataModel.put("orgs", orgs);
088        dataModel.put("rootUid", UserInfo.ROOT.getUid());
089        dataModel.put("adminId", Organization.ADMIN_GROUP.getId());
090        rep = new ClientResource(LocalReference.createClapReference(getClass().getPackage())
091            + "/UserAdmin.ftl").get();
092      }
093      else {
094        try {
095          // regular organization, can manipulate sensors, depositories
096          depot.getOrganization(orgId, true);
097          List<Depository> depos = depot.getDepositories(orgId, false);
098          List<Sensor> sensors = depot.getSensors(orgId, false);
099          List<SensorModel> sensorModels = depot.getSensorModels();
100          List<SensorGroup> sensorGroups = depot.getSensorGroups(orgId, false);
101          List<CollectorProcessDefinition> cpds = depot
102              .getCollectorProcessDefinitions(orgId, false);
103          List<MeasurementType> measurementTypes = depot.getMeasurementTypes();
104          List<GarbageCollectionDefinition> gcds = depot.getGarbageCollectionDefinitions(orgId,
105              false);
106          dataModel.put("depositories", depos);
107          dataModel.put("sensors", sensors);
108          dataModel.put("sensorgroups", sensorGroups);
109          dataModel.put("sensormodels", sensorModels);
110          dataModel.put("cpds", cpds);
111          dataModel.put("measurementtypes", measurementTypes);
112          dataModel.put("gcds", gcds);
113          getLogger().log(Level.INFO, gcds.toString());
114          rep = new ClientResource(LocalReference.createClapReference(getClass().getPackage())
115              + "/OrganizationAdmin.ftl").get();
116        }
117        catch (IdNotFoundException e) {
118          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
119        }
120      }
121      template = new TemplateRepresentation(rep, dataModel, MediaType.TEXT_HTML);
122      return template;
123    }
124    return null;
125  }
126}
