044/**
045 * DepositoryAverageValuesServer - Base class for handling HTTP API
046 * ("/wattdepot/{org-id}/depository/{depository-id}/values/average/").
047 * 
048 * @author Cam Moore
049 * 
050 */
051public class DepositoryAverageValuesServer extends WattDepotServerResource {
052  private String depositoryId;
053  private String sensorId;
054  /** The start of the range. */
055  private String start;
056  /** The end of the range. */
057  private String end;
058  /** The interval to calculate the average over. */
059  private String interval;
060  private String dataType;
061
062  /*
063   * (non-Javadoc)
064   * 
065   * @see org.restlet.resource.Resource#doInit()
066   */
067  @Override
068  protected void doInit() throws ResourceException {
069    super.doInit();
070    this.sensorId = getQuery().getValues(Labels.SENSOR);
071    this.start = getQuery().getValues(Labels.START);
072    this.end = getQuery().getValues(Labels.END);
073    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
074    this.interval = getQuery().getValues(Labels.INTERVAL);
075    this.dataType = getQuery().getValues(Labels.VALUE_TYPE);
076  }
077
078  /**
079   * retrieve the depository measurement list for a sensor.
080   * 
081   * @return measurement list.
082   */
083  public MeasuredValueList doRetrieve() {
084    getLogger().log(
085        Level.INFO,
086        "GET /wattdepot/{" + orgId + "}/" + Labels.DEPOSITORY + "/{" + depositoryId + "}/"
087            + Labels.VALUES + "/" + Labels.AVERAGE + "/?" + Labels.SENSOR + "={" + sensorId + "}&"
088            + Labels.START + "={" + start + "}&" + Labels.END + "={" + end + "}&" + Labels.INTERVAL
089            + "={" + interval + "}&" + Labels.VALUE_TYPE + "={" + dataType + "}");
090    if (isInRole(orgId)) {
091      if (start != null && end != null && interval != null) {
092        MeasuredValueList ret = new MeasuredValueList();
093        try {
094          Depository depository = depot.getDepository(depositoryId, orgId, true);
095          XMLGregorianCalendar startRange = DateConvert.parseCalString(start);
096          XMLGregorianCalendar endRange = DateConvert.parseCalString(end);
097          int intervalMinutes = Integer.parseInt(interval);
098          List<XMLGregorianCalendar> rangeList = Tstamp.getTimestampList(startRange, endRange,
099              intervalMinutes);
100          if (rangeList != null) {
101            for (int i = 1; i < rangeList.size(); i++) {
102              Date valueDate = DateConvert.convertXMLCal(rangeList.get(i));
103              XMLGregorianCalendar startInterval = rangeList.get(i - 1);
104              XMLGregorianCalendar endInterval = rangeList.get(i);
105              List<XMLGregorianCalendar> intervalList = Tstamp.getNTimestampList(24, startInterval,
106                  endInterval);
107              DescriptiveStatistics stats = new DescriptiveStatistics();
108              Date previous = null;
109              for (int j = 0; j < intervalList.size(); j++) {
110                Date timestamp = DateConvert.convertXMLCal(intervalList.get(j));
111                Double value = null;
112                if ("point".equals(dataType)) {
113                  value = getValue(depositoryId, orgId, sensorId, timestamp);
114                }
115                else {
116                  if (previous != null) {
117                    value = getValue(depositoryId, orgId, sensorId, previous, timestamp);
118                  }
119                  previous = timestamp;
120                }
121                if (value != null) {
122                  stats.addValue(value);
123                }
124              }
125
126              if (!Double.isNaN(stats.getMean())) {
127                ret.getMeasuredValues().add(
128                    new InterpolatedValue(sensorId, stats.getMean(), depository
129                        .getMeasurementType(), valueDate));
130              }
131            }
132          }
133        }
134        catch (IdNotFoundException e) {
135          // TODO Auto-generated catch block
136          e.printStackTrace();
137        }
138        catch (ParseException e) {
139          // TODO Auto-generated catch block
140          e.printStackTrace();
141        }
142        catch (DatatypeConfigurationException e) {
143          // TODO Auto-generated catch block
144          e.printStackTrace();
145        }
146        return ret;
147      }
148      else {
149        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Missing start and/or end times or interval.");
150        return null;
151      }
152    }
153    else {
154      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Bad credentials.");
155      return null;
156    }
157  }
158
159  /**
160   * @param depositoryId The depository.
161   * @param orgId The organization.
162   * @param sensorId The sensor.
163   * @param timestamp When to get the value.
164   * @return The value.
165   */
166  private Double getValue(String depositoryId, String orgId, String sensorId, Date timestamp) {
167    try {
168      Sensor sensor = depot.getSensor(sensorId, orgId, false);
169      Double value = 0.0;
170      if (sensor != null) { // just get sensor value
171        try {
172          value = depot.getValue(depositoryId, orgId, sensor.getId(), timestamp, true);
173        }
174        catch (NoMeasurementException e) {
175          return null;
176        }
177      }
178      else { // check for sensor group
179        SensorGroup group = depot.getSensorGroup(sensorId, orgId, false);
180        if (group != null) {
181          for (String s : group.getSensors()) {
182            try {
183              value += depot.getValue(depositoryId, orgId, s, timestamp, true);
184            }
185            catch (NoMeasurementException e) { // NOPMD
186              // add 0 to value so do nothing.
187            }
188          }
189        }
190      }
191      return value;
192    }
193    catch (MisMatchedOwnerException e) {
194      return null;
195    }
196    catch (IdNotFoundException e) {
197      return null;
198    }
199  }

237      return null;
238    }
239    catch (IdNotFoundException e) {
240      return null;
241    }
