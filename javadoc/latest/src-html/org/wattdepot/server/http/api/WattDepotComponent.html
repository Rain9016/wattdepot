040/**
041 * WattDepotComponent - Main class to start the WattDepot Http API component of
042 * the WattDepotServer.
043 * 
044 * @author Cam Moore
045 * 
046 */
047public class WattDepotComponent extends Component {
048
049  /**
050   * Sets up the WattDepotComponent with the given WattDepot.
051   * 
052   * @param depot
053   *          The persitent store.
054   * @param port
055   *          the port number on which the restlet server listens.
056   */
057  public WattDepotComponent(WattDepotPersistence depot, int port) {
058    setName("WattDepot HTTP API Server");
059    setDescription("WattDepot RESTful server.");
060    setAuthor("Cam Moore");
061    getLogService().setLoggerName("org.wattdepot.server");
062    // Add a CLAP client connector
063    getClients().add(Protocol.CLAP);
064    getClients().add(Protocol.FILE);
065    getClients().add(Protocol.HTTP);
066
067    // Adds a HTTP server connector
068    Server server = getServers().add(Protocol.HTTP, port);
069    server.getContext().getParameters().set("tracing", "true");
070
071    WattDepotApplication app = new WattDepotApplication();
072    app.setDepot(depot);
073    // configure the JacksonRepresentation so that it uses ISO-8601 compliant
074    // notation
075    MeasurementImpl source = new MeasurementImpl();
076    Representation rep = app.getConverterService().toRepresentation(source);
077    @SuppressWarnings("rawtypes")
078    ObjectMapper mapper = ((JacksonRepresentation) rep).getObjectMapper();
079    mapper.configure(SerializationConfig.Feature.WRITE_DATES_AS_TIMESTAMPS,
080        false);
081    mapper.configure(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES,
082        false);
083    getDefaultHost().attachDefault(app);
084    app.setComponent(this);
085
086    // Set up the security realm
087    MemoryRealm realm = new MemoryRealm();
088    realm.setName("WattDepot Security");
089    getRealms().add(realm);
090    for (Organization group : app.getDepot().getOrganizations()) {
091      app.getRoles().add(new Role(group.getId()));
092      for (String userId : group.getUsers()) {
093        UserPassword up = null;
094        try {
095          up = app.getDepot().getUserPassword(userId, group.getId());
096        }
097        catch (IdNotFoundException e) {
098          e.printStackTrace();
099        }
100        if (up != null) {
101          User user = null;
102          if (userId.equals(UserInfo.ROOT.getUid())) {
103            // There isn't a UserInfo stored in persistence for ROOT.
104            user = new User(UserInfo.ROOT.getUid(), StrongAES.getInstance()
105                .decrypt(up.getEncryptedPassword()),
106                UserInfo.ROOT.getFirstName(), UserInfo.ROOT.getLastName(),
107                UserInfo.ROOT.getEmail());
108          }
109          else {
110            UserInfo info;
111            try {
112              info = app.getDepot().getUser(userId, group.getId());
113              user = new User(info.getUid(), StrongAES.getInstance().decrypt(
114                  up.getEncryptedPassword()), info.getFirstName(),
115                  info.getLastName(), info.getEmail());
116            }
117            catch (IdNotFoundException e) {
118              e.printStackTrace();
119            }
120          }
121          realm.getUsers().add(user);
122          realm.map(user, app.getRole(group.getId()));
123        }
124      }
125    }
126
127    // Set the realm's default enroler and verifier
128    app.getContext().setDefaultEnroler(realm.getEnroler());
129    app.getContext().setDefaultVerifier(realm.getVerifier());
130
131//    Properties props = new Properties();
132//    InputStream inputStream = this.getClass().getClassLoader()
133//        .getResourceAsStream("log.properties");
134//    if (inputStream != null) {
135//      try {
136//        props.load(inputStream);
137//      }
138//      catch (IOException e) {
139//        // TODO Auto-generated catch block
140//        e.printStackTrace();
141//      }
142//    }
143//    // // // Configure the log service
144//    getLogService().setLoggerName("WattDepot.AccessLog");
145//    try {
146//      LogManager.getLogManager().readConfiguration(inputStream);
147//    }
148//    catch (SecurityException e) {
149//      // TODO Auto-generated catch block
150//      e.printStackTrace();
151//    }
152//    catch (IOException e) {
153//      // TODO Auto-generated catch block
154//      e.printStackTrace();
155//    }
156    //
157    // getLogService().setLogPropertiesRef("log.properties");
158
159  }
