038/**
039 * WattDepotComponent - Main class to start the WattDepot Http API component of
040 * the WattDepotServer.
041 * 
042 * @author Cam Moore
043 * 
044 */
045public class WattDepotComponent extends Component {
046
047  /**
048   * Sets up the WattDepotComponent with the given WattDepot.
049   * 
050   * @param depot
051   *          The persitent store.
052   * @param port
053   *          the port number on which the restlet server listens.
054   */
055  public WattDepotComponent(WattDepotPersistence depot, int port) {
056    setName("WattDepot HTTP API Server");
057    setDescription("WattDepot RESTful server.");
058    setAuthor("Cam Moore");
059    getLogService().setLoggerName("org.wattdepot.server");
060    // Add a CLAP client connector
061    getClients().add(Protocol.CLAP);
062    getClients().add(Protocol.FILE);
063    getClients().add(Protocol.HTTP);
064
065    // Adds a HTTP server connector
066    Server server = getServers().add(Protocol.HTTP, port);
067    server.getContext().getParameters().set("tracing", "true");
068
069    WattDepotApplication app = new WattDepotApplication();
070    app.setDepot(depot);
071    // configure the JacksonRepresentation so that it uses ISO-8601 compliant
072    // notation
073    MeasurementImpl source = new MeasurementImpl();
074    Representation rep = app.getConverterService().toRepresentation(source);
075    @SuppressWarnings("rawtypes")
076    ObjectMapper mapper = ((JacksonRepresentation) rep).getObjectMapper();
077    mapper.configure(SerializationConfig.Feature.WRITE_DATES_AS_TIMESTAMPS,
078        false);
079    mapper.configure(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES,
080        false);
081    getDefaultHost().attachDefault(app);
082    app.setComponent(this);
083
084    // Set up the security realm
085    MemoryRealm realm = new MemoryRealm();
086    realm.setName("WattDepot Security");
087    getRealms().add(realm);
088    for (Organization group : app.getDepot().getOrganizations()) {
089      app.getRoles().add(new Role(group.getId()));
090      for (UserInfo info : group.getUsers()) {
091        UserPassword up = app.getDepot().getUserPassword(info.getId());
092        User user = new User(info.getId(), up.getPlainText(),
093            info.getFirstName(), info.getLastName(), info.getEmail());
094        realm.getUsers().add(user);
095        realm.map(user, app.getRole(group.getId()));
096      }
097    }
098
099    // Set the realm's default enroler and verifier
100    app.getContext().setDefaultEnroler(realm.getEnroler());
101    app.getContext().setDefaultVerifier(realm.getVerifier());
102
103//    Properties props = new Properties();
104//    InputStream inputStream = this.getClass().getClassLoader().getResourceAsStream("log.properties");
105//    if (inputStream != null) {
106//      try {
107//        props.load(inputStream);
108//      }
109//      catch (IOException e) {
110//        // TODO Auto-generated catch block
111//        e.printStackTrace();
112//      }
113//    }
114//    // // Configure the log service
115//    getLogService().setLoggerName("WattDepot.AccessLog");
116//    try {
117//      LogManager.getLogManager().readConfiguration(
118//          inputStream);
119//    }
120//    catch (SecurityException e) {
121//      // TODO Auto-generated catch block
122//      e.printStackTrace();
123//    }
124//    catch (IOException e) {
125//      // TODO Auto-generated catch block
126//      e.printStackTrace();
127//    }
128//    
129//    getLogService().setLogPropertiesRef(
130//        "log.properties");
131
132  }
