583  /**
584   * Ensures the base set of MeasurementTypes are defined in WattDepot.
585   */
586  public void initializeMeasurementTypes() {
587    for (String key : UnitsHelper.quantities.keySet()) {
588      MeasurementType mt = getMeasurementType(Slug.slugify(key));
589      if (mt == null) {
590        try {
591          defineMeasurementType(key, UnitsHelper.quantities.get(key).toString());
592        }
593        catch (UniqueIdException e) {
594          e.printStackTrace();
595        }
596      }
597    }
598  }
599
600  /**
601   * Ensures the base set of SensorModels are defined in WattDepot.
602   */
603  public void initializeSensorModels() {
604    for (String key : SensorModelHelper.models.keySet()) {
605      SensorModel sm = getSensorModel(Slug.slugify(key));
606      if (sm == null) {
607        SensorModel model = SensorModelHelper.models.get(key);
608        try {
609          defineSensorModel(model.getName(), model.getProtocol(), model.getType(),
610              model.getVersion());
611        }
612        catch (UniqueIdException e) {
613          e.printStackTrace();
614        }
615      }
616    }
617
618  }
