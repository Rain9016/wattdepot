069   * processes.

074   * @param pollingInterval The polling interval.

695  /**
696   * Ensures the base set of MeasurementTypes are defined in WattDepot.
697   */
698  public void initializeMeasurementTypes() {
699    for (String key : UnitsHelper.quantities.keySet()) {
700      try {
701        getMeasurementType(Slug.slugify(key));
702      }
703      catch (IdNotFoundException e1) {
704        try {
705          defineMeasurementType(Slug.slugify(key), key, UnitsHelper.quantities
706              .get(key).toString());
707        }
708        catch (UniqueIdException e) {
709          e.printStackTrace();
710        }
711        catch (BadSlugException e) {
712          e.printStackTrace();
713        }
714      }
715    }
716  }
717
718  /**
719   * Ensures the base set of SensorModels are defined in WattDepot.
720   */
721  public void initializeSensorModels() {
722    for (String key : SensorModelHelper.models.keySet()) {
723      try {
724        getSensorModel(Slug.slugify(key));
725      }
726      catch (IdNotFoundException e1) {
727        SensorModel model = SensorModelHelper.models.get(key);
728        try {
729          defineSensorModel(Slug.slugify(model.getName()), model.getName(),
730              model.getProtocol(), model.getType(), model.getVersion());
731        }
732        catch (UniqueIdException e) {
733          e.printStackTrace();
734        }
735        catch (BadSlugException e) {
736          e.printStackTrace();
737        }
738      }
739    }
740
