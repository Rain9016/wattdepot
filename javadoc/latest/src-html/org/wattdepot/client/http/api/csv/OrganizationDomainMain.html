019package org.wattdepot.client.http.api.csv;
020
021import java.io.IOException;
022
023import org.apache.commons.cli.CommandLine;
024import org.apache.commons.cli.CommandLineParser;
025import org.apache.commons.cli.HelpFormatter;
026import org.apache.commons.cli.Options;
027import org.apache.commons.cli.ParseException;
028import org.apache.commons.cli.PosixParser;
029import org.wattdepot.common.exception.BadCredentialException;
030import org.wattdepot.common.exception.BadSensorUriException;
031import org.wattdepot.common.exception.IdNotFoundException;
032
033/**
034 * OrganizationDomainMain - Reads in an Organization domain definition file and
035 * ensures the WattDepot server has all the instances defined in the file.
036 * 
037 * @author Cam Moore
038 * 
039 */
040public class OrganizationDomainMain {
041
042  /**
043   * @param args command line arguments -s <server uri> -u <username> -p
044   *        <password> -o <orgId> -f <fileName> [-d].
045   * @throws BadSensorUriException if there is a problem with the WattDepot
046   *         sensor definition.
047   * @throws IdNotFoundException if there is a problem with the organization id.
048   * @throws BadCredentialException if the credentials are not valid.
049   * @throws IOException if there is a problem with reading the file.
050   */
051  public static void main(String[] args) throws BadCredentialException,
052      IdNotFoundException, BadSensorUriException, IOException {
053    Options options = new Options();
054    CommandLine cmd = null;
055    String serverUri = null;
056    String username = null;
057    String organizationId = null;
058    String password = null;
059    String fileName = null;
060    boolean debug = false;
061
062    options.addOption("h", false,
063        "Usage: OrganizationDomainMain -s <server uri> -u <username>"
064            + " -p <password> -o <orgId> -f <fileName> [-d]");
065    options.addOption("s", "server", true,
066        "WattDepot Server URI. (http://server.wattdepot.org)");
067    options.addOption("u", "username", true, "Username");
068    options.addOption("o", "organizationId", true, "User's Organization id.");
069    options.addOption("p", "password", true, "Password");
070    options.addOption("f", "fileName", true, "Domain definition file name.");
071    options.addOption("d", "debug", false,
072        "Displays debugging information.");
073
074    CommandLineParser parser = new PosixParser();
075    HelpFormatter formatter = new HelpFormatter();
076    try {
077      cmd = parser.parse(options, args);
078    }
079    catch (ParseException e) {
080      System.err.println("Command line parsing failed. Reason: "
081          + e.getMessage() + ". Exiting.");
082      System.exit(1);
083    }
084    if (cmd.hasOption("h")) {
085      formatter.printHelp("OrganizationDomainMain", options);
086      System.exit(0);
087    }
088    if (cmd.hasOption("s")) {
089      serverUri = cmd.getOptionValue("s");
090    }
091    else {
092      serverUri = "http://server.wattdepot.org/";
093    }
094    if (cmd.hasOption("u")) {
095      username = cmd.getOptionValue("u");
096    }
097    else {
098      username = "user";
099    }
100    if (cmd.hasOption("p")) {
101      password = cmd.getOptionValue("p");
102    }
103    else {
104      password = "default";
105    }
106    if (cmd.hasOption("o")) {
107      organizationId = cmd.getOptionValue("o");
108    }
109    else {
110      organizationId = "organization";
111    }
112    if (cmd.hasOption("f")) {
113      fileName = cmd.getOptionValue("f");
114    }
115    else {
116      fileName = "organization.csv";
117    }
118    debug = cmd.hasOption("d");
119    if (debug) {
120      System.out.println("Organization Domain Main:");
121      System.out.println("    WattDepotServer: " + serverUri);
122      System.out.println("    Username: " + username);
123      System.out.println("    OrganizationId: " + organizationId);
124      System.out.println("    Password: ********");
125      System.out.println("    fileName: " + fileName);
126    }
127    OrganizationDomain od = new OrganizationDomain(serverUri, username,
128        organizationId, password, fileName);
129    od.initializeFromFile();
130  }
131
132}
