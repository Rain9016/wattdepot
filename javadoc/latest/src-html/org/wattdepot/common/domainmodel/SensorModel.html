019package org.wattdepot.common.domainmodel;
020
021import org.wattdepot.common.util.Slug;
022
023/**
024 * SensorModel - Information about a Sensor.
025 * 
026 * @author Cam Moore
027 * 
028 */
029public class SensorModel {
030  /** A unique id for the SensorModel. */
031  private String id;
032  /** The name for the SensorModel. */
033  private String name;
034  /** The protocol this sensor uses. */
035  protected String protocol;
036  /** The type of the sensor. */
037  protected String type;
038  /** The version of the sensor model. */
039  protected String version;
040
041  /**
042   * The default constructor.
043   */
044  public SensorModel() {
045
046  }
047
048  /**
049   * @param name
050   *          The name.
051   * @param protocol
052   *          The protocol used by a sensor.
053   * @param type
054   *          The type of the sensor.
055   * @param version
056   *          The version the sensor is using.
057   */
058  public SensorModel(String name, String protocol, String type, String version) {
059    this.id = Slug.slugify(name);
060    this.name = name;
061    this.protocol = protocol;
062    this.type = type;
063    this.version = version;
064  }
065
066  /*
067   * (non-Javadoc)
068   * 
069   * @see java.lang.Object#equals(java.lang.Object)
070   */
071  @Override
072  public boolean equals(Object obj) {
073    if (this == obj) {
074      return true;
075    }
076    if (obj == null) {
077      return false;
078    }
079    if (getClass() != obj.getClass()) {
080      return false;
081    }
082    SensorModel other = (SensorModel) obj;
083    if (id == null) {
084      if (other.id != null) {
085        return false;
086      }
087    }
088    else if (!id.equals(other.id)) {
089      return false;
090    }
091    if (protocol == null) {
092      if (other.protocol != null) {
093        return false;
094      }
095    }
096    else if (!protocol.equals(other.protocol)) {
097      return false;
098    }
099    if (type == null) {
100      if (other.type != null) {
101        return false;
102      }
103    }
104    else if (!type.equals(other.type)) {
105      return false;
106    }
107    if (version == null) {
108      if (other.version != null) {
109        return false;
110      }
111    }
112    else if (!version.equals(other.version)) {
113      return false;
114    }
115    return true;
116  }
117
118  /**
119   * @return the id
120   */
121  public String getId() {
122    return id;
123  }
124
125  /**
126   * @return the name
127   */
128  public String getName() {
129    return name;
130  }
131
132  /**
133   * @return the protocol
134   */
135  public String getProtocol() {
136    return protocol;
137  }
138
139  /**
140   * @return the type
141   */
142  public String getType() {
143    return type;
144  }
145
146  /**
147   * @return the version
148   */
149  public String getVersion() {
150    return version;
151  }
152
153  /*
154   * (non-Javadoc)
155   * 
156   * @see java.lang.Object#hashCode()
157   */
158  @Override
159  public int hashCode() {
160    final int prime = 31;
161    int result = 1;
162    result = prime * result + ((protocol == null) ? 0 : protocol.hashCode());
163    result = prime * result + ((type == null) ? 0 : type.hashCode());
164    result = prime * result + ((version == null) ? 0 : version.hashCode());
165    return result;
166  }
167
168  /**
169   * @param id
170   *          the id to set
171   */
172  public void setId(String id) {
173    this.id = id;
174  }
175
176  /**
177   * @param name
178   *          the name to set
179   */
180  public void setName(String name) {
181    this.name = name;
182    if (this.id == null) {
183      this.id = Slug.slugify(name);
184    }
185  }
186
187  /**
188   * @param protocol
189   *          the protocol to set
190   */
191  public void setProtocol(String protocol) {
192    this.protocol = protocol;
193  }
194
195  /**
196   * @param type
197   *          the type to set
198   */
199  public void setType(String type) {
200    this.type = type;
201  }
202
203  /**
204   * @param version
205   *          the version to set
206   */
207  public void setVersion(String version) {
208    this.version = version;
209  }
210
211  /*
212   * (non-Javadoc)
213   * 
214   * @see java.lang.Object#toString()
215   */
216  @Override
217  public String toString() {
218    return "SensorModel [id=" + id + ", name=" + name + ", protocol=" + protocol + ", type=" + type
219        + ", version=" + version + "]";
220  }
221
222}
