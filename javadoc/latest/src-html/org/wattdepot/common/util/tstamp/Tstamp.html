030  /**
031   * Returns true if the passed string can be parsed into an
032   * XMLGregorianCalendar object.
033   * 
034   * @param lexicalRepresentation The string representation.
035   * @return True if the string is a legal XMLGregorianCalendar.
036   */
037  public static boolean isTimestamp(String lexicalRepresentation) {
038    try {
039      DatatypeFactory factory = DatatypeFactory.newInstance();
040      factory.newXMLGregorianCalendar(lexicalRepresentation);
041      return true;
042
043    }
044    catch (Exception e) {
045      return false;
046    }
047  }

049  /**
050   * Returns an XMLGregorianCalendar, given its string representation. Missing
051   * hours, minutes, second, millisecond, and timezone fields are given
052   * defaults.
053   * 
054   * @param rep The string representation.
055   * @return The timestamp.
056   * @throws Exception If the string cannot be parsed into a timestamp.
057   */
058  public static XMLGregorianCalendar makeTimestamp(String rep) throws Exception {
059    DatatypeFactory factory = DatatypeFactory.newInstance();
060    long mills = factory.newXMLGregorianCalendar(rep).toGregorianCalendar().getTimeInMillis();
061    return makeTimestamp(mills);
062  }

065   * Converts a javax.sql.Timestamp into a
066   * javax.xml.datatype.XMLGregorianCalendar.
067   * 
068   * @param tstamp The javax.sql.Timestamp
069   * @return A new instance of a javax.xml.datatype.XmlGregorianCalendar

071  public static XMLGregorianCalendar makeTimestamp(java.sql.Timestamp tstamp) {
072    DatatypeFactory factory = null;
073    try {
074      factory = DatatypeFactory.newInstance();

077      return factory.newXMLGregorianCalendar(calendar);
078    }

123  /**
124   * Returns a new XMLGregorianCalendar corresponding to the passed tstamp
125   * incremented by the number of days.
126   * 
127   * @param tstamp The base date and time.
128   * @param days The number of days to increment. This can be a negative number.
129   * @return A new XMLGregorianCalendar instance representing the inc'd time.
130   */
131  public static XMLGregorianCalendar incrementDays(XMLGregorianCalendar tstamp, int days) {
132    DatatypeFactory factory = null;
133    try {
134      factory = DatatypeFactory.newInstance();
135      GregorianCalendar calendar = new GregorianCalendar();
136      long millis = tstamp.toGregorianCalendar().getTimeInMillis();
137      millis += 1000L * 60 * 60 * 24 * days;
138      calendar.setTimeInMillis(millis);
139      return factory.newXMLGregorianCalendar(calendar);
140    }
141    catch (DatatypeConfigurationException e) {
142      throw new RuntimeException(factoryErrorMsg, e);
143    }
144  }

269  /**
270   * Returns true if tstamp is equal to or between start and end.
271   * 
272   * @param start The start time.
273   * @param tstamp The timestamp to test.
274   * @param end The end time.
275   * @return True if tstamp is between start and end.
276   */
277  public static boolean inBetween(XMLGregorianCalendar start, XMLGregorianCalendar tstamp,
278      XMLGregorianCalendar end) {
279    long startMillis = start.toGregorianCalendar().getTimeInMillis();
280    long endMillis = end.toGregorianCalendar().getTimeInMillis();
281    long tstampMillis = tstamp.toGregorianCalendar().getTimeInMillis();
282    return ((tstampMillis >= startMillis) && (tstampMillis <= endMillis));
283  }

298  /**
299   * Returns the number of days between time1 and time2. Returns a negative
300   * number if day1 is after day2. Takes into account daylight savings time
301   * issues.
302   * 
303   * @param day1 The first day.
304   * @param day2 The second day.
305   * @return The number of days between the two days.
306   */
307  public static int daysBetween(XMLGregorianCalendar day1, XMLGregorianCalendar day2) {
308    return (int) (getUnixDay(day2) - getUnixDay(day1));
309  }

311  /**
312   * Returns a long representing the number of days since the Unix epoch to the
313   * passed day.
314   * 
315   * @param day The day of interest.
316   * @return The number of days since the epoch.
317   */
318  private static long getUnixDay(XMLGregorianCalendar day) {
319    GregorianCalendar greg = day.toGregorianCalendar();
320    long offset = greg.get(Calendar.ZONE_OFFSET) + greg.get(Calendar.DST_OFFSET);
321    long daysSinceEpoch = (long) Math.floor((double) (greg.getTime().getTime() + offset)
322        / ((double) MILLISECS_PER_DAY));
323    return daysSinceEpoch;
324  }

385  /**
386   * Returns true if the passed timestamp indicates some time today or some time
387   * in the future.
388   * 
389   * @param timestamp The timestamp of interest.
390   * @return True if it's today or some day in the future.
391   */
392  public static boolean isTodayOrLater(XMLGregorianCalendar timestamp) {
393    XMLGregorianCalendar today = Tstamp.makeTimestamp();
394    boolean isToday = (today.getYear() == timestamp.getYear())
395        && (today.getMonth() == timestamp.getMonth()) && (today.getDay() == timestamp.getDay());
396    boolean afterToday = today.toGregorianCalendar().getTimeInMillis() < timestamp
397        .toGregorianCalendar().getTimeInMillis();
398    return (isToday || afterToday);
399  }

401  /**
402   * Returns true if the passed timestamp indicates some time yesterday or some
403   * time in the future. This is useful for the Commit/Churn DPDs, where the
404   * sensor typically sends data not from the current day, but from the day
405   * before.
406   * 
407   * @param timestamp The timestamp of interest.
408   * @return True if it's today or some day in the future.
409   */
410  public static boolean isYesterdayOrLater(XMLGregorianCalendar timestamp) {
411    XMLGregorianCalendar yesterday = Tstamp.incrementDays(Tstamp.makeTimestamp(), -1);
412
413    boolean isYesterday = (yesterday.getYear() == timestamp.getYear())
414        && (yesterday.getMonth() == timestamp.getMonth())
415        && (yesterday.getDay() == timestamp.getDay());
416    boolean afterYesterday = yesterday.toGregorianCalendar().getTimeInMillis() < timestamp
417        .toGregorianCalendar().getTimeInMillis();
418    return (isYesterday || afterYesterday);
419  }

433  /**
434   * Helper function that prepares a List timestamps, between the start time and
435   * end time, at the given sampling interval.
436   * 
437   * @param startTime The start of the range requested.
438   * @param endTime The start of the range requested.
439   * @param intervalMinutes The sampling interval requested in minutes.
440   * @return The List of XMLGregorianCalendars.
441   */
442  public static List<XMLGregorianCalendar> getTimestampList(XMLGregorianCalendar startTime,
443      XMLGregorianCalendar endTime, int intervalMinutes) {
444    long intervalMilliseconds;
445    long rangeLength = Tstamp.diff(startTime, endTime);
446    long minutesToMilliseconds = 60L * 1000L;
447
448    if (intervalMinutes < 0) {
449      return null;
450    }
451    if (rangeLength <= 0) {
452      // either startTime == endTime, or startTime > endTime
453      return null;
454    }
455    else if (intervalMinutes == 0) {
456      // use default interval
457      intervalMilliseconds = rangeLength / 10;
458    }
459    else if ((intervalMinutes * minutesToMilliseconds) > rangeLength) {
460      // TODO BOGUS, should throw an exception so callers can distinguish
461      // between problems
462      return null;
463    }
464    else {
465      // got a good interval
466      intervalMilliseconds = intervalMinutes * minutesToMilliseconds;
467    }
468    // DEBUG
469    // System.out.format("%nstartTime=%s, endTime=%s, interval=%d min%n",
470    // startTime, endTime,
471    // intervalMilliseconds / minutesToMilliseconds);
472
473    // Build list of timestamps, starting with startTime, separated by
474    // intervalMilliseconds
475    List<XMLGregorianCalendar> timestampList = new ArrayList<XMLGregorianCalendar>();
476    XMLGregorianCalendar timestamp = startTime;
477    while (Tstamp.lessThan(timestamp, endTime)) {
478      timestampList.add(timestamp);
479      // System.out.format("timestamp=%s%n", timestamp); // DEBUG
480      timestamp = Tstamp.incrementMilliseconds(timestamp, intervalMilliseconds);
481    }
482    // add endTime to cover the last runt interval which is <=
483    // intervalMilliseconds
484    timestampList.add(endTime);
485    // System.out.format("timestamp=%s%n", endTime); // DEBUG
486    return timestampList;
487  }

489  /**
490   * Produces a list of n + 1 XMLGregorianCalendar instances that span the given
491   * time interval by dividing the interval by n.
492   * 
493   * @param n The number of timestamps to generate.
494   * @param startTime The start of the interval.
495   * @param endTime The end of the interval.
496   * @return A List<XMLGregorianCalendar> of size n.
497   */
498  public static List<XMLGregorianCalendar> getNTimestampList(int n, XMLGregorianCalendar startTime,
499      XMLGregorianCalendar endTime) {
500    long rangeLength = Tstamp.diff(startTime, endTime);
501    long sampleIntervalMilliseconds = rangeLength / n;
502    
503    // Build list of timestamps, starting with startTime, separated by
504    // intervalMilliseconds
505    List<XMLGregorianCalendar> timestampList = new ArrayList<XMLGregorianCalendar>();
506    XMLGregorianCalendar timestamp = startTime;
507    while (Tstamp.lessThan(timestamp, endTime) || Tstamp.equal(timestamp, endTime)) {
508      timestampList.add(timestamp);
509      // System.out.format("timestamp=%s%n", timestamp); // DEBUG
510      timestamp = Tstamp.incrementMilliseconds(timestamp, sampleIntervalMilliseconds);
511    }
512    // add endTime to cover the last runt interval which is <=
513    // intervalMilliseconds
514    timestampList.add(endTime);
515    // System.out.format("timestamp=%s%n", endTime); // DEBUG
516    return timestampList;
517
518  }
