001package org.wattdepot.common.util.logger;
002
003import java.io.File;
004import java.io.IOException;
005import java.util.Enumeration;
006import java.util.logging.FileHandler;
007import java.util.logging.Handler;
008import java.util.logging.LogManager;
009import java.util.logging.Logger;
010import java.util.logging.SimpleFormatter;
011
012/**
013 * Provides a convenience method for Restlet logging that adjusts the output Handlers. Portions of
014 * this code are adapted from http://hackystat-utilities.googlecode.com/
015 * 
016 * @author Philip Johnson
017 * @author Robert Brewer
018 */
019public final class RestletLoggerUtil {
020
021  /** Make this class non-instantiable. */
022  private RestletLoggerUtil() {
023    // Do nothing.
024  }
025
026  /**
027   * Adjusts the Restlet Loggers so that they send their output to a file, not the console.
028   * 
029   * @param serverHome Home directory for this server. Logging files are placed in
030   * [serverHome]/logs. If null, an IllegalArgumentException is thrown.
031   * @throws IllegalArgumentException If serverHome is null.
032   * @throws RuntimeException If there are problems creating directories or opening log file.
033   */
034  public static void useFileHandler(String serverHome) {
035    LogManager logManager = LogManager.getLogManager();
036    // System.out.println("In useFileHandler");
037    for (Enumeration<String> en = logManager.getLoggerNames(); en.hasMoreElements();) {
038      String logName = en.nextElement();
039      if ((logName.startsWith("com.noelios") || logName.startsWith("org.restlet") || "global"
040          .equals(logName))
041          && (logManager.getLogger(logName) != null)) {
042        // First, get rid of current Handlers
043        Logger logger = logManager.getLogger(logName);
044        logger.setFilter(new HTTPClientHelperFilter());
045//        System.out.println("logger is: " + logger + " name = " + logName);
046        logger = logger.getParent();
047//        System.out.println("parent logger is: " + logger);
048        Handler[] handlers = logger.getHandlers();
049        for (Handler handler : handlers) {
050          logger.removeHandler(handler);
051        }
052        // System.out.println("Removed handlers.");
053        // Define a handler that writes to the ~/.wattdepot3/<service>/logs directory
054        // Define a file handler that writes to the logs directory, creating it if nec.
055        if (serverHome == null) {
056          throw new IllegalArgumentException("Attempt to change Restlet logging to null filename");
057        }
058        else {
059          File logDir = new File(serverHome + "/logs/");
060          boolean dirsOk = logDir.mkdirs();
061          if (!dirsOk && !logDir.exists()) {
062            throw new RuntimeException("mkdirs() failed");
063          }
064          // System.out.println("Made this directory: " + logDir);
065          String fileName = logDir + "/" + logName + ".%u.log";
066          FileHandler fileHandler;
067          try {
068            fileHandler = new FileHandler(fileName, 500000, 10, true);
069            fileHandler.setFormatter(new SimpleFormatter());
070            logger.addHandler(fileHandler);
071          }
072          catch (IOException e) {
073            // throw new RuntimeException
074            // ("Could not open the log file for this WattDepot service.", e);
075            System.out.println("Could not open log file: " + fileName + " " + e.getMessage());
076          }
077        }
078      }
079    }
080  }
081
082  /**
083   * Adjusts the Restlet Loggers so that they dump their output rather than sending it to the
084   * console.
085   */
086  public static void removeRestletLoggers() {
087    LogManager logManager = LogManager.getLogManager();
088    for (Enumeration<String> en = logManager.getLoggerNames(); en.hasMoreElements();) {
089      String logName = en.nextElement();
090      if ((logName.startsWith("com.noelios") || logName.startsWith("org.restlet") || "global"
091          .equals(logName) || logName.startsWith("org.hibernate"))
092          && (logManager.getLogger(logName) != null)) {
093        Logger logger = logManager.getLogger(logName);
094        logger = logger.getParent();
095        Handler[] handlers = logger.getHandlers();
096        for (Handler handler : handlers) {
097          logger.removeHandler(handler);
098        }
099      }
100    }
101  }
102  
103  /**
104   * Utility for printing out the known/defined logger names.
105   */
106  public static void showLoggers() {
107    LogManager logManager = LogManager.getLogManager();
108    for (Enumeration<String> en = logManager.getLoggerNames(); en.hasMoreElements();) {
109      String logName = en.nextElement();
110      System.out.println("logger name = '" + logName + "'");
111    }
112  }
113}
