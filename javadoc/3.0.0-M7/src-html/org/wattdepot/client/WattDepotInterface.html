019package org.wattdepot.client;
020
021import java.util.Date;
022
023import org.wattdepot.common.domainmodel.CollectorProcessDefinition;
024import org.wattdepot.common.domainmodel.CollectorProcessDefinitionList;
025import org.wattdepot.common.domainmodel.Depository;
026import org.wattdepot.common.domainmodel.DepositoryList;
027import org.wattdepot.common.domainmodel.InterpolatedValue;
028import org.wattdepot.common.domainmodel.Measurement;
029import org.wattdepot.common.domainmodel.MeasurementList;
030import org.wattdepot.common.domainmodel.MeasurementType;
031import org.wattdepot.common.domainmodel.MeasurementTypeList;
032import org.wattdepot.common.domainmodel.Sensor;
033import org.wattdepot.common.domainmodel.SensorGroup;
034import org.wattdepot.common.domainmodel.SensorGroupList;
035import org.wattdepot.common.domainmodel.SensorList;
036import org.wattdepot.common.domainmodel.SensorModel;
037import org.wattdepot.common.domainmodel.SensorModelList;
038import org.wattdepot.common.exception.IdNotFoundException;
039import org.wattdepot.common.exception.MeasurementGapException;
040import org.wattdepot.common.exception.MeasurementTypeException;
041import org.wattdepot.common.exception.NoMeasurementException;
042
043/**
044 * WattDepotInterface - The CRUD interface to the WattDepot server for regular
045 * users.
046 * 
047 * @author Cam Moore
048 * 
049 */
050public interface WattDepotInterface {
051
052  /**
053   * Deletes the given Depository.
054   * 
055   * @param depository the Depository to delete.
056   * @throws IdNotFoundException if the Depository is not found in the server.
057   */
058  public void deleteDepository(Depository depository) throws IdNotFoundException;
059
060  /**
061   * Deletes the given measurement from the given depository.
062   * 
063   * @param depository the Depository that stores the measurement.
064   * @param measurement the Measurement to delete.
065   * @throws IdNotFoundException if the Measurement is not found in the
066   *         Depository.
067   */
068  public void deleteMeasurement(Depository depository, Measurement measurement)
069      throws IdNotFoundException;
070
071  /**
072   * Deletes the given MeasurementType.
073   * 
074   * @param type the measurement type to delete.
075   * @throws IdNotFoundException if the MeasurementType is not found in the
076   *         Depository.
077   */
078  public void deleteMeasurementType(MeasurementType type) throws IdNotFoundException;
079
080  /**
081   * Deletes the given Sensor.
082   * 
083   * @param sensor the Sensor to delete.
084   * @throws IdNotFoundException if the Sensor is not found in the server.
085   */
086  public void deleteSensor(Sensor sensor) throws IdNotFoundException;
087
088  /**
089   * Deletes the given SensorGroup.
090   * 
091   * @param group the SensorGroup to delete.
092   * @throws IdNotFoundException if the SensorGroup is not found in the server.
093   */
094  public void deleteSensorGroup(SensorGroup group) throws IdNotFoundException;
095
096  /**
097   * Deletes the given SensorModel.
098   * 
099   * @param model the SensorModel to delete.
100   * @throws IdNotFoundException if the SensorModel is not found in the server.
101   */
102  public void deleteSensorModel(SensorModel model) throws IdNotFoundException;
103
104  /**
105   * Deletes the given CollectorProcessDefinitionData.
106   * 
107   * @param process the CollectorProcessDefinitionData to delete.
108   * @throws IdNotFoundException if the CollectorProcessDefinitionData is not
109   *         found in the server.
110   */
111  public void deleteCollectorProcessDefinition(CollectorProcessDefinition process)
112      throws IdNotFoundException;
113
114  /**
115   * @return The defined Depositories.
116   */
117  public DepositoryList getDepositories();
118
119  /**
120   * Retrieves the Depository with the given id from the WattDepot Server.
121   * 
122   * @param id The Depository's id.
123   * @return the Depository with the given id or null.
124   * @exception IdNotFoundException if the given id is not a Depository's id.
125   */
126  public Depository getDepository(String id) throws IdNotFoundException;
127
128  /**
129   * @param depository The Depository storing the Measurements.
130   * @param sensor The Sensor making the Measurements.
131   * @return The earliest Value.
132   */
133  public InterpolatedValue getEarliestValue(Depository depository, Sensor sensor);
134
135  /**
136   * @param depository The Depository storing the Measurements.
137   * @param group The SensorGroup whose sensors are making the Measurements.
138   * @return The earliest Value.
139   */
140  public InterpolatedValue getEarliestValue(Depository depository, SensorGroup group);
141
142  /**
143   * @param depository The Depository storing the Measurements.
144   * @param sensor The Sensor making the Measurements.
145   * @return The latest Measurement.
146   */
147  public InterpolatedValue getLatestValue(Depository depository, Sensor sensor);
148
149  /**
150   * @param depository The Depository storing the Measurements.
151   * @param group The SensorGroup whose sensors are making the Measurements.
152   * @return The latest Value.
153   */
154  public InterpolatedValue getLatestValue(Depository depository, SensorGroup group);
155
156  /**
157   * @param depository The Depository storing the Measurements.
158   * @param sensor The Sensor that made the measurements.
159   * @param start The start time.
160   * @param end The end time.
161   * @return The Measurements stored in the depository made by the sensor
162   *         between start and end.
163   */
164  public MeasurementList getMeasurements(Depository depository, Sensor sensor, Date start, Date end);
165
166  /**
167   * @param depository The Depository storing the Measurements.
168   * @param group The SensorGroup whose sensors made the measurements.
169   * @param start The start time.
170   * @param end The end time.
171   * @return The Measurements stored in the depository made by the sensor
172   *         between start and end.
173   */
174  public MeasurementList getMeasurements(Depository depository, SensorGroup group, Date start,
175      Date end);
176
177  /**
178   * @return The defined MeasurementTypes.
179   */
180  public MeasurementTypeList getMeasurementTypes();
181
182  /**
183   * @param id the unique id for the MeasurementType.
184   * @return The MeasurementType.
185   * @exception IdNotFoundException if the given id is not a MeasurementType's
186   *            id.
187   */
188  public MeasurementType getMeasurementType(String id) throws IdNotFoundException;
189
190  /**
191   * Retrieves the Sensor with the given id from the WattDepot Server.
192   * 
193   * @param id The Sensor's id.
194   * @return the Sensor with the given id or null.
195   * @exception IdNotFoundException if the given id is not a Sensor's id.
196   */
197  public Sensor getSensor(String id) throws IdNotFoundException;
198
199  /**
200   * Retrieves the SensorGroup with the given id from the WattDepot Server.
201   * 
202   * @param id The SensorGroup's id.
203   * @return the SensorGroup with the given id or null.
204   * @exception IdNotFoundException if the given id is not a SensorGroup's id.
205   */
206  public SensorGroup getSensorGroup(String id) throws IdNotFoundException;
207
208  /**
209   * @return The defined SensorGroups.
210   */
211  public SensorGroupList getSensorGroups();
212
213  /**
214   * Retrieves the SensorModel with the given id from the WattDepot Server.
215   * 
216   * @param id The SensorModel's id.
217   * @return the SensorModel with the given id or null.
218   * @exception IdNotFoundException if the given id is not a SensorModel's id.
219   */
220  public SensorModel getSensorModel(String id) throws IdNotFoundException;
221
222  /**
223   * @return The defined SensorModels.
224   */
225  public SensorModelList getSensorModels();
226
227  /**
228   * Retrieves the CollectorProcessDefinitionData with the given id from the
229   * WattDepot Server.
230   * 
231   * @param id The CollectorProcessDefinitionData's id.
232   * @return the CollectorProcessDefinitionData with the given id or null.
233   * @exception IdNotFoundException if the given id is not a
234   *            CollectorProcessDefinitionData's id.
235   */
236  public CollectorProcessDefinition getCollectorProcessDefinition(String id)
237      throws IdNotFoundException;
238
239  /**
240   * @return The defined CollectorProcessDefinitionDatas.
241   */
242  public CollectorProcessDefinitionList getCollectorProcessDefinition();
243
244  /**
245   * @return The defined Sensors.
246   */
247  public SensorList getSensors();
248
249  /**
250   * @param depository The Depository storing the measurements.
251   * @param sensor The sensor making the measurements.
252   * @param timestamp The time for the measured value.
253   * @return The Value 'measured' at the given time, most likely an interpolated
254   *         value.
255   * @throws NoMeasurementException if there aren't any measurements around the
256   *         timestamp.
257   */
258  public Double getValue(Depository depository, Sensor sensor, Date timestamp)
259      throws NoMeasurementException;
260
261  /**
262   * @param depository The Depository storing the measurements.
263   * @param group The SensorGroup whose sensors are making the measurements.
264   * @param timestamp The time for the measured value.
265   * @return The Value 'measured' at the given time, most likely an interpolated
266   *         value.
267   * @throws NoMeasurementException if there aren't any measurements around the
268   *         timestamp.
269   */
270  public Double getValue(Depository depository, SensorGroup group, Date timestamp)
271      throws NoMeasurementException;
272
273  /**
274   * @param depository The Depository storing the measurements.
275   * @param sensor The sensor making the measurements.
276   * @param start The start of the period.
277   * @param end The end of the period.
278   * @return The value measured the difference between the end value and the
279   *         start value.
280   * @throws NoMeasurementException if there are no measurements around the
281   *         start or end time.
282   */
283  public Double getValue(Depository depository, Sensor sensor, Date start, Date end)
284      throws NoMeasurementException;
285
286  /**
287   * @param depository The Depository storing the measurements.
288   * @param group The SensorGroup whose Sensors are making the measurements.
289   * @param start The start of the period.
290   * @param end The end of the period.
291   * @return The value measured the difference between the end value and the
292   *         start value.
293   * @throws NoMeasurementException if there are no measurements around the
294   *         start or end time.
295   */
296  public Double getValue(Depository depository, SensorGroup group, Date start, Date end)
297      throws NoMeasurementException;
298
299  /**
300   * @param depository The Depository storing the measurements.
301   * @param sensor The Sensor making the measurements.
302   * @param start The start of the interval.
303   * @param end The end of the interval
304   * @param gapSeconds The maximum number of seconds that measurements need to
305   *        be within the start and end.
306   * @return The value measured the difference between the end value and the
307   *         start value.
308   * @throws NoMeasurementException if there are no measurements around the
309   *         start or end time.
310   * @throws MeasurementGapException if the measurements around start or end are
311   *         too far apart.
312   */
313  public Double getValue(Depository depository, Sensor sensor, Date start, Date end, Long gapSeconds)
314      throws NoMeasurementException, MeasurementGapException;
315
316  /**
317   * @param depository The Depository storing the measurements.
318   * @param group The SensorGroup whose Sensors are making the measurements.
319   * @param start The start of the interval.
320   * @param end The end of the interval
321   * @param gapSeconds The maximum number of seconds that measurements need to
322   *        be within the start and end.
323   * @return The value measured the difference between the end value and the
324   *         start value.
325   * @throws NoMeasurementException if there are no measurements around the
326   *         start or end time.
327   * @throws MeasurementGapException if the measurements around start or end are
328   *         too far apart.
329   */
330  public Double getValue(Depository depository, SensorGroup group, Date start, Date end, Long gapSeconds)
331      throws NoMeasurementException, MeasurementGapException;
332
333  /**
334   * @param depository The Depository storing the measurements.
335   * @param sensor The Sensor making the measurements.
336   * @param timestamp The time of the value.
337   * @param gapSeconds The maximum number of seconds that measurements need to
338   *        be within the start and end.
339   * @return The Value 'measured' at the given time, most likely an interpolated
340   *         value.
341   * @throws NoMeasurementException If there aren't any measurements around the
342   *         time.
343   * @throws MeasurementGapException if the measurements around timestamp are
344   *         too far apart.
345   */
346  public Double getValue(Depository depository, Sensor sensor, Date timestamp, Long gapSeconds)
347      throws NoMeasurementException, MeasurementGapException;
348
349  /**
350   * @param depository The Depository storing the measurements.
351   * @param group The SensorGroup whose Sensors are making the measurements.
352   * @param timestamp The time of the value.
353   * @param gapSeconds The maximum number of seconds that measurements need to
354   *        be within the start and end.
355   * @return The Value 'measured' at the given time, most likely an interpolated
356   *         value.
357   * @throws NoMeasurementException If there aren't any measurements around the
358   *         time.
359   * @throws MeasurementGapException if the measurements around timestamp are
360   *         too far apart.
361   */
362  public Double getValue(Depository depository, SensorGroup group, Date timestamp, Long gapSeconds)
363      throws NoMeasurementException, MeasurementGapException;
364
365  /**
366   * Determines the health of the WattDepot server the client is communicating
367   * with.
368   * 
369   * @return true if the server is healthy, false if cannot connect.
370   */
371  public boolean isHealthy();
372
373  /**
374   * Stores the given Depository in the WattDepot Server.
375   * 
376   * @param depository the Depository.
377   */
378  public void putDepository(Depository depository);
379
380  /**
381   * @param depository The Depository to store the Measurement.
382   * @param measurement The Measurement to store.
383   * @throws MeasurementTypeException if the type of the measurement doesn't
384   *         match the type of the depository.
385   */
386  public void putMeasurement(Depository depository, Measurement measurement)
387      throws MeasurementTypeException;
388
389  /**
390   * Stores the given MeasurementType in the WattDepot Server.
391   * 
392   * @param type the MeasurementType.
393   */
394  public void putMeasurementType(MeasurementType type);
395
396  /**
397   * Stores the given Sensor in the WattDepot Server.
398   * 
399   * @param sensor the Sensor.
400   */
401  public void putSensor(Sensor sensor);
402
403  /**
404   * Stores the given SensorGroup in the WattDepot Server.
405   * 
406   * @param group the SensorGroup.
407   */
408  public void putSensorGroup(SensorGroup group);
409
410  /**
411   * Stores the given SensorModel in the WattDepot Server.
412   * 
413   * @param model the SensorModel.
414   */
415  public void putSensorModel(SensorModel model);
416
417  /**
418   * Stores the given CollectorProcessDefinitionData in the WattDepot Server.
419   * 
420   * @param process the CollectorProcessDefinitionData.
421   */
422  public void putCollectorProcessDefinition(CollectorProcessDefinition process);
423
424  /**
425   * Updates the given Depository in the WattDepot Server.
426   * 
427   * @param depository The Depository to update.
428   */
429  public void updateDepository(Depository depository);
430
431  /**
432   * Updates the given MeasurementType in the WattDepot Server.
433   * 
434   * @param type the MeasurementType to update.
435   */
436  public void updateMeasurementType(MeasurementType type);
437
438  /**
439   * Updates the given Sensor in the WattDepot Server.
440   * 
441   * @param sensor The Sensor to update.
442   */
443  public void updateSensor(Sensor sensor);
444
445  /**
446   * Updates the given SensorGroup in the WattDepot Server.
447   * 
448   * @param group The SensorGroup to update.
449   */
450  public void updateSensorGroup(SensorGroup group);
451
452  /**
453   * Updates the given SensorModel in the WattDepot Server.
454   * 
455   * @param model the SensorModel to update.
456   */
457  public void updateSensorModel(SensorModel model);
458
459  /**
460   * Updates the given CollectorProcessDefinitionData in the WattDepot Server.
461   * 
462   * @param process The CollectorProcessDefinitionData to update.
463   */
464  public void updateCollectorProcessDefinition(CollectorProcessDefinition process);
465}
