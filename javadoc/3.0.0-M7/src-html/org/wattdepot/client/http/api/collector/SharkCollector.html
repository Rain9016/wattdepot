058/**
059 * SharkCollector - Polls a ElectroIndustries Shark power meter for power &
060 * energy data periodically, and sends the results to a WattDepot server. For
061 * more information about Shark meters, see:
062 * http://www.electroind.com/pdf/Shark_New
063 * /E149721_Shark_200-S_Meter_Manual_V.1.03.pdf
064 * 
065 * Inspired by the wattdepot2 SharkSensor.
066 * 
067 * @author Cam Moore
068 * 
069 */
070public class SharkCollector extends MultiThreadedCollector {
071
072  // Note that the Modbus register map in the Shark manual appears to start at
073  // 1, while jamod
074  // expects it to start at 0, so you must subtract 1 from the register index
075  // listed in the manual!
076  /** Register index for "Power & Energy Format". */
077  private static final int ENERGY_FORMAT_REGISTER = 30006 - 1;
078  /** Number of words (registers) that make up "Power & Energy Format". */
079  private static final int ENERGY_FORMAT_LENGTH = 1;
080  /** Register index for "W-hours, Total". */
081  private static final int ENERGY_REGISTER = 1506 - 1;
082  /** Number of words (registers) that make up "W-hours, Total". */
083  private static final int ENERGY_LENGTH = 2;
084  /** Register index for "Watts, 3-Ph total". */
085  private static final int POWER_REGISTER = 1018 - 1;
086  /** Number of words (registers) that make up "Watts, 3-Ph total". */
087  private static final int POWER_LENGTH = 2;
088  /** InetAddress of the meter to be polled. */
089  private InetAddress meterAddress;
090  /** The Shark sensor. */
091  private Sensor sensor;
092  /**
093   * Stores the energy multiplier configured on the meter, which really means
094   * whether the energy value returned is in Wh, kWh, or MWh.
095   */
096  private double energyMultiplier = 0;
097  /**
098   * Stores the configured number of energy digits after the decimal point. For
099   * example, if the retrieved energy value is "12345678" and the decimals value
100   * is 2, then the actual energy value is "123456.78".
101   */
102  private int energyDecimals = 0;
103  /**
104   * Whether all required parameters (such as energy format parameters from the
105   * meter) have been successfully retrieved.
106   */
107  private boolean initialized = false;
108
109  /** The MeasurementType the Depository stores. */
110  private MeasurementType measType;
111  /** The Unit of the depository. */
112  private Unit<?> measUnit;
113  /** The name of the sensor this collector is polling. */
114  private String sensorName;

116  /**
117   * Initializes the SharkCollector.
118   * 
119   * @param serverUri
120   *          The URI for the WattDepot server.
121   * @param username
122   *          The name of a user defined in the WattDepot server.
123   * @param orgId
124   *          the user's organization id.
125   * @param password
126   *          The password for the user.
127   * @param collectorId
128   *          The CollectorProcessDefinitionId used to initialize this
129   *          collector.
130   * @param debug
131   *          flag for debugging messages.
132   * @throws BadCredentialException
133   *           if the user or password don't match the credentials in WattDepot.
134   * @throws IdNotFoundException
135   *           if the processId is not defined.
136   * @throws BadSensorUriException
137   *           if the Sensor's URI isn't valid.
138   */
139  public SharkCollector(String serverUri, String username, String orgId, String password,
140      String collectorId, boolean debug) throws BadCredentialException, IdNotFoundException,
141      BadSensorUriException {
142    super(serverUri, username, orgId, password, collectorId, debug);
143    this.measType = depository.getMeasurementType();
144    this.measUnit = Unit.valueOf(measType.getUnits());
145    this.sensor = client.getSensor(definition.getSensorId());
146    this.sensorName = sensor.getName();
147
148    URL sensorURL;
149    try {
150      sensorURL = new URL(sensor.getUri());
151      String sensorHostName = sensorURL.getHost();
152      this.meterAddress = InetAddress.getByName(sensorHostName);
153    }
154    catch (MalformedURLException e) {
155      throw new BadSensorUriException(sensor.getUri() + " is not a valid URI.");
156    }
157    catch (UnknownHostException e) {
158      throw new BadSensorUriException("Unable to resolve sensor at " + sensor.getUri());
159    }
160  }
161
162  /*
163   * (non-Javadoc)
164   * 
165   * @see java.util.TimerTask#run()
166   */
167  @Override
168  public void run() {
169    if (!this.initialized) {
170      this.initialized = initialize();
171    }
172    if (initialized) {
173      Measurement meas = pollMeter(this.meterAddress, this.energyMultiplier, this.energyDecimals);
174      if (meas != null) {
175        try {
176          this.client.putMeasurement(depository, meas);
177        }
178        catch (MeasurementTypeException e) {
179          System.err.format("%s does not store %s measurements%n", depository.getName(),
180              meas.getMeasurementType());
181        }
182        if (debug) {
183          System.out.println(meas);
184        }
185      }
186    }
187  }

189  /**
190   * Processes the command line arguments and starts the eGauge Collector.
191   * 
192   * @param args
193   *          command line arguments.
194   */
195  public static void main(String[] args) {
196    Options options = new Options();
197    options.addOption("h", "help", false,
198        "Usage: EGaugeCollector <server uri> <username> <password> <collectorid>");
199    options.addOption("s", "server", true, "WattDepot Server URI. (http://server.wattdepot.org)");
200    options.addOption("u", "username", true, "Username");
201    options.addOption("o", "organizationId", true, "User's Organization id.");
202    options.addOption("p", "password", true, "Password");
203    options.addOption("c", "collectorId", true, "Collector Process Definition Id.");
204    options.addOption("d", "debug", false, "Displays sensor data as it is sent to the server.");
205
206    CommandLine cmd = null;
207    String serverUri = null;
208    String username = null;
209    String organizationId = null;
210    String password = null;
211    String collectorId = null;
212    boolean debug = false;
213
214    CommandLineParser parser = new PosixParser();
215    HelpFormatter formatter = new HelpFormatter();
216    try {
217      cmd = parser.parse(options, args);
218    }
219    catch (ParseException e) {
220      System.err.println("Command line parsing failed. Reason: " + e.getMessage() + ". Exiting.");
221      System.exit(1);
222    }
223    if (cmd.hasOption("h")) {
224      formatter.printHelp("SharkCollector", options);
225      System.exit(0);
226    }
227    if (cmd.hasOption("s")) {
228      serverUri = cmd.getOptionValue("s");
229    }
230    else {
231      serverUri = "http://localhost:8192/";
232    }
233    if (cmd.hasOption("u")) {
234      username = cmd.getOptionValue("u");
235    }
236    else {
237      username = UserInfo.ROOT.getUid();
238    }
239    if (cmd.hasOption("o")) {
240      organizationId = cmd.getOptionValue("o");
241    }
242    else {
243      organizationId = Organization.ADMIN_GROUP.getId();
244    }
245    if (cmd.hasOption("p")) {
246      password = cmd.getOptionValue("p");
247    }
248    else {
249      password = "default";
250    }
251    if (cmd.hasOption("c")) {
252      collectorId = cmd.getOptionValue("c");
253    }
254    else {
255      collectorId = "ilima_6th_power";
256    }
257
258    debug = cmd.hasOption("d");
259
260    if (debug) {
261      System.out.println("WattDepot Server: " + serverUri);
262      System.out.println("Username: " + username);
263      System.out.println("Password: " + password);
264      System.out.println("Collector Process Definition Id: " + collectorId);
265      System.out.println("debug: " + debug);
266      System.out.println();
267    }
268    try {
269      if (!MultiThreadedCollector.start(serverUri, username, organizationId, password, collectorId,
270          debug)) {
271        System.exit(1);
272      }
273    }
274    catch (InterruptedException e) {
275      e.printStackTrace();
276      System.exit(1);
277    }
278    catch (BadCredentialException e) {
279      e.printStackTrace();
280      System.exit(1);
281    }
282  }

284  /**
285   * Need to retrieve some configuration parameters from meter, but only want to
286   * do it once per session. Note that all failures to retrieve the parameters
287   * are considered temporary for caution's sake, so a misconfigured meter might
288   * continue to be polled indefinitely until the problem is resolved.
289   * 
290   * @return True if configuration parameters could be retrieved from meter,
291   *         false otherwise.
292   */
293  private boolean initialize() {
294    ModbusResponse response;
295    ReadMultipleRegistersResponse goodResponse = null;
296    try {
297      response = readRegisters(this.meterAddress, ENERGY_FORMAT_REGISTER, ENERGY_FORMAT_LENGTH);
298    }
299    catch (Exception e) {
300      System.err.format(
301          "%s: Unable to retrieve energy format parameters from meter %s: %s, retrying.%n",
302          Tstamp.makeTimestamp(), this.sensorName, e.getMessage());
303      return false;
304    }
305    if (response instanceof ReadMultipleRegistersResponse) {
306      goodResponse = (ReadMultipleRegistersResponse) response;
307    }
308    else if (response instanceof ExceptionResponse) {
309      System.err
310          .format(
311              "Got Modbus exception response while retrieving energy format parameters from meter %s, code: %s%n",
312              this.sensorName, ((ExceptionResponse) response).getExceptionCode());
313      return false;
314    }
315    else {
316      System.err
317          .format(
318              "Got strange Modbus reply while retrieving energy format parameters from meter %s, aborting.%n",
319              this.sensorName);
320      return false;
321    }
322
323    double energyMultiplier = decodeEnergyMultiplier(goodResponse);
324    if (energyMultiplier == 0) {
325      System.err.format("Got bad energy multiplier from meter %s energy format, aborting.%n",
326          this.sensorName);
327      return false;
328    }
329
330    int energyDecimals = decodeEnergyDecimals(goodResponse);
331    if (energyDecimals == 0) {
332      System.err.format("Got bad energy decimal format from meter %s energy format, aborting.%n",
333          this.sensorName);
334      return false;
335    }
336    // Able to retrieve all the energy format parameters, so set the class
337    // fields for future use
338    this.energyMultiplier = energyMultiplier;
339    this.energyDecimals = energyDecimals;
340    return true;
341
342  }

344  /**
345   * Reads one or more consecutive registers from a device using Modbus/TCP
346   * using the default TCP port (502).
347   * 
348   * @param address
349   *          The IP address of the device to be read from
350   * @param register
351   *          The Modbus register to be read, as a decimal integer
352   * @param length
353   *          The number of registers to read
354   * @return A ReadMultipleRegistersResponse containing the response from the
355   *         device, or null if there was some unexpected error.
356   * @throws Exception
357   *           If an error occurs while attempting to read the registers.
358   */
359  private ModbusResponse readRegisters(InetAddress address, int register, int length)
360      throws Exception {
361    return readRegisters(address, Modbus.DEFAULT_PORT, register, length);
362  }
363
364  /**
365   * Reads one or more consecutive registers from a device using Modbus/TCP.
366   * 
367   * @param address
368   *          The IP address of the device to be read from
369   * @param port
370   *          The destination TCP port to connect to
371   * @param register
372   *          The Modbus register to be read, as a decimal integer
373   * @param length
374   *          The number of registers to read
375   * @return A ReadMultipleRegistersResponse containing the response from the
376   *         device, or null if there was some unexpected error.
377   * @throws Exception
378   *           If an error occurs while attempting to read the registers.
379   */
380  private ModbusResponse readRegisters(InetAddress address, int port, int register, int length)
381      throws Exception {
382    TCPMasterConnection connection = null;
383    ModbusTCPTransaction transaction = null;
384    ReadMultipleRegistersRequest request = null;
385    ModbusResponse response = null;
386
387    // Open the connection
388    connection = new TCPMasterConnection(address);
389    try {
390      connection.setPort(port);
391      connection.connect();
392
393      // Prepare the request
394      request = new ReadMultipleRegistersRequest(register, length);
395
396      // Prepare the transaction
397      transaction = new ModbusTCPTransaction(connection);
398      transaction.setRequest(request);
399      transaction.execute();
400      response = transaction.getResponse();
401
402    }
403    finally {
404      // Close the connection
405      connection.close();
406    }
407
408    return response;
409  }

411  /**
412   * Decodes the energy multiplier configured on the meter, which really means
413   * whether the energy value returned is in Wh, kWh, or MWh.
414   * 
415   * @param response
416   *          The response from the meter containing the power and energy
417   *          format.
418   * @return A double that represents the scale which energy readings should be
419   *         multiplied by, or 0 if there was some problem decoding the value.
420   */
421  private double decodeEnergyMultiplier(ReadMultipleRegistersResponse response) {
422    if ((response != null) && (response.getWordCount() == 1)) {
423      // From Shark manual, bitmap looks like this ("-" is unused bit
424      // apparently):
425      // ppppiinn feee-ddd
426      //
427      // pppp = power scale (0-unit, 3-kilo, 6-mega, 8-auto)
428      // ii = power digits after decimal point (0-3),
429      // applies only if f=1 and pppp is not auto
430      // nn = number of energy digits (5-8 --> 0-3)
431      // eee = energy scale (0-unit, 3-kilo, 6-mega)
432      // f = decimal point for power
433      // (0=data-dependant placement, 1=fixed placement per ii value)
434      // ddd = energy digits after decimal point (0-6)
435
436      // Get energy scale by shifting off 4 bits and then mask with 111
437      int energyScale = (response.getRegisterValue(0) >>> 4) & 7;
438      switch (energyScale) {
439      case 0:
440        // watts
441        return 1.0;
442      case 3:
443        // kilowatts
444        return 1000.0;
445      case 6:
446        // megawatts
447        return 1000000.0;
448      default:
449        // should never happen, according to manual, so return 0
450        // System.err.println("Unknown energy scale from meter, defaulting to kWh");
451        return 0.0;
452      }
453    }
454    else {
455      return 0.0;
456    }
457  }

459  /**
460   * Decodes the configured number of energy digits after the decimal point. For
461   * example, if the retrieved energy value is "12345678" and the decimals value
462   * is 2, then the actual energy value is "123456.78".
463   * 
464   * @param response
465   *          The response from the meter containing the power and energy
466   *          format.
467   * @return An int that represents the number of ending digits from the energy
468   *         reading that should be considered as decimals.
469   */
470  private int decodeEnergyDecimals(ReadMultipleRegistersResponse response) {
471    if ((response != null) && (response.getWordCount() == 1)) {
472      // From Shark manual, bitmap looks like this ("-" is unused bit
473      // apparently):
474      // ppppiinn feee-ddd
475      //
476      // pppp = power scale (0-unit, 3-kilo, 6-mega, 8-auto)
477      // ii = power digits after decimal point (0-3),
478      // applies only if f=1 and pppp is not auto
479      // nn = number of energy digits (5-8 --> 0-3)
480      // eee = energy scale (0-unit, 3-kilo, 6-mega)
481      // f = decimal point for power
482      // (0=data-dependant placement, 1=fixed placement per ii value)
483      // ddd = energy digits after decimal point (0-6)
484
485      // Get # of energy digits after decimal point by masking with 111
486      return response.getRegisterValue(0) & 7;
487    }
488    else {
489      return 0;
490    }
491  }

500  /**
501   * Connects to the meter, retrieves the latest data, and returns it as a
502   * Measurement.
503   * 
504   * @param sensorAddress
505   *          The address of the meter.
506   * @param energyMultiplier
507   *          The amount to multiply energy readings to account for unit of
508   *          measurement.
509   * @param energyDecimals
510   *          The configured number of decimals include in the energy reading.
511   * @return The meter data as a Measurement.
512   */
513  private Measurement pollMeter(InetAddress sensorAddress, double energyMultiplier,
514      int energyDecimals) {
515    return pollMeter(sensorAddress, Modbus.DEFAULT_PORT, energyMultiplier, energyDecimals);
516  }

518  /**
519   * @param sensorAddress
520   *          The address of the meter.
521   * @param port
522   *          The port to communicate with.
523   * @param energyMultiplier
524   *          The amount to multiply energy readings to account for unit of
525   *          measurement.
526   * @param energyDecimals
527   *          The configured number of decimals include in the energy reading.
528   * @return The meter data as a Measurement.
529   */
530  private Measurement pollMeter(InetAddress sensorAddress, int port, double energyMultiplier,
531      int energyDecimals) {
532    Measurement meas = null;
533    // Record current time as close approximation to time for reading we are
534    // about to make
535    Date timestamp = new Date();
536    ModbusResponse energyResponse, powerResponse;
537    ReadMultipleRegistersResponse goodResponse = null;
538    try {
539      // Make both queries in rapid succession to reduce any lag on sensor side
540      energyResponse = readRegisters(sensorAddress, port, ENERGY_REGISTER, ENERGY_LENGTH);
541      powerResponse = readRegisters(sensorAddress, port, POWER_REGISTER, POWER_LENGTH);
542    }
543    catch (Exception e) {
544      System.err.format("%s Unable to retrieve energy data from meter %s: %s.%n", new Date(),
545          this.sensorName, e.getMessage());
546      return null;
547    }
548
549    // First handle energy response
550    if (energyResponse instanceof ReadMultipleRegistersResponse) {
551      goodResponse = (ReadMultipleRegistersResponse) energyResponse;
552    }
553    else if (energyResponse instanceof ExceptionResponse) {
554      System.err.format(
555          "Got Modbus exception response while retrieving energy data from meter %s, code: %s%n",
556          this.sensorName, ((ExceptionResponse) energyResponse).getExceptionCode());
557      return null;
558    }
559    else {
560      System.err.format("Got strange Modbus reply while retrieving energy data from meter %s.%n",
561          this.sensorName);
562      return null;
563    }
564    int wattHoursInt = ModbusUtil.registersToInt(extractByteArray(goodResponse));
565    // Take integer value, divide by 10^energyDecimals to move decimal point to
566    // right place,
567    // then multiply by a value depending on units (nothing, kilo, or mega).
568    double wattHours = (wattHoursInt / (Math.pow(10.0, energyDecimals))) * energyMultiplier;
569
570    // Then handle power response
571    if (powerResponse instanceof ReadMultipleRegistersResponse) {
572      goodResponse = (ReadMultipleRegistersResponse) powerResponse;
573    }
574    else if (powerResponse instanceof ExceptionResponse) {
575      System.err.format(
576          "Got Modbus exception response while retrieving power data from meter %s, code: %s%n",
577          this.sensorName, ((ExceptionResponse) powerResponse).getExceptionCode());
578      return null;
579    }
580    else {
581      System.err.println("Got strange Modbus reply while retrieving power data from meter.");
582      return null;
583    }
584    float watts = ModbusUtil.registersToFloat(extractByteArray(goodResponse));
585    Double value = null;
586    // power is given in W
587    Amount<?> power = Amount.valueOf(watts, SI.WATT);
588    // energy given in Wh
589    Amount<?> energy = Amount.valueOf(wattHours, SI.WATT.times(NonSI.HOUR));
590
591    if (isPower()) {
592      value = power.to(measUnit).getEstimatedValue();
593    }
594    else {
595      value = energy.to(measUnit).getEstimatedValue();
596    }
597    meas = new Measurement(sensor.getId(), timestamp, value, measUnit);
598    return meas;
599  }

601  /**
602   * Given a response with two consecutive registers, extract the values as a 4
603   * byte array so they can be passed to methods in ModbusUtil. It seems like
604   * there should be a better way to do this.
605   * 
606   * @param response
607   *          The response containing the two registers
608   * @return a byte[4] array or null if there is a problem with the response.
609   */
610  private byte[] extractByteArray(ReadMultipleRegistersResponse response) {
611    byte[] regBytes = new byte[4];
612    if (response.getWordCount() == 2) {
613      regBytes[0] = response.getRegister(0).toBytes()[0];
614      regBytes[1] = response.getRegister(0).toBytes()[1];
615      regBytes[2] = response.getRegister(1).toBytes()[0];
616      regBytes[3] = response.getRegister(1).toBytes()[1];
617      return regBytes;
618    }
619    else {
620      return null;
621    }
622  }
