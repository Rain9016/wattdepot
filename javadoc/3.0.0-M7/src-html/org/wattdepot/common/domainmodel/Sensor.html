019package org.wattdepot.common.domainmodel;
020
021import java.util.HashSet;
022import java.util.Set;
023
024import org.wattdepot.common.util.Slug;
025
026/**
027 * Sensor - Represents the device making measurements.
028 * 
029 * @author Cam Moore
030 * 
031 */
032public class Sensor implements IDomainModel {
033  /** A unique id for the Sensor usable in URLs. */
034  private String id;
035  /** The name of the Sensor. */
036  private String name;
037  /** The URI to the sensor. */
038  private String uri;
039  /** The id of the model of the sensor. */
040  private String modelId;
041  /** Additional properties of the sensor. */
042  private Set<Property> properties;
043  /** The owner of this sensor. */
044  private String ownerId;
045
046  /**
047   * Default constructor.
048   */
049  public Sensor() {
050    this.properties = new HashSet<Property>();
051  }
052
053  /**
054   * @param name The name.
055   * @param uri The URI to the meter.
056   * @param modelId The id of the meter's model.
057   * @param ownerId the id of the owner of the sensor.
058   */
059  public Sensor(String name, String uri, String modelId, String ownerId) {
060    this(Slug.slugify(name), name, uri, modelId, new HashSet<Property>(), ownerId);
061  }
062
063  /**
064   * @param id The unique id.
065   * @param name The name.
066   * @param uri The URI to the meter.
067   * @param modelId The id of the meter's model.
068   * @param properties The properties associated with this Sensor.
069   * @param ownerId the id of the owner of the sensor.
070   */
071  public Sensor(String id, String name, String uri, String modelId, Set<Property> properties,
072      String ownerId) {
073    this.id = id;
074    this.name = name;
075    this.uri = uri;
076    this.modelId = modelId;
077    this.properties = properties;
078    this.ownerId = ownerId;
079  }
080
081  /**
082   * @param e The Property to add.
083   * @return true if added.
084   * @see java.util.List#add(java.lang.Object)
085   */
086  public boolean addProperty(Property e) {
087    return properties.add(e);
088  }
089
090  /*
091   * (non-Javadoc)
092   * 
093   * @see java.lang.Object#equals(java.lang.Object)
094   */
095  @Override
096  public boolean equals(Object obj) {
097    if (this == obj) {
098      return true;
099    }
100    if (obj == null) {
101      return false;
102    }
103    if (!getClass().isAssignableFrom(obj.getClass())
104        && !obj.getClass().isAssignableFrom(getClass()) && getClass() != obj.getClass()) {
105      return false;
106    }
107    Sensor other = (Sensor) obj;
108    if (id == null) {
109      if (other.id != null) {
110        return false;
111      }
112    }
113    else if (!id.equals(other.id)) {
114      return false;
115    }
116    if (name == null) {
117      if (other.name != null) {
118        return false;
119      }
120    }
121    else if (!name.equals(other.name)) {
122      return false;
123    }
124    if (modelId == null) {
125      if (other.modelId != null) {
126        return false;
127      }
128    }
129    else if (!modelId.equals(other.modelId)) {
130      return false;
131    }
132    if (ownerId == null) {
133      if (other.ownerId != null) {
134        return false;
135      }
136    }
137    else if (!ownerId.equals(other.ownerId)) {
138      return false;
139    }
140    if (properties == null) {
141      if (other.properties != null) {
142        return false;
143      }
144    }
145    else if (!properties.equals(other.properties)) {
146      return false;
147    }
148    if (uri == null) {
149      if (other.uri != null) {
150        return false;
151      }
152    }
153    else if (!uri.equals(other.uri)) {
154      return false;
155    }
156    return true;
157  }
158
159  /**
160   * @return the id
161   */
162  public String getId() {
163    return id;
164  }
165
166  /**
167   * @return the modelId
168   */
169  public String getModelId() {
170    return modelId;
171  }
172
173  /**
174   * @return the name
175   */
176  public String getName() {
177    return name;
178  }
179
180  /**
181   * @return the id of the owner
182   */
183  public String getOrganizationId() {
184    return ownerId;
185  }
186
187  /**
188   * @return the properties
189   */
190  public Set<Property> getProperties() {
191    return properties;
192  }
193
194  /**
195   * @param key The key.
196   * @return The value of associated with the key.
197   */
198  public Property getProperty(String key) {
199    for (Property p : properties) {
200      if (p.getKey().equals(key)) {
201        return p;
202      }
203    }
204    return null;
205  }
206
207  /**
208   * @return the uri
209   */
210  public String getUri() {
211    return uri;
212  }
213
214  /*
215   * (non-Javadoc)
216   * 
217   * @see java.lang.Object#hashCode()
218   */
219  @Override
220  public int hashCode() {
221    final int prime = 31;
222    int result = 1;
223    result = prime * result + ((id == null) ? 0 : id.hashCode());
224    result = prime * result + ((name == null) ? 0 : name.hashCode());
225    result = prime * result + ((modelId == null) ? 0 : modelId.hashCode());
226    result = prime * result + ((ownerId == null) ? 0 : ownerId.hashCode());
227    result = prime * result + ((properties == null) ? 0 : properties.hashCode());
228    result = prime * result + ((uri == null) ? 0 : uri.hashCode());
229    return result;
230  }
231
232  /**
233   * Determines if the given group is the owner of this location.
234   * 
235   * @param group the UserGroup to check.
236   * @return True if the group owns the Location or the group is the
237   *         ADMIN_GROUP.
238   */
239  public boolean isOwner(Organization group) {
240    if (ownerId != null
241        && (ownerId.equals(group.getId()) || group.equals(Organization.ADMIN_GROUP))) {
242      return true;
243    }
244    return false;
245  }
246
247  /**
248   * @param o The Property to remove.
249   * @return true if removed.
250   * @see java.util.List#remove(java.lang.Object)
251   */
252  public boolean removeProperty(Object o) {
253    return properties.remove(o);
254  }
255
256  /**
257   * @param id the id to set
258   */
259  public void setId(String id) {
260    this.id = id;
261  }
262
263  /**
264   * @param model the id of the model to set
265   */
266  public void setModelId(String model) {
267    this.modelId = model;
268  }
269
270  /**
271   * @param name the name to set
272   */
273  public void setName(String name) {
274    this.name = name;
275    if (this.id == null) {
276      this.id = Slug.slugify(name);
277    }
278  }
279
280  /**
281   * @param ownerId the id of the owner to set
282   */
283  public void setOrganizationId(String ownerId) {
284    this.ownerId = ownerId;
285  }
286
287  /**
288   * @param properties the properties to set
289   */
290  public void setProperties(Set<Property> properties) {
291    this.properties = properties;
292  }
293
294  /**
295   * @param uri the uri to set
296   */
297  public void setUri(String uri) {
298    this.uri = uri;
299  }
300
301  /*
302   * (non-Javadoc)
303   * 
304   * @see java.lang.Object#toString()
305   */
306  @Override
307  public String toString() {
308    return "Sensor [id=" + id + ", name=" + name + ", uri=" + uri + ", modelId=" + modelId
309        + ", properties=" + properties + "]";
310  }
