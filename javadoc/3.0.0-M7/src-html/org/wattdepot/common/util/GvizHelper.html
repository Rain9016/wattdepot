045/**
046 * GvizHelper - Utility class that handles Google Visualization using the Google Visualization
047 * Datasource library. 
048 *
049 * * @see <a
050 * href="http://code.google.com/apis/chart/interactive/docs/dev/implementing_data_source.html">Google
051 * Visualization Datasource API</a>
052 * 
053 * @author Yongwen Xu
054 *
055 */
056public class GvizHelper { 
057  /** Conversion factor for milliseconds per minute. */
058  private static final long MILLISECONDS_PER_MINUTE = 60L * 1000;
059  
060  /**
061   * @param server
062   *          the restlet ServerResource
063   * @param type
064   *          type of the gviz query string, such as tqx, or tq
065   * @return the value of the gviz query string from the request
066   */
067  public static String getGvizQueryString(ServerResource server, String type) {
068    String qs = server.getRequest().getResourceRef().getQueryAsForm().getFirstValue(type);
069    try {
070      if (qs != null) {
071        qs = URLDecoder.decode(qs, "UTF-8");
072      }
073    }
074    catch (UnsupportedEncodingException e) {
075      qs = null;
076    }
077    return qs;
078  }
079  
080  /**
081   * @param resource
082   *          server resource object
083   * @param tqxString
084   *          gviz tqx query string, i.e., request id
085   * @param tqString
086   *          gviz tq query string, selectable fields
087   * @return gviz response 
088   */
089  public static String getGvizResponse(Object resource, String tqxString, String tqString) {
090    DataTable table = null;
091    try {
092      if (resource instanceof InterpolatedValue) {
093        table = getDataTable((InterpolatedValue)resource);
094      }
095      if (resource instanceof MeasurementList) {
096        table = getDataTable((MeasurementList)resource);
097      }
098      if (resource instanceof MeasuredValueList) {
099        table = getDataTable((MeasuredValueList)resource);
100      }
101    } 
102    catch (DataSourceException e) {
103      return getGvizDataErrorResponse(e);
104    }
105
106    return getGvizResponseFromDataTable(table, tqxString, tqString);
107  }
108  
109  /**
110   * @param mValue 
111   *          mesured value
112   * @return gviz DataTable
113   * @throws DataSourceException 
114   *          data source exception
115   */
116  private static DataTable getDataTable(InterpolatedValue mValue) throws DataSourceException {    
117    
118    MeasurementType mType = mValue.getMeasurementType();
119    
120    DataTable data = new DataTable();
121
122    try {
123      data.addColumn(
124          new ColumnDescription("Date", ValueType.DATETIME, "Date & Time"));
125      data.addColumn(
126          new ColumnDescription(mType.getName(), ValueType.NUMBER, mType.toString()));
127      
128      TableRow row = new TableRow();
129      
130      XMLGregorianCalendar xgcal = DateConvert.convertDate(mValue.getDate());
131      row.addCell(new DateTimeValue(convertTimestamp(xgcal)));
132      row.addCell(mValue.getValue());
133
134      data.addRow(row);
135    }
136    catch (TypeMismatchException | DatatypeConfigurationException e) {
137      throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Problem adding data to table"); // NOPMD
138    }
139    return data;
140  }
141  
142  /**
143   * @param mList
144   *          measurement list
145   * @return gviz data table
146   * @throws DataSourceException
147   *          data source exception
148   */
149  private static DataTable getDataTable(MeasurementList mList) throws DataSourceException {
150    DataTable data = new DataTable();
151
152    // Sets up the columns requested by any SELECT in the datasource query
153    ArrayList<Measurement> measurements = mList.getMeasurements();
154    if (! measurements.isEmpty()) {
155      String type = measurements.get(0).getMeasurementType();
156      try {
157        data.addColumn(
158            new ColumnDescription("Timestamp", ValueType.DATETIME, "Date & Time"));
159        data.addColumn(
160            new ColumnDescription(type, ValueType.NUMBER, type));
161        
162        for (Measurement measurement : measurements) {
163          TableRow row = new TableRow();
164          XMLGregorianCalendar xgcal = DateConvert.convertDate(measurement.getDate());
165          row.addCell(new DateTimeValue(convertTimestamp(xgcal)));
166          row.addCell(measurement.getValue());
167          data.addRow(row);
168        }
169      }
170      catch (NumberFormatException e) {
171        // String value in database couldn't be converted to a number.
172        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Found bad number in database"); // NOPMD
173      }
174      catch (TypeMismatchException | DatatypeConfigurationException e) {
175          throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Problem adding data to table"); // NOPMD
176      } 
177    }
178    return data;
179  }

181  /**
182   * @param mList
183   *          measured value list
184   * @return gviz data table
185   * @throws DataSourceException
186   *          data source exception
187   */
188  private static DataTable getDataTable(MeasuredValueList mList) throws DataSourceException {
189    DataTable data = new DataTable();
190
191    // Sets up the columns requested by any SELECT in the datasource query
192    ArrayList<InterpolatedValue> values = mList.getMeasuredValues();
193    if (! values.isEmpty()) {
194      String type = values.get(0).getMeasurementType().getUnits();
195      try {
196        data.addColumn(
197            new ColumnDescription("Timestamp", ValueType.DATETIME, "Date & Time"));
198        data.addColumn(
199            new ColumnDescription(type, ValueType.NUMBER, type));
200        
201        for (InterpolatedValue value : values) {
202          TableRow row = new TableRow();
203          XMLGregorianCalendar xgcal = DateConvert.convertDate(value.getDate());
204          row.addCell(new DateTimeValue(convertTimestamp(xgcal)));
205          row.addCell(value.getValue());
206          data.addRow(row);
207        }
208      }
209      catch (NumberFormatException e) {
210        // String value in database couldn't be converted to a number.
211        throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Found bad number in database"); // NOPMD
212      }
213      catch (TypeMismatchException | DatatypeConfigurationException e) {
214          throw new DataSourceException(ReasonType.INTERNAL_ERROR, "Problem adding data to table"); // NOPMD
215      } 
216    }
217    return data;
218  }

220  /**
221   * @param e
222   *          DataSourceException
223   * @return the error message
224   */
225  private static String getGvizDataErrorResponse(DataSourceException e) {
226    return "({status:'error',errors:[{reason:'"
227        + e.getReasonType().toString() + "',message:'" + e.getMessageToUser() + "'}]});";
228  }

230  /**
231   * @param table
232   *          gviz data table
233   * @param tqxString
234   *          gviz query string
235   * @param tqString
236   *          gviz query string
237   * @return the gviz json response
238   */
239  private static String getGvizResponseFromDataTable(DataTable table, String tqxString, String tqString) {  
240    String response = "google.visualization.Query.setResponse";
241    
242    String reqId = null;
243    if (tqxString != null) {
244      String[] tqxArray = tqxString.split(";");
245      for (String s : tqxArray) {
246        if (s.contains("reqId")) {
247          reqId = s.substring(s.indexOf(":") + 1, s.length());
248        }
249      }
250    }
251    
252    String tableString = JsonRenderer.renderDataTable(table, true, true, true).toString();
253    
254    response += "({status:'ok',";
255    
256    if (reqId != null) {
257      response += "reqId:'" + reqId + "',";
258    }
259    response += "table:" + tableString + "});";
260    
261    return response;
262  }

264  /**
265   * Takes an XMLGregorianCalendar (the native WattDepot timestamp format) and returns a 
266   * com.ibm.icu.util.GregorianCalendar (<b>note:</b>
267   * this is <b>not</b> the same as a java.util.GregorianCalendar!) with the time zone set to GMT,
268   * but with the timestamp adjusted by the provided offset.
269   * 
270   * For example, if the input value is 10 AM Hawaii Standard Time (HST is -10 hours from UTC), and
271   * the offset is -600 minutes then the output will be 10 AM UTC, effectively subtracting 10 hours
272   * from the timestamp (in addition to the conversion between types).
273   * 
274   * This rigamarole is needed because the Google Visualization data source library only accepts
275   * timestamps in UTC, and the Annonated Timeline visualization displays the UTC values. Without
276   * this conversion, any graphs of meter data recorded in the Hawaii time zone (for instance) would
277   * display the data 10 hours later in the graph, which is not acceptable. The timezoneOffset is
278   * provided if in the future we want the capability to display data normalized to an arbitrary
279   * time zone, rather than just the one the data was collected in.
280   * 
281   * @param timestamp the input XMLGregorianCalendar timestamp
282   * @return a com.ibm.icu.util.GregorianCalendar suitable for use by the Google visualization data
283   * source library, and normalized by timeZoneOffset but with timeZone field set to GMT.
284   */
285  private static com.ibm.icu.util.GregorianCalendar convertTimestamp(XMLGregorianCalendar timestamp) {
286    // Calendar conversion hell. GViz library uses com.ibm.icu.util.GregorianCalendar, which is
287    // different from java.util.GregorianCalendar. So we go from our native format
288    // XMLGregorianCalendar -> GregorianCalendar, extract milliseconds since epoch, feed that
289    // to the IBM GregorianCalendar.
290
291    int timeZoneOffset = timestamp.getTimezone();
292    
293    com.ibm.icu.util.GregorianCalendar gvizCal = new com.ibm.icu.util.GregorianCalendar();
294    
295    // Added bonus, DateTimeValue only accepts GregorianCalendars if the timezone is GMT.
296    gvizCal.setTimeZone(com.ibm.icu.util.TimeZone.getTimeZone("GMT"));
297    
298    java.util.GregorianCalendar standardCal = timestamp.toGregorianCalendar();
299
300    // Add the timeZoneOffset to the epoch time after converting to milliseconds
301    gvizCal.setTimeInMillis(standardCal.getTimeInMillis()
302        + (timeZoneOffset * MILLISECONDS_PER_MINUTE));
303
304    return gvizCal;
305  }
