132  /**
133   * Creates a root Router to dispatch call to server resources.
134   * 
135   * @return the inbound root.
136   */
137  @Override
138  public Restlet createInboundRoot() {
139    Router router = new Router(getContext());
140    String webRoot = "file:///" + System.getProperty("user.dir") + "/target/classes";
141    Directory directory = new Directory(getContext(), webRoot);
142    directory.setListingAllowed(true);
143    router.attach("/webroot/", directory);
144    // router.attach("/wattdepot/", LoginPageServerResource.class);
145    // router.attach("/wattdepot/login/", LoginServerResource.class);
146    router.attach("/wattdepot/measurementtype/{measurementtype_id}",
147        MeasurementTypeServerResource.class);
148    router.attach("/wattdepot/measurementtypes/", MeasurementTypesServerResource.class);
149    router.attach("/wattdepot/sensormodel/", SensorModelServerResource.class);
150    router.attach("/wattdepot/sensormodel/{sensormodel_id}",
151        SensorModelServerResource.class);
152    router.attach("/wattdepot/sensormodels/", SensorModelsServerResource.class);
153    router.attach("/wattdepot/{group_id}/", AdminServerResource.class);
154    router.attach("/wattdepot/{group_id}/depository/", DepositoryServerResource.class);
155    router.attach("/wattdepot/{group_id}/depository/{depository_id}",
156        DepositoryServerResource.class);
157    router.attach("/wattdepot/{group_id}/depository/{depository_id}/measurement/",
158        DepositoryMeasurementServerResource.class);
159    router.attach("/wattdepot/{group_id}/depository/{depository_id}/measurement/{meas_id}",
160        DepositoryMeasurementServerResource.class);
161    router.attach("/wattdepot/{group_id}/depository/{depository_id}/measurements/",
162        DepositoryMeasurementsServerResource.class);
163    router.attach("/wattdepot/{group_id}/depository/{depository_id}/sensors/",
164        DepositorySensorsServerResource.class);
165    router.attach("/wattdepot/{group_id}/depository/{depository_id}/value/",
166        DepositoryValueServerResource.class);
167    router.attach("/wattdepot/{group_id}/depositories/", DepositoriesServerResource.class);
168    router.attach("/wattdepot/{group_id}/location/", SensorLocationServerResource.class);
169    router.attach("/wattdepot/{group_id}/location/{location_id}", SensorLocationServerResource.class);
170    router.attach("/wattdepot/{group_id}/locations/", SensorLocationsServerResource.class);
171    router.attach("/wattdepot/{group_id}/sensorgroup/", SensorGroupServerResource.class);
172    router.attach("/wattdepot/{group_id}/sensorgroup/{sensorgroup_id}",
173        SensorGroupServerResource.class);
174    router.attach("/wattdepot/{group_id}/sensorgroups/", SensorGroupsServerResource.class);
175    router.attach("/wattdepot/{group_id}/collectormetadata/", CollectorMetaDataServerResource.class);
176    router.attach("/wattdepot/{group_id}/collectormetadata/{collectormetadata_id}",
177        CollectorMetaDataServerResource.class);
178    router.attach("/wattdepot/{group_id}/collectormetadatas/", CollectorMetaDatasServerResource.class);
179    router.attach("/wattdepot/{group_id}/sensor/", SensorServerResource.class);
180    router.attach("/wattdepot/{group_id}/sensor/{sensor_id}", SensorServerResource.class);
181    router.attach("/wattdepot/{group_id}/sensors/", SensorsServerResource.class);
182    router.attach("/wattdepot/{group_id}/user/{user_id}", UserInfoServerResource.class);
183    router.attach("/wattdepot/{group_id}/userpassword/{user_id}", UserPasswordServerResource.class);
184    router.attach("/wattdepot/{group_id}/usergroup/", UserGroupServerResource.class);
185    router.attach("/wattdepot/{group_id}/usergroup/{usergroup_id}", UserGroupServerResource.class);
186    router.attach("/wattdepot/{group_id}/usergroups/", UserGroupsServerResource.class);
187
188    ChallengeAuthenticator authenticator = new ChallengeAuthenticator(getContext(),
189        ChallengeScheme.HTTP_BASIC, "WattDepot Realm");
190    authenticator.setNext(router);
191    return authenticator;
192
193    // return router;
194  }
