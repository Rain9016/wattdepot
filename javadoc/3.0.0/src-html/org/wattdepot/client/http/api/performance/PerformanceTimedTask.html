019package org.wattdepot.client.http.api.performance;

021import java.util.HashSet;
022import java.util.TimerTask;
023
024import org.wattdepot.client.http.api.WattDepotClient;
025import org.wattdepot.common.domainmodel.CollectorProcessDefinition;
026import org.wattdepot.common.domainmodel.Depository;
027import org.wattdepot.common.domainmodel.DepositoryList;
028import org.wattdepot.common.domainmodel.MeasurementType;
029import org.wattdepot.common.domainmodel.Property;
030import org.wattdepot.common.domainmodel.Sensor;
031import org.wattdepot.common.domainmodel.SensorList;
032import org.wattdepot.common.domainmodel.SensorModel;
033import org.wattdepot.common.exception.BadCredentialException;
034import org.wattdepot.common.exception.BadSensorUriException;
035import org.wattdepot.common.exception.IdNotFoundException;
036import org.wattdepot.common.util.SensorModelHelper;

038/**
039 * PerformanceTimedTask times a query from the WattDepot Server.
040 * 
041 * @author Cam Moore
042 * 
043 */
044public abstract class PerformanceTimedTask extends TimerTask {
045  /** Flag for debugging messages. */
046  protected boolean debug;
047  /** The definition about the collector. */
048  protected CollectorProcessDefinition definition;
049  /** The client used to communicate with the WattDepot server. */
050  protected WattDepotClient client;
051  /** The Depository for storing measurements. */
052  protected Depository depository;
053  /** The sensor to get the latest value for. */
054  protected Sensor sensor;
055
056  private Long totalTime;
057  private Long numberOfRuns;
058  private Long minGetTime;
059  private Long maxGetTime;
060  private String collectorId;
061
062  /**
063   * Initializes the GetLatestValueTask.
064   * 
065   * @param serverUri The URI for the WattDepot server.
066   * @param username The name of a user defined in the WattDepot server.
067   * @param orgId the id of the organization the user is in.
068   * @param password The password for the user.
069   * @param debug flag for debugging messages.
070   * @throws BadCredentialException if the user or password don't match the
071   *         credentials in WattDepot.
072   * @throws IdNotFoundException if the processId is not defined.
073   * @throws BadSensorUriException if the Sensor's URI isn't valid.
074   */
075  public PerformanceTimedTask(String serverUri, String username, String orgId, String password,
076      boolean debug) throws BadCredentialException, IdNotFoundException, BadSensorUriException {
077    this(serverUri, username, orgId, password, debug, "performance-evaluation-collector");
078  }
079
080  /**
081   * Initializes the GetLatestValueTask.
082   * 
083   * @param serverUri The URI for the WattDepot server.
084   * @param username The name of a user defined in the WattDepot server.
085   * @param orgId the id of the organization the user is in.
086   * @param password The password for the user.
087   * @param debug flag for debugging messages.
088   * @param collectorId the CollectorProcessDefinition id.
089   * @throws BadCredentialException if the user or password don't match the
090   *         credentials in WattDepot.
091   * @throws IdNotFoundException if the processId is not defined.
092   * @throws BadSensorUriException if the Sensor's URI isn't valid.
093   */
094  public PerformanceTimedTask(String serverUri, String username, String orgId, String password,
095      boolean debug, String collectorId) throws BadCredentialException, IdNotFoundException,
096      BadSensorUriException {
097    this.totalTime = 0l;
098    this.numberOfRuns = 0l;
099    this.maxGetTime = -1l;
100    this.minGetTime = Long.MAX_VALUE;
101    this.client = new WattDepotClient(serverUri, username, orgId, password);
102    this.debug = debug;
103    this.collectorId = collectorId;
104    if (!client.isDefinedCollectorProcessDefinition(this.collectorId)) {
105      // need to find a depository and sensor with data.
106      DepositoryList depositories = client.getDepositories();
107      if (depositories != null) {
108        for (Depository d : depositories.getDepositories()) {
109          SensorList sensors = client.getDepositorySensors(d.getId());
110          if (sensors != null && sensors.getSensors().size() > 0) {
111            this.depository = d;
112            this.sensor = sensors.getSensors().get(0);
113            break;
114          }
115        }
116      }
117    }
118    else {
119      if (!client.isDefinedCollectorProcessDefinition(this.collectorId)) {
120        initializePerformanceItems(this.collectorId);
121      }
122      this.definition = client.getCollectorProcessDefinition(this.collectorId);
123      this.depository = client.getDepository(definition.getDepositoryId());
124      this.sensor = client.getSensor(definition.getSensorId());
125    }
126  }
127
128  /**
129   * @return average put time.
130   */
131  public Double getAverageTime() {
132    if (debug) {
133      System.out.println(totalTime + " / " + numberOfRuns);
134    }
135    return 1.0 * (totalTime) / numberOfRuns;
136  }
137
138  /**
139   * @return the maxGetTime
140   */
141  public Long getMaxTime() {
142    return maxGetTime;
143  }
144
145  /**
146   * @return the minGetTime
147   */
148  public Long getMinTime() {
149    return minGetTime;
150  }
151
152  /**
153   * @return the numberOfRuns
154   */
155  public Long getNumberOfRuns() {
156    return numberOfRuns;
157  }
158
159  /**
160   * @return the totalTime
161   */
162  public Long getTotalTime() {
163    return totalTime;
164  }
165
166  /**
167   * Ensures that there is a Performance Evaluation sensor, depository, and
168   * collector process definition.
169   * 
170   * @param collectorId the id of the collector to define.
171   */
172  private void initializePerformanceItems(String collectorId) {
173    SensorModel model = SensorModelHelper.models.get(SensorModelHelper.STRESS);
174    Sensor stressSensor = new Sensor("Performance Evaluation Sensor",
175        "http://performance.evaluation.sensor", model.getId(), client.getOrganizationId());
176    if (!client.isDefinedSensor(stressSensor.getId())) {
177      client.putSensor(stressSensor);
178    }
179    MeasurementType measType = null;
180    if (client.isDefinedMeasurementType("energy-wh")) {
181      try {
182        measType = client.getMeasurementType("energy-wh");
183        Depository depository = new Depository("Performance Evaluation Energy", measType,
184            client.getOrganizationId());
185        if (!client.isDefinedDepository(depository.getId())) {
186          client.putDepository(depository);
187        }
188        CollectorProcessDefinition cpd = new CollectorProcessDefinition(collectorId,
189            "Performance Evaluation Collector", stressSensor.getId(), 1L, depository.getId(),
190            new HashSet<Property>(), client.getOrganizationId());
191        if (!client.isDefinedCollectorProcessDefinition(cpd.getId())) {
192          client.putCollectorProcessDefinition(cpd);
193        }
194      }
195      catch (IdNotFoundException e) { // NOPMD
196        // can't happen
197      }
198    }
199  }
200
201  /**
202   * The client task to time.
203   */
204  public abstract void clientTask();
205
206  /*
207   * (non-Javadoc)
208   * 
209   * @see java.util.TimerTask#run()
210   */
211  @Override
212  public void run() {
213    numberOfRuns++;
214    Long startTime = System.nanoTime();
215    clientTask();
216    Long endTime = System.nanoTime();
217    Long diff = endTime - startTime;
218    totalTime += diff;
219    if (debug) {
220      System.out.println("time = " + (diff / 1E9) + " seconds.");
221    }
222    if (minGetTime > diff) {
223      minGetTime = diff;
224    }
225    if (maxGetTime < diff) {
226      maxGetTime = diff;
227    }
228  }
