019package org.wattdepot.client.http.api.performance;
020
021import java.util.Timer;
022import java.util.TimerTask;
023
024import org.apache.commons.cli.CommandLine;
025import org.apache.commons.cli.CommandLineParser;
026import org.apache.commons.cli.HelpFormatter;
027import org.apache.commons.cli.Options;
028import org.apache.commons.cli.ParseException;
029import org.apache.commons.cli.PosixParser;
030import org.apache.commons.math3.stat.descriptive.DescriptiveStatistics;
031import org.wattdepot.common.exception.BadCredentialException;
032import org.wattdepot.common.exception.BadSensorUriException;
033import org.wattdepot.common.exception.IdNotFoundException;
034
035/**
036 * GetEarliestValueThroughput - Attempts to determine the maximum rate of
037 * getting the Earliest value for a Sensor in a WattDepot installation.
038 * 
039 * @author Cam Moore
040 * 
041 */
042public class GetEarliestValueThroughput extends TimerTask {
043
044  /** Manages the GetEarliestValueTasks. */
045  private Timer timer;
046  /** The GetEarliestValueTask we will sample. */
047  private GetEarliestValueTask sampleTask;
048  /** The WattDepot server's URI. */
049  private String serverUri;
050  /** The WattDepot User. */
051  private String username;
052  /** The WattDepot User's organization. */
053  private String orgId;
054  /** The WattDepot User's password. */
055  private String password;
056  /** Flag for debugging. */
057  private boolean debug;
058  /** The number of times we've checked the stats. */
059  private Integer numChecks;
060  private DescriptiveStatistics averageGetTime;
061  private DescriptiveStatistics averageMinGetTime;
062  private DescriptiveStatistics averageMaxGetTime;
063
064  private Long getsPerSec;
065  private Long calculatedGetsPerSec;
066
067  /**
068   * Initializes the FindGetEarliestValueThroughput instance.
069   * 
070   * @param serverUri The URI for the WattDepot server.
071   * @param username The name of a user defined in the WattDepot server.
072   * @param orgId the id of the organization the user is in.
073   * @param password The password for the user.
074   * @param debug flag for debugging messages.
075   * @throws BadCredentialException if the user or password don't match the
076   *         credentials in WattDepot.
077   * @throws IdNotFoundException if the processId is not defined.
078   * @throws BadSensorUriException if the Sensor's URI isn't valid.
079   */
080  public GetEarliestValueThroughput(String serverUri, String username, String orgId,
081      String password, boolean debug) throws BadCredentialException, IdNotFoundException,
082      BadSensorUriException {
083    this.serverUri = serverUri;
084    this.username = username;
085    this.orgId = orgId;
086    this.password = password;
087    this.debug = debug;
088    this.numChecks = 0;
089    this.getsPerSec = 1l;
090    this.calculatedGetsPerSec = 1l;
091    this.averageMaxGetTime = new DescriptiveStatistics();
092    this.averageMinGetTime = new DescriptiveStatistics();
093    this.averageGetTime = new DescriptiveStatistics();
094    this.timer = new Timer("throughput");
095    this.sampleTask = new GetEarliestValueTask(serverUri, username, orgId, password, debug);
096    // Starting at 1 get/second
097    this.timer.schedule(sampleTask, 0, 1000);
098  }
099
100  /**
101   * @param args command line arguments -s <server uri> -u <username> -p
102   *        <password> -o <orgId> -n <numSamples> [-d].
103   * @throws BadSensorUriException if there is a problem with the WattDepot
104   *         sensor definition.
105   * @throws IdNotFoundException if there is a problem with the organization id.
106   * @throws BadCredentialException if the credentials are not valid.
107   */
108  public static void main(String[] args) throws BadCredentialException, IdNotFoundException,
109      BadSensorUriException {
110    Options options = new Options();
111    CommandLine cmd = null;
112    String serverUri = null;
113    String username = null;
114    String organizationId = null;
115    String password = null;
116    Integer numSamples = null;
117    boolean debug = false;
118
119    options.addOption("h", false, "Usage: FindMeasurementThroughput -s <server uri> -u <username>"
120        + " -p <password> -o <orgId> [-d]");
121    options.addOption("s", "server", true, "WattDepot Server URI. (http://server.wattdepot.org)");
122    options.addOption("u", "username", true, "Username");
123    options.addOption("o", "organizationId", true, "User's Organization id.");
124    options.addOption("p", "password", true, "Password");
125    options.addOption("n", "numSamples", true, "Number of puts to sample.");
126    options.addOption("d", "debug", false, "Displays statistics as the Measurements are stored.");
127    CommandLineParser parser = new PosixParser();
128    HelpFormatter formatter = new HelpFormatter();
129    try {
130      cmd = parser.parse(options, args);
131    }
132    catch (ParseException e) {
133      System.err.println("Command line parsing failed. Reason: " + e.getMessage() + ". Exiting.");
134      System.exit(1);
135    }
136    if (cmd.hasOption("h")) {
137      formatter.printHelp("GetEarliestValueThroughput", options);
138      System.exit(0);
139    }
140    if (cmd.hasOption("s")) {
141      serverUri = cmd.getOptionValue("s");
142    }
143    else {
144      serverUri = "http://server.wattdepot.org/";
145    }
146    if (cmd.hasOption("u")) {
147      username = cmd.getOptionValue("u");
148    }
149    else {
150      username = "user";
151    }
152    if (cmd.hasOption("p")) {
153      password = cmd.getOptionValue("p");
154    }
155    else {
156      password = "default";
157    }
158    if (cmd.hasOption("o")) {
159      organizationId = cmd.getOptionValue("o");
160    }
161    else {
162      organizationId = "organization";
163    }
164    if (cmd.hasOption("n")) {
165      numSamples = Integer.parseInt(cmd.getOptionValue("n"));
166    }
167    else {
168      numSamples = 13;
169    }
170    debug = cmd.hasOption("d");
171    if (debug) {
172      System.out.println("GetEarliestValue Throughput:");
173      System.out.println("    WattDepotServer: " + serverUri);
174      System.out.println("    Username: " + username);
175      System.out.println("    OrganizationId: " + organizationId);
176      System.out.println("    Password: ********");
177      System.out.println("    Samples: " + numSamples);
178    }
179
180    Timer t = new Timer("monitoring");
181    t.schedule(
182        new GetEarliestValueThroughput(serverUri, username, organizationId, password, debug), 0,
183        numSamples * 1000);
184  }
185
186  /*
187   * (non-Javadoc)
188   * 
189   * @see java.util.TimerTask#run()
190   */
191  @Override
192  public void run() {
193    // wake up to check the stats.
194    if (this.numChecks == 0) {
195      // haven't actually run so do nothing.
196      this.numChecks++;
197      // should I put a bogus first rate so we don't start too fast?
198      // this.averageGetTime.addValue(1.0); // took 1 second per so we start with a
199                                         // low average.
200    }
201    else {
202      this.timer.cancel();
203      this.numChecks++;
204      Double aveTime = sampleTask.getAverageTime();
205      this.averageGetTime.addValue((aveTime / 1E9));
206      this.averageMinGetTime.addValue((sampleTask.getMinTime() / 1E9));
207      this.averageMaxGetTime.addValue((sampleTask.getMaxTime() / 1E9));
208      this.calculatedGetsPerSec = calculateGetRate(averageGetTime);
209      this.getsPerSec = calculatedGetsPerSec;
210      // System.out.println("Min put time = " + (sampleTask.getMinGetTime() /
211      // 1E9));
212      System.out.println("Ave get earliest value time = " + (aveTime / 1E9) + " => "
213          + Math.round(1.0 / (aveTime / 1E9)) + " gets/sec.");
214      // System.out.println("Max put time = " + (sampleTask.getMaxGetTime() /
215      // 1E9));
216      // System.out.println("Max put rate = " +
217      // calculateGetRate(averageMinGetTime));
218      System.out.println("Setting rate to " + this.calculatedGetsPerSec);
219      // System.out.println("Min put rate = " +
220      // calculateGetRate(averageMaxGetTime));
221      this.timer = new Timer("throughput");
222      this.sampleTask = null;
223      // if (debug) {
224      // System.out.println("Starting " + this.measPerSec +
225      // " threads @ 1 meas/s");
226      // }
227      for (int i = 0; i < getsPerSec; i++) {
228        try {
229          if (this.sampleTask == null) {
230            this.sampleTask = new GetEarliestValueTask(serverUri, username, orgId, password, debug);
231            this.timer.schedule(sampleTask, 0, 1000);
232            if (debug) {
233              System.out.println("Starting task " + i);
234            }
235          }
236          else {
237            this.timer.schedule(new GetEarliestValueTask(serverUri, username, orgId, password, debug),
238                0, 1000);
239            if (debug) {
240              System.out.println("Starting task " + i);
241            }
242          }
243          Thread.sleep(10);
244        }
245        catch (BadCredentialException e) { // NOPMD
246          // should not happen.
247          e.printStackTrace();
248        }
249        catch (IdNotFoundException e) { // NOPMD
250          // should not happen.
251          e.printStackTrace();
252        }
253        catch (BadSensorUriException e) { // NOPMD
254          // should not happen
255          e.printStackTrace();
256        }
257        catch (InterruptedException e) {  // NOPMD
258          // should not happen
259          e.printStackTrace();
260        }
261      }
262    }
263  }
264
265  /**
266   * @param stats the DescriptiveStatistics to calculate the mean put time.
267   * @return The estimated put rate based upon the time it takes to put a single
268   *         measurement.
269   */
270  private Long calculateGetRate(DescriptiveStatistics stats) {
271    double putTime = stats.getMean();
272    Long ret = null;
273    double numPuts = 1.0 / putTime;
274    ret = Math.round(numPuts);
275    return ret;
276  }
