019package org.wattdepot.client.http.api.csv;
020
021import java.io.IOException;
022import java.util.Set;
023
024import org.wattdepot.client.http.api.WattDepotClient;
025import org.wattdepot.common.domainmodel.CollectorProcessDefinition;
026import org.wattdepot.common.domainmodel.Depository;
027import org.wattdepot.common.domainmodel.MeasurementPruningDefinition;
028import org.wattdepot.common.domainmodel.Sensor;
029import org.wattdepot.common.domainmodel.SensorGroup;
030import org.wattdepot.common.exception.BadCredentialException;
031import org.wattdepot.common.exception.BadSensorUriException;
032import org.wattdepot.common.exception.IdNotFoundException;
033import org.wattdepot.common.util.csv.DefinitionFileReader;
034import org.wattdepot.common.util.csv.DefinitionFileWriter;
035
036/**
037 * OrganizationDomain - Provides way of initializing an Organization's domain
038 * from a definition file and saving out an Organization's domain to a file.
039 * 
040 * @author Cam Moore
041 * 
042 */
043public class OrganizationDomain {
044
045  /** The client to communicate with the WattDepot server. */
046  private WattDepotClient client;
047  /** The name of the definition file. */
048  private String fileName;
049
050  /**
051   * Initializes the OrganizationDomain.
052   * 
053   * @param serverUri The URI for the WattDepot server.
054   * @param username The name of a user defined in the WattDepot server.
055   * @param orgId the id of the organization the user is in.
056   * @param password The password for the user.
057   * @param fileName The name of the definition file.
058   * @throws BadCredentialException if the user or password don't match the
059   *         credentials in WattDepot.
060   * @throws IdNotFoundException if the processId is not defined.
061   * @throws BadSensorUriException if the Sensor's URI isn't valid.
062   */
063  public OrganizationDomain(String serverUri, String username, String orgId, String password,
064      String fileName) throws BadCredentialException, IdNotFoundException, BadSensorUriException {
065    this.client = new WattDepotClient(serverUri, username, orgId, password);
066    this.fileName = fileName;
067  }
068
069  /**
070   * Ensures that the WattDepot server has all the instances defined in the
071   * definition file.
072   * 
073   * @throws IOException if there is a problem reading the file.
074   */
075  public void initializeFromFile() throws IOException {
076    this.initializeFromFile(fileName);
077  }
078
079  /**
080   * Ensures that the WattDepot server has all the instances defined in the
081   * definition file.
082   * 
083   * @param name the name of the definition file.
084   * @throws IOException if there is a problem reading the file.
085   */
086  public void initializeFromFile(String name) throws IOException {
087    DefinitionFileReader reader = new DefinitionFileReader(name);
088    Set<String> orgIds = reader.getOrgIds();
089    if (orgIds.size() != 1) {
090      throw new IOException("Bad WattDepot definition file has more than one organization id.");
091    }
092    if (!orgIds.contains(client.getOrganizationId())) {
093      throw new IOException(client.getOrganizationId()
094          + " does not match the Organization ID in the file.");
095    }
096    for (Depository d : reader.getDepositories()) {
097      if (client.getOrganizationId().equals(d.getOrganizationId())
098          && !client.isDefinedDepository(d.getId())) {
099        client.putDepository(d);
100      }
101    }
102    for (Sensor s : reader.getSensors()) {
103      if (client.getOrganizationId().equals(s.getOrganizationId())
104          && !client.isDefinedSensor(s.getId())) {
105        client.putSensor(s);
106      }
107    }
108    for (SensorGroup g : reader.getSensorGroups()) {
109      if (client.getOrganizationId().equals(g.getOrganizationId())
110          && !client.isDefinedSensorGroup(g.getId())) {
111        client.putSensorGroup(g);
112      }
113    }
114    for (CollectorProcessDefinition cpd : reader.getCollectorProcessDefinitions()) {
115      if (client.getOrganizationId().equals(cpd.getOrganizationId())
116          && !client.isDefinedCollectorProcessDefinition(cpd.getId())) {
117        client.putCollectorProcessDefinition(cpd);
118      }
119    }
120    for (MeasurementPruningDefinition gcd : reader.getMeasurementPruningDefinitions()) {
121      if (client.getOrganizationId().equals(gcd.getOrganizationId())
122          && !client.isDefinedMeasurementPruningDefinition(gcd.getId())) {
123        client.putMeasurementPruningDefinition(gcd);
124      }
125    }
126  }
127
128  /**
129   * Saves out the Organization's Depositories, Sensors, SensorGroups, and
130   * CollectorProcessDefinitions.
131   * 
132   * @throws IOException if there is a problem writing the file.
133   */
134  public void saveCurrentDomain() throws IOException {
135    this.saveCurrentDomain(fileName);
136  }
137
138  /**
139   * Saves out the Organization's Depositories, Sensors, SensorGroups, and
140   * CollectorProcessDefinitions.
141   * 
142   * @param name the name of the file to save to.
143   * @throws IOException if there is a problem writing the file.
144   */
145  public void saveCurrentDomain(String name) throws IOException {
146    DefinitionFileWriter writer = new DefinitionFileWriter(name);
147    for (Depository d : client.getDepositories().getDepositories()) {
148      writer.add(d);
149    }
150    for (Sensor s : client.getSensors().getSensors()) {
151      writer.add(s);
152    }
153    for (SensorGroup g : client.getSensorGroups().getGroups()) {
154      writer.add(g);
155    }
156    for (CollectorProcessDefinition cpd : client.getCollectorProcessDefinitions().getDefinitions()) {
157      writer.add(cpd);
158    }
159    writer.writeFile();
160  }
161}
