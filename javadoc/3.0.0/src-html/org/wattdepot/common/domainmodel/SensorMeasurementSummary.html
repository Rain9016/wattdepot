019package org.wattdepot.common.domainmodel;
020
021import java.util.Date;
022
023/**
024 * SensorMeasurementSummary a summary of the Measurements created by a given
025 * Sensor in a depository. The summary has a start date, end date and number of
026 * Measurements in the period.
027 * 
028 * @author Cam Moore
029 * 
030 */
031public class SensorMeasurementSummary {
032  /** The id of the sensor making the measurements. */
033  private String sensorId;
034  /** The id of the depository storing the measurements. */
035  private String depositoryId;
036  /** The number of measurements in the interval. */
037  private Integer numMeasurements;
038  /** The start of the interval. */
039  private Date start;
040  /** The end of the interval. */
041  private Date end;
042
043  /**
044   * Default constructor.
045   */
046  public SensorMeasurementSummary() {
047
048  }
049
050  /**
051   * @param sensorId2 The sensor id.
052   * @param depositoryId2 The depository id.
053   * @param start The start of the period.
054   * @param end The endo of the period.
055   * @param size The number of measurements in the period.
056   */
057  public SensorMeasurementSummary(String sensorId2, String depositoryId2, Date start, Date end,
058      int size) {
059    this.sensorId = sensorId2;
060    this.depositoryId = depositoryId2;
061    this.start = new Date(start.getTime());
062    this.end = new Date(end.getTime());
063    this.numMeasurements = size;
064  }
065
066  /**
067   * @return the sensorId
068   */
069  public String getSensorId() {
070    return sensorId;
071  }
072
073  /**
074   * @param sensorId the sensorId to set
075   */
076  public void setSensorId(String sensorId) {
077    this.sensorId = sensorId;
078  }
079
080  /**
081   * @return the depositoryId
082   */
083  public String getDepositoryId() {
084    return depositoryId;
085  }
086
087  /**
088   * @param depositoryId the depositoryId to set
089   */
090  public void setDepositoryId(String depositoryId) {
091    this.depositoryId = depositoryId;
092  }
093
094  /**
095   * @return the numMeasurements
096   */
097  public Integer getNumMeasurements() {
098    return numMeasurements;
099  }
100
101  /**
102   * @param numMeasurements the numMeasurements to set
103   */
104  public void setNumMeasurements(Integer numMeasurements) {
105    this.numMeasurements = numMeasurements;
106  }
107
108  /**
109   * @return the start
110   */
111  public Date getStart() {
112    return new Date(start.getTime());
113  }
114
115  /**
116   * @param start the start to set
117   */
118  public void setStart(Date start) {
119    this.start = new Date(start.getTime());
120  }
121
122  /**
123   * @return the end
124   */
125  public Date getEnd() {
126    return new Date(end.getTime());
127  }
128
129  /**
130   * @param end the end to set
131   */
132  public void setEnd(Date end) {
133    this.end = new Date(end.getTime());
134  }
135
136  /*
137   * (non-Javadoc)
138   * 
139   * @see java.lang.Object#hashCode()
140   */
141  @Override
142  public int hashCode() {
143    final int prime = 31;
144    int result = 1;
145    result = prime * result + ((depositoryId == null) ? 0 : depositoryId.hashCode());
146    result = prime * result + ((end == null) ? 0 : end.hashCode());
147    result = prime * result + ((numMeasurements == null) ? 0 : numMeasurements.hashCode());
148    result = prime * result + ((sensorId == null) ? 0 : sensorId.hashCode());
149    result = prime * result + ((start == null) ? 0 : start.hashCode());
150    return result;
151  }
152
153  /*
154   * (non-Javadoc)
155   * 
156   * @see java.lang.Object#equals(java.lang.Object)
157   */
158  @Override
159  public boolean equals(Object obj) {
160    if (this == obj) {
161      return true;
162    }
163    if (obj == null) {
164      return false;
165    }
166    if (getClass() != obj.getClass()) {
167      return false;
168    }
169    SensorMeasurementSummary other = (SensorMeasurementSummary) obj;
170    if (depositoryId == null) {
171      if (other.depositoryId != null) {
172        return false;
173      }
174    }
175    else if (!depositoryId.equals(other.depositoryId)) {
176      return false;
177    }
178    if (end == null) {
179      if (other.end != null) {
180        return false;
181      }
182    }
183    else if (!end.equals(other.end)) {
184      return false;
185    }
186    if (numMeasurements == null) {
187      if (other.numMeasurements != null) {
188        return false;
189      }
190    }
191    else if (!numMeasurements.equals(other.numMeasurements)) {
192      return false;
193    }
194    if (sensorId == null) {
195      if (other.sensorId != null) {
196        return false;
197      }
198    }
199    else if (!sensorId.equals(other.sensorId)) {
200      return false;
201    }
202    if (start == null) {
203      if (other.start != null) {
204        return false;
205      }
206    }
207    else if (!start.equals(other.start)) {
208      return false;
209    }
210    return true;
211  }
212
213  /*
214   * (non-Javadoc)
215   * 
216   * @see java.lang.Object#toString()
217   */
218  @Override
219  public String toString() {
220    return "SensorMeasurementSummary [sensorId=" + sensorId + ", depositoryId=" + depositoryId
221        + ", numMeasurements=" + numMeasurements + ", start=" + start + ", end=" + end + "]";
222  }
223
224}
