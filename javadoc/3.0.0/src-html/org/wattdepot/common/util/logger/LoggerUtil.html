001package org.wattdepot.common.util.logger;

003import java.io.File;

006import java.util.logging.ConsoleHandler;

014/**
015 * Provides a convenience method for Restlet logging that adjusts the output
016 * Handlers. Portions of this code are adapted from
017 * http://hackystat-utilities.googlecode.com/
018 * 
019 * @author Philip Johnson
020 * @author Robert Brewer
021 */
022public final class LoggerUtil {
023
024  /** Make this class non-instantiable. */
025  private LoggerUtil() {
026    // Do nothing.
027  }
028
029  /**
030   * Forces all Loggers to use the Console for logging output.
031   */
032  public static void useConsoleHandler() {
033    LogManager logManager = LogManager.getLogManager();
034    for (Enumeration<String> en = logManager.getLoggerNames(); en
035        .hasMoreElements();) {
036      String logName = en.nextElement();
037      Logger logger = logManager.getLogger(logName);
038      if (logger != null) {
039        // remove the old handlers
040        Handler[] handlers = logger.getHandlers();
041        for (Handler handler : handlers) {
042          logger.removeHandler(handler);
043        }
044      }
045    }
046    Logger logger = logManager.getLogger("");
047    ConsoleHandler handler = new ConsoleHandler();
048    logger.addHandler(handler);
049  }
050
051  /**
052   * Removes all the handlers from all the defined loggers. This should
053   */
054  public static void disableLogging() {
055    Logger base = LogManager.getLogManager().getLogger(
056        Logger.GLOBAL_LOGGER_NAME);
057    base = base.getParent();
058    base.setLevel(Level.SEVERE);
059  }
060
061  /**
062   * Adjusts the Restlet Loggers so that they send their output to a file, not
063   * the console.
064   * 
065   * @param serverHome Home directory for this server. Logging files are placed
066   *        in [serverHome]/logs. If null, an IllegalArgumentException is
067   *        thrown.
068   * @throws IllegalArgumentException If serverHome is null.
069   * @throws RuntimeException If there are problems creating directories or
070   *         opening log file.
071   */
072  public static void useFileHandler(String serverHome) {
073    LogManager logManager = LogManager.getLogManager();
074    // System.out.println("In useFileHandler");
075    for (Enumeration<String> en = logManager.getLoggerNames(); en
076        .hasMoreElements();) {
077      String logName = en.nextElement();
078      if ((logName.startsWith("com.noelios")
079          || logName.startsWith("org.restlet") || "global".equals(logName))
080          && (logManager.getLogger(logName) != null)) {
081        // First, get rid of current Handlers
082        Logger logger = logManager.getLogger(logName);
083        logger.setFilter(new HTTPClientHelperFilter());
084        // System.out.println("logger is: " + logger + " name = " + logName);
085        logger = logger.getParent();
086        // System.out.println("parent logger is: " + logger);
087        Handler[] handlers = logger.getHandlers();
088        for (Handler handler : handlers) {
089          logger.removeHandler(handler);
090        }
091        // System.out.println("Removed handlers.");
092        // Define a handler that writes to the ~/.wattdepot3/<service>/logs
093        // directory
094        // Define a file handler that writes to the logs directory, creating it
095        // if nec.
096        if (serverHome == null) {
097          throw new IllegalArgumentException(
098              "Attempt to change Restlet logging to null filename");
099        }
100        else {
101          File logDir = new File(serverHome + "/logs/");
102          boolean dirsOk = logDir.mkdirs();
103          if (!dirsOk && !logDir.exists()) {
104            throw new RuntimeException("mkdirs() failed");
105          }
106          // System.out.println("Made this directory: " + logDir);
107          String fileName = logDir + "/" + logName + ".%u.log";
108          FileHandler fileHandler;
109          try {
110            fileHandler = new FileHandler(fileName, 500000, 10, true);
111            fileHandler.setFormatter(new SimpleFormatter());
112            logger.addHandler(fileHandler);
113          }
114          catch (IOException e) {
115            // throw new RuntimeException
116            // ("Could not open the log file for this WattDepot service.", e);
117            System.out.println("Could not open log file: " + fileName + " "
118                + e.getMessage());
119          }
120        }
121      }
122    }
123  }
124
125  /**
126   * Adjusts the Restlet Loggers so that they dump their output rather than
127   * sending it to the console.
128   */
129  public static void removeRestletLoggers() {
130    LogManager logManager = LogManager.getLogManager();
131    for (Enumeration<String> en = logManager.getLoggerNames(); en
132        .hasMoreElements();) {
133      String logName = en.nextElement();
134      if ((logName.startsWith("com.noelios")
135          || logName.startsWith("org.restlet") || "global".equals(logName) || logName
136            .startsWith("org.hibernate"))
137          && (logManager.getLogger(logName) != null)) {
138        Logger logger = logManager.getLogger(logName);
139        logger = logger.getParent();
140        Handler[] handlers = logger.getHandlers();
141        for (Handler handler : handlers) {
142          logger.removeHandler(handler);
143        }
144      }
145    }
146  }
147
148  /**
149   * Sets the logging level to be used for this WattDepot logger. If the passed
150   * string cannot be parsed into a Level, then INFO is set by default.
151   * 
152   * @param logger The logger whose level is to be set.
153   * @param level The new Level.
154   */
155  public static void setLoggingLevel(Logger logger, String level) {
156    Level newLevel = Level.INFO;
157    try {
158      newLevel = Level.parse(level);
159    }
160    catch (Exception e) {
161      logger.info("Couldn't set Logging level to: " + level);
162    }
163    logger.setLevel(newLevel);
164    for (Handler handler : logger.getHandlers()) {
165      handler.setLevel(newLevel);
166    }
167  }
168
169  /**
170   * Utility for printing out the known/defined logger names.
171   */
172  public static void showLoggers() {
173    LogManager logManager = LogManager.getLogManager();
174    for (Enumeration<String> en = logManager.getLoggerNames(); en
175        .hasMoreElements();) {
176      String logName = en.nextElement();
177      Logger logger = logManager.getLogger(logName);
178      Logger parent = logger.getParent();
179      System.out.print("logger name = '" + logName + "'");
180      if (parent != null) {
181        System.out.print(" parent = '" + parent.getName() + "'");
182      }
183      System.out.println();
184    }
185  }
