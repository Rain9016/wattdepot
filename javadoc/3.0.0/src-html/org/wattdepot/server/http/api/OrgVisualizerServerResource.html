046/**
047 * OrgVisualizerServerResource - Summary Web page generator for WattDepot
048 * Organizations. It handles ("/wattdepot/{org-id}/visualize/").
049 * 
050 * @author Cam Moore
051 * 
052 */
053public class OrgVisualizerServerResource extends WattDepotServerResource {
054
055  /**
056   * @return The Organization summary information as an HTML Representation.
057   */
058  @Get()
059  public Representation toHtml() {
060    getLogger().log(Level.INFO,
061        "GET " + API.BASE_URI + "{" + orgId + "}/" + Labels.VISUALIZE + "/");
062    if (isInRole(orgId) || isInRole(Organization.ADMIN_GROUP.getId())) {
063      Map<String, Object> dataModel = new HashMap<String, Object>();
064      dataModel.put("orgId", orgId);
065      Representation rep = null;
066      TemplateRepresentation template = null;
067      try {
068        // Long startTime = System.nanoTime();
069        depot.getOrganization(orgId, true);
070        List<Depository> depos = depot.getDepositories(orgId, false);
071        List<Sensor> sensors = depot.getSensors(orgId, false);
072        List<SensorGroup> sensorGroups = depot.getSensorGroups(orgId, false);
073        // Long endTime = System.nanoTime();
074        // Long diff = endTime - startTime;
075        // getLogger().log(Level.INFO,
076        // "getDepositories took " + (diff / 1E9) + " seconds");
077        Map<String, List<Sensor>> depoSensors = new HashMap<String, List<Sensor>>();
078        Map<String, Map<String, List<Date>>> depotSensorInfo = new HashMap<String, Map<String, List<Date>>>();
079        for (Depository d : depos) {
080          Map<String, List<Date>> sensorInfo = new HashMap<String, List<Date>>();
081          depotSensorInfo.put(d.getId(), sensorInfo);
082          List<Sensor> sensorList = new ArrayList<Sensor>();
083          depoSensors.put(d.getId(), sensorList);
084          for (String sensorId : depot.listSensors(d.getId(), orgId, false)) {
085            // startTime = System.nanoTime();
086            Sensor s = depot.getSensor(sensorId, orgId, false);
087            // endTime = System.nanoTime();
088            // diff = endTime - startTime;
089            // getLogger().log(Level.INFO,
090            // "getSensor took " + (diff / 1E9) + " seconds");
091            sensorList.add(s);
092            try {
093              InterpolatedValue earliest = depot.getEarliestMeasuredValue(
094                  d.getId(), orgId, sensorId, false);
095              InterpolatedValue latest = depot.getLatestMeasuredValue(
096                  d.getId(), orgId, sensorId, false);
097              List<Date> info = new ArrayList<Date>();
098              info.add(earliest.getDate());
099              info.add(latest.getDate());
100              sensorInfo.put(sensorId, info);
101            }
102            catch (NoMeasurementException e) {
103              // TODO Auto-generated catch block
104              e.printStackTrace();
105            }
106          }
107          for (String groupId : depot.getSensorGroupIds(orgId, false)) {
108            SensorGroup group = depot.getSensorGroup(groupId, orgId, false);
109            List<Date> info = new ArrayList<Date>();
110            InterpolatedValue earliest = null;
111            InterpolatedValue latest = null;
112            for (String sensorId : group.getSensors()) {
113              try {
114                InterpolatedValue temp = depot.getEarliestMeasuredValue(
115                    d.getId(), orgId, sensorId, false);
116                if (temp != null) {
117                  if (earliest == null) {
118                    earliest = temp;
119                  }
120                  else if (earliest.getDate().after(temp.getDate())) {
121                    earliest = temp;
122                  }
123                }
124                temp = depot.getLatestMeasuredValue(d.getId(), orgId, sensorId,
125                    false);
126                if (temp != null) {
127                  if (latest == null) {
128                    latest = temp;
129                  }
130                  else if (latest.getDate().before(temp.getDate())) {
131                    latest = temp;
132                  }
133                }
134              }
135              catch (NoMeasurementException e) {
136                // TODO Auto-generated catch block
137                e.printStackTrace();
138              }
139            }
140            if (earliest != null) {
141              info.add(earliest.getDate());
142            }
143            else {
144              // CAM what is a value we can use for no data?
145              info.add(new Date(0l));
146            }
147            if (latest != null) {
148              info.add(latest.getDate());
149            }
150            else {
151              // CAM what is a value we can use for no data?
152              info.add(new Date(0l));
153            }
154            sensorInfo.put(groupId, info);
155          }
156        }
157        dataModel.put("depositories", depos);
158        dataModel.put("sensors", sensors);
159        dataModel.put("sensorgroups", sensorGroups);
160        dataModel.put("depoSensors", depoSensors);
161        dataModel.put("depotSensorInfo", depotSensorInfo);
162        rep = new ClientResource(LocalReference.createClapReference(getClass()
163            .getPackage()) + "/Visualizer.ftl").get();
164      }
165      catch (IdNotFoundException e) {
166        e.printStackTrace();
167        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
168        return null;
169      }
170      catch (MisMatchedOwnerException e) {
171        e.printStackTrace();
172        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
173        return null;
174      }
175      template = new TemplateRepresentation(rep, dataModel, MediaType.TEXT_HTML);
176      return template;
177    }
178    return null;
179  }
180}
