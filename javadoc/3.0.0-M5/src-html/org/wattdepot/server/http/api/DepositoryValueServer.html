019package org.wattdepot.server.http.api;
020
021import java.text.ParseException;
022import java.util.Date;
023import java.util.logging.Level;
024
025import javax.xml.datatype.DatatypeConfigurationException;
026
027import org.restlet.data.Status;
028import org.restlet.resource.ResourceException;
029import org.wattdepot.common.domainmodel.Labels;
030import org.wattdepot.common.domainmodel.MeasuredValue;
031import org.wattdepot.common.domainmodel.Sensor;
032import org.wattdepot.common.exception.MeasurementGapException;
033import org.wattdepot.common.exception.MissMatchedOwnerException;
034import org.wattdepot.common.exception.NoMeasurementException;
035import org.wattdepot.common.util.DateConvert;
036import org.wattdepot.server.depository.impl.hibernate.DepositoryImpl;
037
038/**
039 * DepositoryValueServerResource - ServerResouce that handles the GET
040 * /wattdepot/{group_id}/depository/{depository_id}/value/ response.
041 * 
042 * @author Cam Moore
043 * 
044 */
045public class DepositoryValueServer extends WattDepotServerResource {
046  private String depositoryId;
047  private String sensorId;
048  private String start;
049  private String end;
050  private String timestamp;
051  private String gapSeconds;
052
053  /*
054   * (non-Javadoc)
055   * 
056   * @see org.restlet.resource.Resource#doInit()
057   */
058  @Override
059  protected void doInit() throws ResourceException {
060    super.doInit();
061    this.sensorId = getQuery().getValues(Labels.SENSOR);
062    this.start = getQuery().getValues(Labels.START);
063    this.end = getQuery().getValues(Labels.END);
064    this.timestamp = getQuery().getValues(Labels.TIMESTAMP);
065    this.groupId = getAttribute(Labels.GROUP_ID);
066    this.depositoryId = getAttribute(Labels.DEPOSITORY_ID);
067    this.gapSeconds = getQuery().getValues(Labels.GAP);
068  }
069
070  /**
071   * retrieve the depository value for a sensor.
072   * 
073   * @return measured value. 
074   */
075  public MeasuredValue doRetrieve() {
076    getLogger().log(Level.INFO, "GET /wattdepot/{" + groupId + "}/depository/{" + depositoryId
077        + "}/value/?sensor={" + sensorId + "}&start={" + start + "}&end={" + end + "}&timestamp={"
078        + timestamp + "}&gap={" + gapSeconds + "}");
079    try {
080      DepositoryImpl deposit = (DepositoryImpl) depot.getWattDeposiory(depositoryId, groupId);
081      if (deposit != null) {
082        Sensor sensor = depot.getSensor(sensorId, groupId);
083        if (sensor != null) {
084          Double value = null;
085          Date startDate = null;
086          Date endDate = null;
087          Date time = null;
088          if (start != null && end != null) {
089            startDate = DateConvert.parseCalStringToDate(start);
090            endDate = DateConvert.parseCalStringToDate(end);
091            if (gapSeconds != null) {
092              value = deposit.getValue(sensor, startDate, endDate, Long.parseLong(gapSeconds));
093            }
094            else {
095              value = deposit.getValue(sensor, startDate, endDate);
096            }
097          }
098          else if (timestamp != null) {
099            time = DateConvert.parseCalStringToDate(timestamp);
100            if (gapSeconds != null) {
101              value = deposit.getValue(sensor, time, Long.parseLong(gapSeconds));
102            }
103            else {
104              value = deposit.getValue(sensor, time);
105            }
106          }
107          MeasuredValue val = new MeasuredValue(sensorId, value, deposit.getMeasurementType());
108          if (end != null) {
109            val.setDate(endDate);
110          }
111          else if (time != null) {
112            val.setDate(time);
113          }
114          return val;
115        }
116        else {
117          setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Could not find sensor " + sensorId);
118        }
119      }
120      else {
121        setStatus(Status.CLIENT_ERROR_BAD_REQUEST, "Could not find depository " + depositoryId);
122      }
123    }
124    catch (MissMatchedOwnerException e) {
125      setStatus(Status.CLIENT_ERROR_CONFLICT, e.getMessage());
126    }
127    catch (NoMeasurementException e1) {
128      setStatus(Status.CLIENT_ERROR_EXPECTATION_FAILED, e1.getMessage());
129    }
130    catch (NumberFormatException e) {
131      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
132    }
133    catch (MeasurementGapException e) {
134      setStatus(Status.CLIENT_ERROR_EXPECTATION_FAILED, e.getMessage());
135    }
136    catch (ParseException e) {
137      setStatus(Status.CLIENT_ERROR_BAD_REQUEST, e.getMessage());
138      e.printStackTrace();
139    }
140    catch (DatatypeConfigurationException e) {
141      setStatus(Status.SERVER_ERROR_INTERNAL, e.getMessage());
142    }
143    return null;
144  }
145}
