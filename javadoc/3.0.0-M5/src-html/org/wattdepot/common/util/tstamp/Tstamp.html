030  /**
031   * Returns true if the passed string can be parsed into an XMLGregorianCalendar object.
032   * 
033   * @param lexicalRepresentation The string representation.
034   * @return True if the string is a legal XMLGregorianCalendar.
035   */
036  public static boolean isTimestamp(String lexicalRepresentation) {
037    try {
038      DatatypeFactory factory = DatatypeFactory.newInstance();
039      factory.newXMLGregorianCalendar(lexicalRepresentation);
040      return true;
041
042    }
043    catch (Exception e) {
044      return false;
045    }
046  }

048  /**
049   * Returns an XMLGregorianCalendar, given its string representation. Missing hours, minutes,
050   * second, millisecond, and timezone fields are given defaults.
051   * 
052   * @param rep The string representation.
053   * @return The timestamp.
054   * @throws Exception If the string cannot be parsed into a timestamp.
055   */
056  public static XMLGregorianCalendar makeTimestamp(String rep) throws Exception {
057    DatatypeFactory factory = DatatypeFactory.newInstance();
058    long mills = factory.newXMLGregorianCalendar(rep).toGregorianCalendar().getTimeInMillis();
059    return makeTimestamp(mills);
060  }

063   * Converts a javax.sql.Timestamp into a javax.xml.datatype.XMLGregorianCalendar.
064   * 
065   * @param tstamp The javax.sql.Timestamp

068  public static XMLGregorianCalendar makeTimestamp(java.sql.Timestamp tstamp) {
069    DatatypeFactory factory = null;
070    try {
071      factory = DatatypeFactory.newInstance();

074      return factory.newXMLGregorianCalendar(calendar);
075    }

077      throw new RuntimeException(factoryErrorMsg, e);
078    }

118  /**
119   * Returns a new XMLGregorianCalendar corresponding to the passed tstamp incremented by the number
120   * of days.
121   * 
122   * @param tstamp The base date and time.
123   * @param days The number of days to increment. This can be a negative number.
124   * @return A new XMLGregorianCalendar instance representing the inc'd time.
125   */
126  public static XMLGregorianCalendar incrementDays(XMLGregorianCalendar tstamp, int days) {
127    DatatypeFactory factory = null;
128    try {
129      factory = DatatypeFactory.newInstance();
130      GregorianCalendar calendar = new GregorianCalendar();
131      long millis = tstamp.toGregorianCalendar().getTimeInMillis();
132      millis += 1000L * 60 * 60 * 24 * days;
133      calendar.setTimeInMillis(millis);
134      return factory.newXMLGregorianCalendar(calendar);
135    }
136    catch (DatatypeConfigurationException e) {
137      throw new RuntimeException(factoryErrorMsg, e);
138    }
139  }

259  /**
260   * Returns true if tstamp is equal to or between start and end.
261   * 
262   * @param start The start time.
263   * @param tstamp The timestamp to test.
264   * @param end The end time.
265   * @return True if tstamp is between start and end.
266   */
267  public static boolean inBetween(XMLGregorianCalendar start, XMLGregorianCalendar tstamp,
268      XMLGregorianCalendar end) {
269    long startMillis = start.toGregorianCalendar().getTimeInMillis();
270    long endMillis = end.toGregorianCalendar().getTimeInMillis();
271    long tstampMillis = tstamp.toGregorianCalendar().getTimeInMillis();
272    return ((tstampMillis >= startMillis) && (tstampMillis <= endMillis));
273  }

288  /**
289   * Returns the number of days between time1 and time2. Returns a negative number if day1 is after
290   * day2. Takes into account daylight savings time issues.
291   * 
292   * @param day1 The first day.
293   * @param day2 The second day.
294   * @return The number of days between the two days.
295   */
296  public static int daysBetween(XMLGregorianCalendar day1, XMLGregorianCalendar day2) {
297    return (int) (getUnixDay(day2) - getUnixDay(day1));
298  }

300  /**
301   * Returns a long representing the number of days since the Unix epoch to the passed day.
302   * 
303   * @param day The day of interest.
304   * @return The number of days since the epoch.
305   */
306  private static long getUnixDay(XMLGregorianCalendar day) {
307    GregorianCalendar greg = day.toGregorianCalendar();
308    long offset = greg.get(Calendar.ZONE_OFFSET) + greg.get(Calendar.DST_OFFSET);
309    long daysSinceEpoch =
310        (long) Math.floor((double) (greg.getTime().getTime() + offset)
311            / ((double) MILLISECS_PER_DAY));
312    return daysSinceEpoch;
313  }

374  /**
375   * Returns true if the passed timestamp indicates some time today or some time in the future.
376   * 
377   * @param timestamp The timestamp of interest.
378   * @return True if it's today or some day in the future.
379   */
380  public static boolean isTodayOrLater(XMLGregorianCalendar timestamp) {
381    XMLGregorianCalendar today = Tstamp.makeTimestamp();
382    boolean isToday =
383        (today.getYear() == timestamp.getYear()) && (today.getMonth() == timestamp.getMonth())
384            && (today.getDay() == timestamp.getDay());
385    boolean afterToday =
386        today.toGregorianCalendar().getTimeInMillis() < timestamp.toGregorianCalendar()
387            .getTimeInMillis();
388    return (isToday || afterToday);
389  }

391  /**
392   * Returns true if the passed timestamp indicates some time yesterday or some time in the future.
393   * This is useful for the Commit/Churn DPDs, where the sensor typically sends data not from the
394   * current day, but from the day before.
395   * 
396   * @param timestamp The timestamp of interest.
397   * @return True if it's today or some day in the future.
398   */
399  public static boolean isYesterdayOrLater(XMLGregorianCalendar timestamp) {
400    XMLGregorianCalendar yesterday = Tstamp.incrementDays(Tstamp.makeTimestamp(), -1);
401
402    boolean isYesterday =
403        (yesterday.getYear() == timestamp.getYear())
404            && (yesterday.getMonth() == timestamp.getMonth())
405            && (yesterday.getDay() == timestamp.getDay());
406    boolean afterYesterday =
407        yesterday.toGregorianCalendar().getTimeInMillis() < timestamp.toGregorianCalendar()
408            .getTimeInMillis();
409    return (isYesterday || afterYesterday);
410  }

424  /**
425   * Helper function that prepares a List timestamps, between the start time and end time, at the
426   * given sampling interval.
427   * 
428   * @param startTime The start of the range requested.
429   * @param endTime The start of the range requested.
430   * @param intervalMinutes The sampling interval requested in minutes.
431   * @return The List of XMLGregorianCalendars.
432   */
433  public static List<XMLGregorianCalendar> getTimestampList(XMLGregorianCalendar startTime,
434      XMLGregorianCalendar endTime, int intervalMinutes) {
435    long intervalMilliseconds;
436    long rangeLength = Tstamp.diff(startTime, endTime);
437    long minutesToMilliseconds = 60L * 1000L;
438
439    if (intervalMinutes < 0) {
440      return null;
441    }
442    if (rangeLength <= 0) {
443      // either startTime == endTime, or startTime > endTime
444      return null;
445    }
446    else if (intervalMinutes == 0) {
447      // use default interval
448      intervalMilliseconds = rangeLength / 10;
449    }
450    else if ((intervalMinutes * minutesToMilliseconds) > rangeLength) {
451      // TODO BOGUS, should throw an exception so callers can distinguish between problems
452      return null;
453    }
454    else {
455      // got a good interval
456      intervalMilliseconds = intervalMinutes * minutesToMilliseconds;
457    }
458    // DEBUG
459    // System.out.format("%nstartTime=%s, endTime=%s, interval=%d min%n", startTime, endTime,
460    // intervalMilliseconds / minutesToMilliseconds);
461
462    // Build list of timestamps, starting with startTime, separated by intervalMilliseconds
463    List<XMLGregorianCalendar> timestampList = new ArrayList<XMLGregorianCalendar>();
464    XMLGregorianCalendar timestamp = startTime;
465    while (Tstamp.lessThan(timestamp, endTime)) {
466      timestampList.add(timestamp);
467      // System.out.format("timestamp=%s%n", timestamp); // DEBUG
468      timestamp = Tstamp.incrementMilliseconds(timestamp, intervalMilliseconds);
469    }
470    // add endTime to cover the last runt interval which is <= intervalMilliseconds
471    timestampList.add(endTime);
472    // System.out.format("timestamp=%s%n", endTime); // DEBUG
473    return timestampList;
474  }
