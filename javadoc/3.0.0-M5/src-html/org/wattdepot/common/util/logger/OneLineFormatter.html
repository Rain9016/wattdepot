001package org.wattdepot.common.util.logger;
002
003import java.text.SimpleDateFormat;
004import java.util.Date;
005import java.util.Locale;
006import java.util.logging.Formatter;
007import java.util.logging.LogRecord;
008
009/**
010 * Provides a one line formatter for use with WattDepot logging. Supports optional date stamp prefix
011 * and optional appending of a newline. Portions of this code are adapted from
012 * http://hackystat-utilities.googlecode.com/
013 * 
014 * @author Philip Johnson
015 * @author Robert Brewer
016 */
017public class OneLineFormatter extends Formatter {
018
019  /** Whether or not to include the date stamp in the format string. */
020  private boolean enableDateStamp = true;
021
022  /** Whether or not to add a newline. */
023  private boolean enableNewline = true;
024
025  /**
026   * Default constructor that enables the date stamp and new line.
027   */
028  public OneLineFormatter() {
029    this(true, true);
030  }
031
032  /**
033   * One line format string with optional date stamp. Always adds a newline.
034   * 
035   * @param enableDateStamp If true, a date stamp is inserted.
036   */
037  public OneLineFormatter(boolean enableDateStamp) {
038    this(enableDateStamp, true);
039  }
040
041  /**
042   * One line format string with optional date stamp and optional newline.
043   * 
044   * @param enableDateStamp If true, a date stamp is inserted.
045   * @param enableNewline If true, a newline is always inserted.
046   */
047  public OneLineFormatter(boolean enableDateStamp, boolean enableNewline) {
048    this.enableDateStamp = enableDateStamp;
049    this.enableNewline = enableNewline;
050  }
051
052  /**
053   * Formats the passed log string as a single line. Prefixes the log string with a date stamp if
054   * enabled, and adds a newline if enabled.
055   * 
056   * @param record A log record.
057   * @return The message string.
058   */
059  @Override
060  public String format(LogRecord record) {
061    StringBuffer buff = new StringBuffer();
062    if (this.enableDateStamp) {
063      SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy/MM/dd HH:mm:ss", Locale.US);
064      buff.append(dateFormat.format(new Date()));
065      buff.append("  ");
066    }
067    buff.append(record.getMessage());
068    if (this.enableNewline) {
069      buff.append(System.getProperty("line.separator"));
070    }
071    return buff.toString();
072  }
073}
