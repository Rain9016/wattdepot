045/**
046 * AdministratorServerResource - Administrative interface for WattDepot. It
047 * handles the HTTP API ("/wattdepot/{group_id}/").
048 * 
049 * @author Cam Moore
050 * 
051 */
052public class AdminServerResource extends WattDepotServerResource {
053
054  /**
055   * @return The admin user interface as an HTML Representation.
056   */
057  @Get()
058  public Representation toHtml() {
059    getLogger().log(Level.INFO, "GET /wattdepot/{" + groupId + "}/");
060    if (!isInRole(groupId) && !isInRole("admin")) {
061      User user = getClientInfo().getUser();
062      UserInfo info = depot.getUser(user.getIdentifier());
063      UserGroup group = depot.getUsersGroup(info);
064      if (group != null) {
065        redirectPermanent("/wattdepot/" + group.getId() + "/");
066      }
067      else {
068        setStatus(Status.CLIENT_ERROR_FORBIDDEN);
069      }
070    }
071    Map<String, Object> dataModel = new HashMap<String, Object>();
072    // get some stuff from the database
073    List<UserInfo> users = depot.getUsers();
074    List<UserGroup> groups = depot.getUserGroups();
075    List<Depository> depos = depot.getWattDepositories(groupId);
076    List<SensorLocation> locs = depot.getLocations(groupId);
077    List<Sensor> sensors = depot.getSensors(groupId);
078    List<SensorModel> sensorModels = depot.getSensorModels();
079    List<SensorGroup> sensorGroups = depot.getSensorGroups(groupId);
080    List<CollectorMetaData> sensorProcesses = depot.getCollectorMetaDatas(groupId);
081    List<MeasurementType> measurementTypes = depot.getMeasurementTypes();
082    dataModel.put("users", users);
083    dataModel.put("groups", groups);
084    dataModel.put("groupId", groupId);
085    dataModel.put("depositories", depos);
086    dataModel.put("locations", locs);
087    dataModel.put("sensors", sensors);
088    dataModel.put("sensorgroups", sensorGroups);
089    dataModel.put("sensormodels", sensorModels);
090    dataModel.put("sensorprocesses", sensorProcesses);
091    dataModel.put("measurementtypes", measurementTypes);
092    dataModel.put("opens", ((WattDepotImpl) depot).getSessionOpen());
093    dataModel.put("closes", ((WattDepotImpl) depot).getSessionClose());
094    Representation rep = new ClientResource(LocalReference.createClapReference(getClass()
095        .getPackage()) + "/Admin.ftl").get();
096
097    return new TemplateRepresentation(rep, dataModel, MediaType.TEXT_HTML);
098  }
099}
