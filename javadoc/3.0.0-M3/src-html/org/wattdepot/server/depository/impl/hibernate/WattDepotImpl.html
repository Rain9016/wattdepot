060  /**
061   * Creates a new WattDepotImpl instance with the given ServerProperties.
062   * 
063   * @param properties
064   *          The ServerProperties.
065   */
066  public WattDepotImpl(ServerProperties properties) {
067    super();
068    setServerProperties(properties);
069    UserPassword adminPassword = getUserPassword(UserInfo.ADMIN.getId());
070    if (getSessionClose() != getSessionOpen()) {
071      throw new RuntimeException("opens and closed mismatched.");
072    }
073    if (adminPassword == null) {
074      try {
075        defineUserPassword(UserPassword.ADMIN.getId(), UserPassword.ADMIN.getPlainText());
076        if (getSessionClose() != getSessionOpen()) {
077          throw new RuntimeException("opens and closed mismatched.");
078        }
079      }
080      catch (UniqueIdException e1) {
081        // what do we do here?
082        e1.printStackTrace();
083      }
084    }
085    else {
086      updateUserPassword(adminPassword);
087      if (getSessionClose() != getSessionOpen()) {
088        throw new RuntimeException("opens and closed mismatched.");
089      }
090    }
091    UserGroup admin = getUserGroup(UserGroup.ADMIN_GROUP.getId());
092    if (getSessionClose() != getSessionOpen()) {
093      throw new RuntimeException("opens and closed mismatched.");
094    }
095    if (admin == null) {
096      try {
097        defineUserGroup(UserGroup.ADMIN_GROUP.getId(), UserGroup.ADMIN_GROUP.getUsers());
098        if (getSessionClose() != getSessionOpen()) {
099          throw new RuntimeException("opens and closed mismatched.");
100        }
101      }
102      catch (UniqueIdException e) {
103        // what do we do here?
104        e.printStackTrace();
105      }
106    }
107    else {
108      updateUserGroup(admin);
109      if (getSessionClose() != getSessionOpen()) {
110        throw new RuntimeException("opens and closed mismatched.");
111      }
112    }
113    UserInfo adminUser = getUser(UserInfo.ADMIN.getId());
114    if (getSessionClose() != getSessionOpen()) {
115      throw new RuntimeException("opens and closed mismatched.");
116    }
117    if (adminUser == null) {
118      try {
119        defineUserInfo(UserInfo.ADMIN.getId(), UserInfo.ADMIN.getFirstName(),
120            UserInfo.ADMIN.getLastName(), UserInfo.ADMIN.getEmail(), UserInfo.ADMIN.getAdmin(),
121            UserInfo.ADMIN.getProperties());
122        if (getSessionClose() != getSessionOpen()) {
123          throw new RuntimeException("opens and closed mismatched.");
124        }
125      }
126      catch (UniqueIdException e) {
127        // what do we do here?
128        e.printStackTrace();
129      }
130    }
131    else {
132      updateUserInfo(UserInfo.ADMIN);
133      if (getSessionClose() != getSessionOpen()) {
134        throw new RuntimeException("opens and closed mismatched.");
135      }
136    }
137  }

146  @Override
147  public SensorLocation defineLocation(String id, Double latitude, Double longitude,
148      Double altitude, String description, UserGroup owner) throws UniqueIdException {
149    SensorLocation l = null;
150    try {
151      l = getLocation(id, UserGroup.ADMIN_GROUP_NAME);
152    }
153    catch (MissMatchedOwnerException e) {
154      // can't happen
155      e.printStackTrace();
156    }
157    if (l != null) {
158      throw new UniqueIdException(id + " is already a Location id.");
159    }
160    Session session = Manager.getFactory(getServerProperties()).openSession();
161    sessionOpen++;
162    session.beginTransaction();
163    l = new SensorLocation(id, latitude, longitude, altitude, description, owner);
164    session.save(l);
165    session.getTransaction().commit();
166    session.close();
167    sessionClose++;
168    return l;
169  }

207      UserGroup owner) throws UniqueIdException, MissMatchedOwnerException {
208    if (!owner.equals(l.getOwner())) {
209      throw new MissMatchedOwnerException(owner.getId() + " does not match location's.");
210    }
211    Sensor s = null;
212    s = getSensor(id, UserGroup.ADMIN_GROUP_NAME);
213    if (s != null) {
214      throw new UniqueIdException(id + " is already a Sensor id.");
215    }
216    Session session = Manager.getFactory(getServerProperties()).openSession();
217    sessionOpen++;
218    session.beginTransaction();
219    s = new Sensor(id, uri, l, sm, owner);
220    saveSensor(session, s);
221    session.getTransaction().commit();
222    session.close();
223    sessionClose++;
224    return s;
225  }

228   * (non-Javadoc)
229   * 
230   * @see org.wattdepot.server.WattDepot#defineSensorGroup(java.lang.String,
231   * java.util.List, org.wattdepot.datamodel.UserGroup)
232   */
233  @Override
234  public SensorGroup defineSensorGroup(String id, Set<Sensor> sensors, UserGroup owner)
235      throws UniqueIdException, MissMatchedOwnerException {
236    for (Sensor s : sensors) {
237      if (!owner.equals(s.getOwner())) {
238        throw new MissMatchedOwnerException(owner.getId() + " is not the owner of all the sensors.");
239      }
240    }
241    SensorGroup sg = null;
242    sg = getSensorGroup(id, UserGroup.ADMIN_GROUP_NAME);
243    if (sg != null) {
244      throw new UniqueIdException(id + " is already a SensorGroup id.");
245    }
246    Session session = Manager.getFactory(getServerProperties()).openSession();
247    sessionOpen++;
248    session.beginTransaction();
249    sg = new SensorGroup(id, sensors, owner);
250    for (Sensor s : sensors) {
251      saveSensor(session, s);
252    }
253    session.save(sg);
254    session.getTransaction().commit();
255    session.close();
256    sessionClose++;
257    return sg;
258  }

324  @Override
325  public UserGroup defineUserGroup(String id, Set<UserInfo> users) throws UniqueIdException {
326    UserGroup g = getUserGroup(id);
327    if (g != null) {
328      throw new UniqueIdException(id + " is already a UserGroup id.");
329    }
330    Session session = Manager.getFactory(getServerProperties()).openSession();
331    sessionOpen++;
332    session.beginTransaction();
333    for (UserInfo u : users) {
334      for (Property p : u.getProperties()) {
335        session.saveOrUpdate(p);
336      }
337      session.saveOrUpdate(u);
338    }
339    g = new UserGroup(id, users);
340    session.saveOrUpdate(g);
341    session.getTransaction().commit();
342    session.close();
343    sessionClose++;
344    return g;
345  }

623  /*
624   * (non-Javadoc)
625   * 
626   * @see org.wattdepot.server.WattDepot#deleteUserGroup(java.lang.String)
627   */
628  @Override
629  public void deleteUserGroup(String id) throws IdNotFoundException {
630    UserGroup g = getUserGroup(id);
631    if (g == null) {
632      throw new IdNotFoundException(id + " is not a defined user group id.");
633    }
634    Session session = Manager.getFactory(getServerProperties()).openSession();
635    sessionOpen++;
636    session.beginTransaction();
637    for (CollectorMetaData sp : getCollectorMetaDatas(session, id)) {
638      session.delete(sp);
639    }
640    session.getTransaction().commit();
641    session.close();
642    sessionClose++;
643    session = Manager.getFactory(getServerProperties()).openSession();
644    sessionOpen++;
645    session.beginTransaction();
646    for (SensorGroup sg : getSensorGroups(session, id)) {
647      session.delete(sg);
648    }
649    session.getTransaction().commit();
650    session.close();
651    sessionClose++;
652    session = Manager.getFactory(getServerProperties()).openSession();
653    sessionOpen++;
654    session.beginTransaction();
655    for (Depository d : getWattDepositories(session, id)) {
656      for (Sensor s : d.listSensors(session)) {
657        for (Measurement m : d.getMeasurements(session, s)) {
658          session.delete(m);
659        }
660      }
661    }
662    session.getTransaction().commit();
663    session.close();
664    sessionClose++;
665    session = Manager.getFactory(getServerProperties()).openSession();
666    sessionOpen++;
667    session.beginTransaction();
668    List<Depository> depositories = getWattDepositories(session, id);
669    session.getTransaction().commit();
670    session.close();
671    sessionClose++;
672    session = Manager.getFactory(getServerProperties()).openSession();
673    sessionOpen++;
674    session.beginTransaction();
675    for (Depository d : depositories) {
676      session.delete(d);
677    }
678    for (Sensor s : getSensors(session, id)) {
679      session.delete(s);
680    }
681    session.getTransaction().commit();
682    session.close();
683    sessionClose++;
684    session = Manager.getFactory(getServerProperties()).openSession();
685    sessionOpen++;
686    session.beginTransaction();
687    for (SensorModel sm : getSensorModels(session)) {
688      if (!SensorModelHelper.models.containsKey(sm.getName())) {
689        session.delete(sm);
690      }
691    }
692    session.getTransaction().commit();
693    session.close();
694    sessionClose++;
695    session = Manager.getFactory(getServerProperties()).openSession();
696    sessionOpen++;
697    session.beginTransaction();
698    for (SensorLocation l : getLocations(session, id)) {
699      session.delete(l);
700    }
701    session.getTransaction().commit();
702    session.close();
703    sessionClose++;
704    session = Manager.getFactory(getServerProperties()).openSession();
705    sessionOpen++;
706    session.beginTransaction();
707    session.delete(g);
708    session.getTransaction().commit();
709    session.close();
710    sessionClose++;
711  }
