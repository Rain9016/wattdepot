019package org.wattdepot.common.httpapi;
020
021import org.restlet.resource.Delete;
022import org.restlet.resource.Get;
023import org.restlet.resource.Put;
024import org.wattdepot.common.domainmodel.UserPassword;
025
026/**
027 * UserResource - HTTP Interface for data model User.
028 * 
029 * @author Cam Moore
030 * 
031 */
032public interface UserPasswordResource {
033
034  /**
035   * Defines GET /wattdepot/userpassword/{user_id} API call.
036   * 
037   * @return The User with the given id. The id is sent in the request.
038   */
039  @Get("json") // Use JSON as transport encoding.
040  public UserPassword retrieve();
041
042  /**
043   * Defines the PUT /wattdepot/userpassword/ API call.
044   * 
045   * @param user
046   *          The User to store.
047   */
048  @Put()
049  public void store(UserPassword user);
050
051  /**
052   * Defined the DEL /wattdepot/userpassword/{user_id} API call. The id is sent in the
053   * request.
054   */
055  @Delete
056  public void remove();
057
058}
