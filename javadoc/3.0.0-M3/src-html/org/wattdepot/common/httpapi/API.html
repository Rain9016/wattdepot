019package org.wattdepot.common.httpapi;
020
021/**
022 * API - Defines the Strings that make up the HTTP API.
023 *
024 * @author Cam Moore
025 *
026 */
027public class API {
028
029  /** The base URI used for all WattDepot URIs. */
030  public static final String BASE_URI = "wattdepot/";
031  
032  /** The URI to get to the administration UI. */
033  public static final String ADMIN_URI = "admin/";
034  
035  /** The URI to get all the Depositories. */
036  public static final String DEPOSITORIES_URI = "depositories/";
037  
038  /** The URI to manipulate a given Depository. */
039  public static final String DEPOSITORY_URI = "depository/";
040  
041  /** The URI to manipulate a given Location. */
042  public static final String LOCATION_URI = "location/";
043  
044  /** The URI to get all the Locations. */
045  public static final String LOCATIONS_URI = "locations/";
046  
047  /** The URI to manipulate a given Measurement. */
048  public static final String MEASUREMENT_URI = "measurement/";
049
050  /** The URI to get all the Measurements. */
051  public static final String MEASUREMENTS_URI = "measurements/";
052  
053  /** The URI to manipulate a given MeasurementType. */
054  public static final String MEASUREMENT_TYPE_URI = "measurementtype/";
055
056  /** The URI to get all the MeasurementTypes. */
057  public static final String MEASUREMENT_TYPES_URI = "measurementtypes/";
058  
059  /** The URI to manipulate a given Sensor. */
060  public static final String SENSOR_URI = "sensor/";
061
062  /** The URI to get all Sensors. */
063  public static final String SENSORS_URI = "sensors/";
064  
065  /** The URI to manipulate a given SensorGroup. */
066  public static final String SENSOR_GROUP_URI = "sensorgroup/";
067
068  /** The URI to get all SensorGroups. */
069  public static final String SENSOR_GROUPS_URI = "sensorgroups/";
070
071  /** The URI to manipulate a given SensorModel. */
072  public static final String SENSOR_MODEL_URI = "sensormodel/";
073
074  /** The URI to get all SensorModels. */
075  public static final String SENSOR_MODELS_URI = "sensormodels/";
076
077  /** The URI to manipulate a given SensorProcess. */
078  public static final String SENSOR_PROCESS_URI = "collectormetadata/";
079
080  /** The URI to get all SensorProcesses. */
081  public static final String SENSOR_PROCESSES_URI = "collectormetadatas/";
082  
083  /** The URI for getting a 'measured' value. */
084  public static final String VALUE_URI = "value/";
085  
086  /** The URI for manipulating a given UserInfo. */
087  public static final String USER_URI = "user/";
088
089  /** The URI for manipulating a given UserInfo. */
090  public static final String USERS_URI = "users/";
091
092  /** The URI for manipulating a given UserGroup. */
093  public static final String USER_GROUP_URI = "usergroup/";
094
095  /** The URI for manipulating a given UserGroup. */
096  public static final String USER_GROUPS_URI = "usergroups/";
097
098  /** The URI for manipulating a given UserGroup. */
099  public static final String USER_PASSWORD_URI = "userpassword/";
100
101}
