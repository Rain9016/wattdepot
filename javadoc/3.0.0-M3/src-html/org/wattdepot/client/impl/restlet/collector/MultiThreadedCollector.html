019package org.wattdepot.client.impl.restlet.collector;
020
021import java.util.Timer;
022import java.util.TimerTask;
023
024import org.apache.commons.validator.routines.UrlValidator;
025import org.wattdepot.client.impl.restlet.WattDepotClient;
026import org.wattdepot.common.domainmodel.CollectorMetaData;
027import org.wattdepot.common.domainmodel.Depository;
028import org.wattdepot.common.domainmodel.Sensor;
029import org.wattdepot.common.domainmodel.SensorModel;
030import org.wattdepot.common.exception.BadCredentialException;
031import org.wattdepot.common.exception.BadSensorUriException;
032import org.wattdepot.common.exception.IdNotFoundException;
033import org.wattdepot.common.util.SensorModelHelper;
034import org.wattdepot.common.util.Slug;
035import org.wattdepot.common.util.tstamp.Tstamp;
036
037/**
038 * MultiThreadedCollector - Abstract base class for all Multi-Threaded
039 * Collectors.
040 * 
041 * @author Cam Moore
042 * 
043 */
044public abstract class MultiThreadedCollector extends TimerTask {
045
046  /** Flag for debugging messages. */
047  protected boolean debug;
048  /** The metadata about the collector. */
049  protected CollectorMetaData metaData;
050
051  /** The client used to communicate with the WattDepot server. */
052  protected WattDepotClient client;
053  /** The Depository for storing measurements. */
054  protected Depository depository;
055
056  /**
057   * Initializes the MultiThreadedCollector.
058   * 
059   * @param serverUri
060   *          The URI for the WattDepot server.
061   * @param username
062   *          The name of a user defined in the WattDepot server.
063   * @param password
064   *          The password for the user.
065   * @param collectorId
066   *          The CollectorMetaDataId used to initialize this collector.
067   * @param debug
068   *          flag for debugging messages.
069   * @throws BadCredentialException
070   *           if the user or password don't match the credentials in WattDepot.
071   * @throws IdNotFoundException
072   *           if the processId is not defined.
073   * @throws BadSensorUriException
074   *           if the Sensor's URI isn't valid.
075   */
076  public MultiThreadedCollector(String serverUri, String username, String password,
077      String collectorId, boolean debug) throws BadCredentialException, IdNotFoundException,
078      BadSensorUriException {
079    this.client = new WattDepotClient(serverUri, username, password);
080    this.debug = debug;
081    this.metaData = client.getCollectorMetaData(collectorId);
082    this.depository = client.getDepository(metaData.getDepositoryId());
083    validate();
084  }
085
086  /**
087   * @param serverUri
088   *          The URI for the WattDepot server.
089   * @param username
090   *          The name of a user defined in the WattDepot server.
091   * @param password
092   *          The password for the user.
093   * @param sensor
094   *          The Sensor to poll.
095   * @param pollingInterval
096   *          The polling interval in seconds.
097   * @param depository
098   *          The Depository to store the measurements.
099   * @param debug
100   *          flag for debugging messages.
101   * @throws BadCredentialException
102   *           if the user or password don't match the credentials in WattDepot.
103   * @throws BadSensorUriException
104   *           if the Sensor's URI isn't valid.
105   */
106  public MultiThreadedCollector(String serverUri, String username, String password, Sensor sensor,
107      Long pollingInterval, Depository depository, boolean debug) throws BadCredentialException,
108      BadSensorUriException {
109    this.client = new WattDepotClient(serverUri, username, password);
110    this.debug = debug;
111    this.metaData = new CollectorMetaData(Slug.slugify(sensor.getId() + " " + pollingInterval + " "
112        + depository.getName()), sensor, pollingInterval, depository.getName(), null);
113    client.putCollectorMetaData(metaData);
114    this.depository = depository;
115    client.putDepository(depository);
116    validate();
117  }
118
119  /**
120   * @return true if everything is good to go.
121   */
122  public boolean isValid() {
123    if (this.client != null && this.metaData != null) {
124      return true;
125    }
126    return false;
127  }
128
129  /**
130   * @param serverUri
131   *          The URI for the WattDepot server.
132   * @param username
133   *          The name of a user defined in the WattDepot server.
134   * @param password
135   *          The password for the user.
136   * @param collectorId
137   *          The CollectorMetaDataId used to initialize this collector.
138   * @param debug
139   *          flag for debugging messages.
140   * @param debug
141   * @return true if sensor starts successfully.
142   * @throws InterruptedException
143   *           If sleep is interrupted for some reason.
144   * @throws BadCredentialException
145   *           if the username and password are invalid.
146   */
147  public static boolean start(String serverUri, String username, String password,
148      String collectorId, boolean debug) throws InterruptedException, BadCredentialException {
149
150    // Before starting any sensors, confirm that we can connect to the WattDepot
151    // server. We do
152    // this here because if the server and sensors are running on the same
153    // system, at boot time the
154    // sensor might start before the server, causing client calls to fail. If
155    // this happens, we
156    // want to catch it at the top level, where it will result in the sensor
157    // process terminating.
158    // If we wait to catch it at the per-sensor level, it might cause a sensor
159    // to abort for what
160    // might be a short-lived problem. The sensor process should be managed by
161    // some other process
162    // (such as launchd), so it is OK to terminate because it should get
163    // restarted if the server
164    // isn't up quite yet.
165    WattDepotClient staticClient = new WattDepotClient(serverUri, username, password);
166    if (!staticClient.isHealthy()) {
167      System.err.format("Could not connect to server %s. Aborting.%n", serverUri);
168      // Pause briefly to rate limit restarts if server doesn't come up for a
169      // long time
170      Thread.sleep(2000);
171      return false;
172    }
173    // Get the collector metadata
174    CollectorMetaData metaData = null;
175    try {
176      metaData = staticClient.getCollectorMetaData(collectorId);
177      staticClient.getDepository(metaData.getDepositoryId());
178    }
179    catch (IdNotFoundException e) {
180      System.err.println(e.getMessage());
181      return false;
182    }
183    // Get SensorModel to determine what type of collector to start.
184    SensorModel model = metaData.getSensor().getModel();
185    if (model.getName().equals(SensorModelHelper.EGAUGE) && model.getVersion().equals("1.0")) {
186      Timer t = new Timer();
187      try {
188        EGaugeCollector collector = new EGaugeCollector(serverUri, username, password, collectorId,
189            debug);
190        if (collector.isValid()) {
191          System.out.format("Started polling %s sensor at %s%n", metaData.getSensor().getName(),
192              Tstamp.makeTimestamp());
193          t.schedule(collector, 0, metaData.getPollingInterval() * 1000);
194        }
195        else {
196          System.err.format("Cannot poll %s sensor%n", metaData.getSensor().getName());
197          return false;
198        }
199      }
200      catch (BadSensorUriException e) {
201        // TODO Auto-generated catch block
202        e.printStackTrace();
203      }
204      catch (IdNotFoundException e) {
205        // TODO Auto-generated catch block
206        e.printStackTrace();
207      }
208    }
209    else if (model.getName().equals(SensorModelHelper.SHARK) && model.getVersion().equals("1.03")) {
210      Timer t = new Timer();
211      try {
212        SharkCollector collector = new SharkCollector(serverUri, username, password, collectorId,
213            debug);
214        if (collector.isValid()) {
215          System.out.format("Started polling %s sensor at %s%n", metaData.getSensor().getName(),
216              Tstamp.makeTimestamp());
217          t.schedule(collector, 0, metaData.getPollingInterval() * 1000);
218        }
219        else {
220          System.err.format("Cannot poll %s sensor%n", metaData.getSensor().getName());
221          return false;
222        }
223      }
224      catch (IdNotFoundException e) {
225        // TODO Auto-generated catch block
226        e.printStackTrace();
227      }
228      catch (BadSensorUriException e) {
229        // TODO Auto-generated catch block
230        e.printStackTrace();
231      }
232    }
233    return true;
234  }
235
236  /**
237   * @throws BadSensorUriException
238   *           if the Sensor's URI isn't valid.
239   */
240  private void validate() throws BadSensorUriException {
241    Sensor s = metaData.getSensor();
242    String[] schemes = { "http", "https" };
243    UrlValidator urlValidator = new UrlValidator(schemes);
244    if (!urlValidator.isValid(s.getUri())) {
245      throw new BadSensorUriException(s.getUri() + " is not a valid URI.");
246    }
247  }
